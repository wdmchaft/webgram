/* obtain a JavaScript reference to the main canvas element */
var canvasElement = document.getElementById('mainCanvas');

/* create a webgram canvas around the canvas element */
var canvas = new Webgram.Canvas(canvasElement.getContext('2d'));

/* create the webgram instance */
var webgram = new Webgram(canvasElement, canvas);

/* install the mouse and keyboard event handlers */
webgram.attachHandlers();

/* create a rectangle element class */
var RectangleElement = Webgram.Class('RectangleElement', Webgram.DrawingElements.RectangularElement, {
    draw: function () {
        this.drawRect(this.getDrawingRectangle());
        this.paint();
        this.drawText('Webgram is cool!', this.getDrawingRectangle());
    }
});

/* create a rectangle element object */
var rectangleElement = new RectangleElement('my_rectangle', -50, -50, 201, 101);

/* add the rectangle element object to the webgram */
webgram.addDrawingElement(rectangleElement);

/* create a stroke style */
var strokeStyle = new Webgram.Styles.StrokeStyle(5, 'round', 'round', 0, '#A52800');

/* obtain a fill style by modifying the default one */
var fillStyle = Webgram.Styles.getFillStyle('default').replace({'colors': 'rgba(166, 40, 0, 0.5)'});

/* obtain a text style by modifying the element's text style */
var textStyle = rectangleElement.getTextStyle().replace({'size': 32});

/* set the new styles to the rectangle element */
rectangleElement.setStrokeStyle(strokeStyle);
rectangleElement.setFillStyle(fillStyle);
rectangleElement.setTextStyle(textStyle);
