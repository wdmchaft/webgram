 24     /**
 25      * Calculates the <tt>cursor</tt> CSS property indicating the direction
 26      * the closest to the given <tt>angle</tt>.
 27      * @param {Number} angle the angle
 28      * @returns {String} the <tt>cursor</tt> CSS property 
 29      */
 30     getCursorByAngle: function (angle) {
 31         var cursors = [
 32             'e-resize',
 33             'se-resize',
 34             's-resize',
 35             'sw-resize',
 36             'w-resize',
 37             'nw-resize',
 38             'n-resize',
 39             'ne-resize'
 40         ];
 41     
 42         angle = Webgram.Utils.normalizeAngle(angle);
 43         
 44         var index = Math.round(4 / Math.PI * angle);
 45         
 46         if (index < 0) {
 47             index = 8 + index;
 48         }
 49         if (index > 7) {
 50             index = index - 8;
 51         }
 52         
 53         return cursors[index];
 54     },

 56     /**
 57      * Normalizes an angle by translating it into the range 0..2 PI.
 58      * @param {Number} angle the angle to normalize
 59      * @returns {Number} the normalized angle, between 0 and 2 PI
 60      */
 61     normalizeAngle: function (angle) {
 62         /* transform the angle to degrees and then back to radians */
 63         angle = Math.round(angle * 180 * 10 / Math.PI) / 10;
 64         
 65         while (angle < 0) {
 66             angle = 2 * 180 + angle;
 67         }
 68         
 69         angle = angle % 360;
 70         angle = angle * Math.PI / 180;
 71         
 72         return angle;
 73     },

 75     /**
 76      * Snaps the given angle to multiples of a <tt>snapAngle</tt>.
 77      * @param {Number} angle the angle to snap
 78      * @param {Number} snapAngle the angle whose multiples will be used to snap
 79      * @param {Number} threshold the maximal distance beyond which the angle will not be snapped
 80      * @returns {Number} the snapped angle, or <tt>null</tt> if no snapping was performed
 81      */
 82     getSnappedAngle: function (angle, snapAngle, threshold) {
 83         angle = Webgram.Utils.normalizeAngle(angle);
 84         
 85         /* transform everything to degrees */
 86         snapAngle = snapAngle * 180 / Math.PI;
 87         angle = angle * 180 / Math.PI;
 88         threshold = threshold * 180 / Math.PI;
 89         
 90         var div = Math.floor(angle / snapAngle);
 91         var mod = angle % snapAngle;
 92         
 93         if (mod < threshold / 2) { /* close to the smaller snapping angle */
 94             angle = snapAngle * div;
 95         }
 96         else if (snapAngle - mod < threshold / 2) { /* close to the higher snapping angle */
 97             angle = snapAngle * (div + 1);
 98         }
 99         else {
100             return null; /* no snapping has been performed */
101         }
102         
103         /* transform it back to radians */
104         angle = angle * Math.PI / 180;
105         
106         return Webgram.Utils.normalizeAngle(angle);
107     },

109     /**
110      * Determines whether a point is inside (over) a given segment or not.
111      * The segment is formed by two points and has a <tt>thickness</tt>.
112      * @param {Webgram.Geometry.Point} point point to test
113      * @param {Webgram.Geometry.Point} point1 the first point of the segment
114      * @param {Webgram.Geometry.Point} point2 the second point of the segment
115      * @param {Number} thickness the thickness of the segment
116      * @returns {Boolean} <tt>true</tt> if the point is inside the segment, <tt>false</tt> otherwise
117      */
118     pointInSegment: function (point, point1, point2, thickness) {
119         var angle = Math.atan2(point2.y - point1.y, point2.x - point1.x);
120         var centerX = (point1.x + point2.x) / 2;
121         var centerY = (point1.y + point2.y) / 2;
122         
123         angle = -angle; /* reverse rotation */
124         
125         var sine = Math.sin(angle);
126         var cosine = Math.cos(angle);
127         
128         var rotatedX = centerX + (point.x - centerX) * cosine - (point.y - centerY) * sine;
129         var rotatedY = centerY + (point.y - centerY) * cosine + (point.x - centerX) * sine;
130     
131         var rotatedX1 = centerX + (point1.x - centerX) * cosine - (point1.y - centerY) * sine;
132         var rotatedY1 = centerY + (point1.y - centerY) * cosine + (point1.x - centerX) * sine;
133     
134         var rotatedX2 = centerX + (point2.x - centerX) * cosine - (point2.y - centerY) * sine;
135         
136         return (rotatedX >= Math.min(rotatedX1, rotatedX2)) && (rotatedX <= Math.max(rotatedX1, rotatedX2)) && 
137             (rotatedY >= rotatedY1 - thickness) && (rotatedY <= rotatedY1 + thickness);
138     },

140     /**
141      * Determines whether a point is inside (over) a bezier curve or not.
142      * The bezier curve is given by two end points, two control points and
143      * a certain <tt>thickness</tt>.
144      * @param {Webgram.Geometry.Point} point to test
145      * @param {Webgram.Geometry.Point} point1 the first end point of the bezier curve
146      * @param {Webgram.Geometry.Point} point2 the second end point of the bezier curve
147      * @param {Webgram.Geometry.Point} controlPoint1 the first control point of the bezier curve
148      * @param {Webgram.Geometry.Point} controlPoint2 the second control point of the bezier curve
149      * @param {Number} thickness the thickness of the bezier curve
150      * @returns {Boolean} <tt>true</tt> if the point is inside the bezier curve, <tt>false</tt> otherwise
151      */
152     pointInBezier: function (point, point1, point2, controlPoint1, controlPoint2, thickness) {
153         /* what we actually do here is to follow a bezier path,
154          * with a very coarse resolution; for every point so found,
155          * we check if the given point is within a small radial area */
156         
157         var distance = point1.getDistanceTo(point2);
158         var step, radius, x, y;
159         if (controlPoint2 != null) { /* cubic */
160             distance += (point1.getDistanceTo(controlPoint1) + 
161                     controlPoint1.getDistanceTo(controlPoint2) +
162                     controlPoint2.getDistanceTo(point2)) / 6;
163 
164             step = 5 / distance;
165             radius = thickness * 2;
166             
167             for (var t = 0; t < 1; t += step) {
168                 x = 
169                     Math.pow(1 - t, 3) * point1.x +
170                     3 * Math.pow(1 - t, 2) * t * controlPoint1.x + 
171                     3 * (1 - t) * t * t * controlPoint2.x + 
172                     Math.pow(t, 3) * point2.x;
173                 
174                 y = 
175                     Math.pow(1 - t, 3) * point1.y +
176                     3 * Math.pow(1 - t, 2) * t * controlPoint1.y + 
177                     3 * (1 - t) * t * t * controlPoint2.y + 
178                     Math.pow(t, 3) * point2.y;
179                 
180                 if (point.getDistanceTo(new Webgram.Geometry.Point(x, y)) < radius) {
181                     return true;
182                 }
183             }
184             
185             return false;
186         }
187         else { /* quadratic */
188             distance += (point1.getDistanceTo(controlPoint1) + 
189                     controlPoint1.getDistanceTo(point2)) / 4;
190             
191             step = 5 / distance;
192             radius = thickness * 2;
193             
194             for (var t = 0; t < 1; t += step) {
195                 x = 
196                     Math.pow(1 - t, 2) * point1.x +
197                     2 * (1 - t) * t * controlPoint1.x + 
198                     Math.pow(t, 2) * point2.x;
199                 
200                 y = 
201                     Math.pow(1 - t, 2) * point1.y +
202                     2 * (1 - t) * t * controlPoint1.y + 
203                     Math.pow(t, 2) * point2.y;
204                 
205                 if (point.getDistanceTo(new Webgram.Geometry.Point(x, y)) < radius) {
206                     return true;
207                 }
208             }
209             
210             return false;
211         }
212     },

214     /**
215      * Determines whether an angle is a multiple of a value.
216      * The comparison is made using a small threshold.
217      * @param {Number} angle the angle to test
218      * @param {Number} to the value whose multiples are to be considered
219      * @returns {Boolean} <tt>true</tt> if the angle is a multiple of the given angle, <tt>false</tt> otherwise
220      */
221     angleMultipleOf: function (angle, to) {
222         var value = angle / to;
223         return Math.abs(value - Math.round(value)) < 0.0001;
224     },
225     
226     /**
227      * Determines the position (index) of an object in an array.
228      * @param {Array} array the array to search
229      * @param {Object} value the object to search for
230      * @returns {Number} the position of the object inside the array, or <tt>-1</tt> if not found
231      */
232     indexOf: function (array, value) {
233         for (var i = 0; i < array.length; i++) {
234             if (array[i] === value) {
235                 return i;
236             }
237         }
238         
239         return -1;
240     },
241     
242     /**
243      * Tells if a string is entirely made of whitespace characters.
244      * @param {String} s the string to test
245      */
246     isSpace: function (s) {
247         for (var i = 0; i < s.length; i++) {
248             if (s[i] != ' ' &&
249                 s[i] != '\f' &&
250                 s[i] != '\n' &&
251                 s[i] != '\r' &&
252                 s[i] != '\t' &&
253                 s[i] != '\v') {
254                 
255                 return false;
256             }
257         }
258             
259         return true;
260     }
