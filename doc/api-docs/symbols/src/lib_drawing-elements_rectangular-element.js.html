 24 /**
 25  * @class A base drawing element class for all rectangular drawing elements.
 26  * A rectangular element has two main points: <em>top-left</em> and <em>bottom-right</em>.
 27  * @extends Webgram.DrawingElement
 28  * @param {String} id a given identifier, can be <tt>null</tt> 
 29  * @param {Number} x the coordinate of the left side
 30  * @param {Number} y the coordinate of the top side
 31  * @param {Number} width the width of the rectangular element
 32  * @param {Number} height the height of the rectangular element
 33  */
 34 Webgram.DrawingElements.RectangularElement = function (id, x, y, width, height) {
 35     this._minSize = new Webgram.Geometry.Size(1, 1); /* the default minimum size */
 36     
 37     if (x == null) { /* default attributes */
 38         x = 0;
 39         y = 0;
 40         width = this._minSize.width;
 41         height = this._minSize.height;
 42     }
 43     
 44     var points = [
 45         new Webgram.Geometry.Point(x, y),
 46         new Webgram.Geometry.Point(x + width - 1, y + height - 1)
 47     ];
 48     
 49     Webgram.DrawingElement.call(this, id, points);
 50     
 51     this._aspectRatio = null;
 52     this._keepSquareEnabled = false;
 53     
 54     this.setResizeEnabled(true);
 55     
 56     /**
 57      * An event that is triggered when the rectangular element is resized.<br>
 58      * Handlers receive no arguments. 
 59      * @type Webgram.Event
 60      */
 61     this.onResize = new Webgram.Event('resize', this); /* () */
 62     
 63     this.onResize.bind(function () {
 64         if (this.rotationCenterControlPoint) {
 65             this.rotationCenterControlPoint.reset();
 66         }
 67     });
 68 };

211     restoreParentRelativePosition: function () {
212         if (!this.isResizeEnabled()) {
213             return;
214         }
215         
216         var parentWidth = this._parent.getWidth();
217         var parentHeight = this._parent.getHeight();
218         
219         /* compute the two temporary new corners */
220         var topLeft = new Webgram.Geometry.Point(
221                 this._parentRelativePositions[Webgram.DrawingElements.RectangularElement.TOP_LEFT].x * (parentWidth - 1),
222                 this._parentRelativePositions[Webgram.DrawingElements.RectangularElement.TOP_LEFT].y * (parentHeight - 1)
223         );
224         var bottomRight = new Webgram.Geometry.Point(
225                 this._parentRelativePositions[Webgram.DrawingElements.RectangularElement.BOTTOM_RIGHT].x * (parentWidth - 1),
226                 this._parentRelativePositions[Webgram.DrawingElements.RectangularElement.BOTTOM_RIGHT].y * (parentHeight - 1)
227         );
228         
229         /* compute the new center */
230         var newCenter = topLeft.getCenterTo(bottomRight);
231         
232         /* rotate them inversely around the new center */
233         topLeft = topLeft.getRotated(-this.getRotationAngle(), newCenter);
234         bottomRight = bottomRight.getRotated(-this.getRotationAngle(), newCenter);
235         
236         var width = bottomRight.x - topLeft.x + 1;
237         var height = bottomRight.y - topLeft.y + 1;
238         
239         var constrainedSize = this._applySizeConstraints(new Webgram.Geometry.Size(width, height));
240         width = constrainedSize.width;
241         height = constrainedSize.height;
242         
243         /* honor the aspect ratio constraint, if any */
244         if (this.isKeepSquareEnabled()) {
245             if (height < width) {
246                 width = height;
247             }
248             else {
249                 height = width;
250             }
251         }
252         else if (this.getAspectRatio() != null) {
253             if (this.getAspectRatio() * height < width) {
254                 width = height * this.getAspectRatio();
255             }
256             else {
257                 height = width / this.getAspectRatio();
258             }
259         }
260         
261         /* update the shape points */
262         var left = newCenter.x - width / 2;
263         var top = newCenter.y - height / 2;
264         var right = newCenter.x + width / 2;
265         var bottom = newCenter.y + height / 2;
266         
267         this.setShapePoint(Webgram.DrawingElements.RectangularElement.TOP_LEFT, new Webgram.Geometry.Point(left, top));
268         this.setShapePoint(Webgram.DrawingElements.RectangularElement.BOTTOM_RIGHT, new Webgram.Geometry.Point(right, bottom));

464     _setTop: function (point, noSnapFeedback, noFlip) {
465         if (!noFlip) {
466             if (this.isFlippedVertically()) {
467                 return this._setBottom(point, noSnapFeedback, true);
468             }
469         }
470         
471         var shape = this._getFourPointsShape();
472         var rotatedShape = this.rotateDirect(shape);
473         var rotationAngle = this.getRotationAngle();
474         var oldCenter = this.getShape().getCenter();
475         
476         var nrBottomLeft = shape.points[Webgram.DrawingElements.RectangularElement.BOTTOM_LEFT];
477         var nrBottomRight = shape.points[Webgram.DrawingElements.RectangularElement.BOTTOM_RIGHT];
478         var nrBottom = nrBottomLeft.getCenterTo(nrBottomRight);
479         var nrTop = point.getRotated(-rotationAngle, oldCenter);
480         
481         /* snap the real shape point and apply
482          * the result to the fake point */
483         var nrTopLeft = shape.points[Webgram.DrawingElements.RectangularElement.TOP_LEFT];
484         nrTopLeft.y = nrTop.y;
485         
486         /* only snap when angle is multiple of PI/2 */
487         var snapDetails = null;
488         if (Webgram.Utils.angleMultipleOf(rotationAngle, Math.PI / 2)) { 
489             snapDetails = this._snapShapePoint(
490                     Webgram.DrawingElements.RectangularElement.TOP_LEFT, nrTopLeft, [nrBottomRight], rotationAngle, oldCenter);
491             nrTopLeft = snapDetails.point;
492             
493             if (Webgram.Utils.angleMultipleOf(rotationAngle, Math.PI)) { /* even multiple of PI */
494                 if (snapDetails.ySnapped && !noSnapFeedback) {
495                     snapDetails.snapVisualFeedback.x = null;
496                     this.setSnapVisualFeedback(snapDetails.snapVisualFeedback);
497                 }
498             }
499             else { /* odd multiple of PI */
500                 if (snapDetails.xSnapped && !noSnapFeedback) {
501                     snapDetails.snapVisualFeedback.y = null;
502                     this.setSnapVisualFeedback(snapDetails.snapVisualFeedback);
503                 }
504             }
505         }
506         
507         nrTop.y = nrTopLeft.y;
508         
509         /* apply other constraints */
510         var width = nrBottomRight.x - nrBottomLeft.x + 1;
511         var constrained = this._applyTopConstraints(nrTop, nrBottom, width);
512         nrTop = constrained.nrTop;
513         width = constrained.width;
514         
515         var top = nrTop.getRotated(rotationAngle, oldCenter);
516         var bottomLeft = rotatedShape.points[Webgram.DrawingElements.RectangularElement.BOTTOM_LEFT];
517         var bottomRight = rotatedShape.points[Webgram.DrawingElements.RectangularElement.BOTTOM_RIGHT];
518         var bottom = bottomLeft.getCenterTo(bottomRight);
519         var newCenter = top.getCenterTo(bottom);
520         
521         nrTop = top.getRotated(-this.getRotationAngle(), newCenter);
522         nrBottom = bottom.getRotated(-this.getRotationAngle(), newCenter);
523         
524         var x1 = newCenter.x - width / 2 + 0.5;
525         var y1 = nrTop.y;
526         var x2 = newCenter.x + width / 2 - 0.5;
527         var y2 = nrBottom.y;
528         
529         this.setShapePoint(Webgram.DrawingElements.RectangularElement.TOP_LEFT, new Webgram.Geometry.Point(x1, y1));
530         this.setShapePoint(Webgram.DrawingElements.RectangularElement.BOTTOM_RIGHT, new Webgram.Geometry.Point(x2, y2));
531         
532         this.onResize.trigger();
533     },

537             if (this.isFlippedHorizontally()) {
538                 return this._setLeft(point, noSnapFeedback, true);
539             }
540         }
541         
542         var shape = this._getFourPointsShape();
543         var rotatedShape = this.rotateDirect(shape);
544         var rotationAngle = this.getRotationAngle();
545         var oldCenter = this.getShape().getCenter();
546         
547         var nrTopLeft = shape.points[Webgram.DrawingElements.RectangularElement.TOP_LEFT];
548         var nrBottomLeft = shape.points[Webgram.DrawingElements.RectangularElement.BOTTOM_LEFT];
549         var nrLeft = nrTopLeft.getCenterTo(nrBottomLeft);
550         var nrRight = point.getRotated(-rotationAngle, oldCenter);
551         
552         /* snap the real shape point and apply
553          * the result to the fake point */
554         var nrBottomRight = shape.points[Webgram.DrawingElements.RectangularElement.BOTTOM_RIGHT];
555         nrBottomRight.x = nrRight.x;
556         
557         /* only snap when angle is multiple of PI/2 */
558         var snapDetails = null;
559         if (Webgram.Utils.angleMultipleOf(rotationAngle, Math.PI / 2)) { 
560             snapDetails = this._snapShapePoint(
561                     Webgram.DrawingElements.RectangularElement.BOTTOM_RIGHT, nrBottomRight, [nrTopLeft], rotationAngle, oldCenter);
562             nrBottomRight = snapDetails.point;
563         
564             if (Webgram.Utils.angleMultipleOf(rotationAngle, Math.PI)) { /* even multiple of PI */
565                 if (snapDetails.xSnapped && !noSnapFeedback) {
566                     snapDetails.snapVisualFeedback.y = null;
567                     this.setSnapVisualFeedback(snapDetails.snapVisualFeedback);
568                 }
569             }
570             else { /* odd multiple of PI */
571                 if (snapDetails.ySnapped && !noSnapFeedback) {
572                     snapDetails.snapVisualFeedback.x = null;
573                     this.setSnapVisualFeedback(snapDetails.snapVisualFeedback);

588         var bottomLeft = rotatedShape.points[Webgram.DrawingElements.RectangularElement.BOTTOM_LEFT];
589         var left = topLeft.getCenterTo(bottomLeft);
590         var newCenter = left.getCenterTo(right);
591         
592         nrLeft = left.getRotated(-this.getRotationAngle(), newCenter);
593         nrRight = right.getRotated(-this.getRotationAngle(), newCenter);
594 
595         var x1 = nrLeft.x;
596         var y1 = newCenter.y - height / 2 + 0.5;
597         var x2 = nrRight.x;
598         var y2 = newCenter.y + height / 2 - 0.5;
599         
600         this.setShapePoint(Webgram.DrawingElements.RectangularElement.TOP_LEFT, new Webgram.Geometry.Point(x1, y1));
601         this.setShapePoint(Webgram.DrawingElements.RectangularElement.BOTTOM_RIGHT, new Webgram.Geometry.Point(x2, y2));
602         
603         this.onResize.trigger();
604     },

1168     /**
1169      * Applies the size constraints (minimum size, aspect ratio, keep square) to the given size.
1170      * @param {Webgram.Geometry.Size} size the size to be evaluated and constrained
1171      * @param {Boolean} keepWidth true to adjust the height based on width,
1172      *  false to adjust the width based on height,
1173      *  null to choose the best way (minimal size)
1174      * @returns {Webgram.Geometry.Size} the constrained size
1175      */
1176     _applySizeConstraints: function (size, keepWidth) {
1177         // FIXME this could be improved to not call getMinSize & getAspectRatio two times
1178         
1179         if (keepWidth == null) {
1180             var sizeKeepWidth = this._applySizeConstraints(size, true);
1181             var sizeKeepHeight = this._applySizeConstraints(size, false);
1182             
1183             /* prefer the variant that yields the minimum size */
1184             if (sizeKeepHeight.width + sizeKeepHeight.height > sizeKeepWidth.width + sizeKeepWidth.height) {
1185                 return sizeKeepWidth;
1186             }
1187             else {
1188                 return sizeKeepHeight;
1189             }
1190         }
1191         
1192         var minSize = this.getMinSize();
1193         var aspectRatio = this.getAspectRatio();
1194         var width = size.width;
1195         var height = size.height;
1196 
1197         if (this.isKeepSquareEnabled()) {
1198             aspectRatio = 1;
1199         }
1200         
1201         if (aspectRatio != null) {
1202             if (keepWidth) {
1203                 height = width / aspectRatio;
1204             }
1205             else {
1206                 width = height * aspectRatio;
1207             }
1208             
1209             if (width < minSize.width) {
1210                 width = minSize.width;
1211                 height = width / aspectRatio;
1212             }
1213             if (height < minSize.height) {
1214                 height = minSize.height;
1215                 width = height * aspectRatio;
1216             }         
1217         }
1218         else {
1219             if (width < minSize.width) {
1220                 width = minSize.width;
1221             }
1222             if (height < minSize.height) {
1223                 height = minSize.height;
1224             }            
1225         }
1226         
1227         return new Webgram.Geometry.Size(width, height);
1228     },
