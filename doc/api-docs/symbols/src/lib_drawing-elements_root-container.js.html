 23 /**
 24  * @class The parent container of all the drawing elements of a webgram instance.
 25  * @extends Webgram.DrawingElements.ContainerElement
 26  * @param {Webgram} webgram the webgram instance
 27  * @param {Webgram.Canvas} canvas the canvas to be used with this root container
 28  */
 29 Webgram.DrawingElements.RootContainer = function (webgram, canvas) {
 30     Webgram.DrawingElements.ContainerElement.call(this, ':special:root-container',
 31         -canvas.getWidth() / 2, -canvas.getHeight() / 2,
 32         canvas.getWidth(), canvas.getHeight());
 33     
 34     /**
 35      * The webgram instance associated with this root container.
 36      * @type Webgram
 37      */
 38     this.webgram = webgram;
 39     
 40     /**
 41      * The canvas used by this root container to draw.
 42      * @type Webgram.Canvas
 43      */
 44     this.canvas = canvas;
 45     
 46     this._controlPoints = []; /* remove all control points inherited from a normal block element */    
 47     this._parent = null; /* no parent */
 48     this._fillStyle = Webgram.Styles.getFillStyle('background');
 49 
 50     /* drawing controls */
 51     this._activeDrawingControl = null;
 52     this._prevDrawingControl = null;
 53     this._lastMousePoint = null;
 54     
 55     /* zooming & panning */
 56     this._noZoom = false;
 57     this._zoomLevel = this.getSetting('zoomLevel');
 58     this._zoomFactor = this.getSetting('zoomFactors')[this._zoomLevel];
 59     this._offsetX = -this.canvas.getWidth() / 2;
 60     this._offsetY = -this.canvas.getHeight() / 2;
 61     
 62     /* snap visual feedback */
 63     this._snapVisualFeedbackList = null;
 64     this._snapVisualFeedbackTimer = null;
 65     
 66     /* benchmarking */
 67     this._drawingStartTime = 0;
 68     this._drawingEndTime = 0;
 69 
 70     this.setMoveEnabled(false);
 71     this.setRotateEnabled(false);
 72     this.setResizeEnabled(false);
 73     this.setSelectEnabled(false);    
 74 };

 94     /**
 95      * Draws the grid of the root container.
 96      * This is internally called by {@link Webgram.DrawingElements.RootContainer#draw}.<br><br>
 97      * <em>(could be overridden)</em>
 98      */
 99     drawGrid: function () {
100         var visibleWidth = this.canvas.getWidth() / this._zoomFactor;
101         var visibleHeight = this.canvas.getHeight() / this._zoomFactor;
102         var lineWidth = 1 / this._zoomFactor;
103         var lineNo, gmin, gmax, x, y;
104         
105         var snapGrid = this.getSetting('snapGrid');
106         var mainGrid = this.getSetting('mainGrid');
107         var mainGridStrokeStyle = Webgram.Styles.getStrokeStyle('main-grid').replace({lineWidth: lineWidth});
108         var axisStrokeStyle = Webgram.Styles.getStrokeStyle('main-grid-axes').replace({lineWidth: lineWidth});
109         
110         var snapGridStrokeStyle = null;
111         if (snapGrid != null && snapGrid.sizeX != null) {
112             snapGridStrokeStyle = Webgram.Styles.getStrokeStyle('snap-grid').replace({lineWidth: lineWidth});
113         }
114         
115         /* _mini property is true while the root container element
116          * is being drawn on the mini webgram */
117         if (!this._mini) {
118             /* draw the snap grid only when zoomed and enabled */
119             if (this._zoomFactor > 1 && snapGridStrokeStyle != null) {
120                 /* draw the vertical snap grid */
121                 gmin = Math.floor(this._offsetX / snapGrid.sizeX) + 1;
122                 gmax = Math.floor((this._offsetX + visibleWidth) / snapGrid.sizeX);
123                 for (lineNo = gmin; lineNo <= gmax; lineNo++) {
124                     x = snapGrid.sizeX * lineNo;
125                     this.drawLine(new Webgram.Geometry.Point(x, this._offsetY), new Webgram.Geometry.Point(x, visibleHeight + this._offsetY - 1));
126                     this.paint(snapGridStrokeStyle, null);
127                 }
128                 
129                 /* draw the horizontal snap grid */
130                 gmin = Math.floor(this._offsetY / snapGrid.sizeY) + 1;
131                 gmax = Math.floor((this._offsetY + visibleHeight) / snapGrid.sizeY);
132                 for (lineNo = gmin; lineNo <= gmax; lineNo++) {
133                     y = snapGrid.sizeY * lineNo;
134                     this.drawLine(new Webgram.Geometry.Point(this._offsetX, y), new Webgram.Geometry.Point(visibleWidth + this._offsetX - 1, y));
135                     this.paint(snapGridStrokeStyle, null);
136                 }
137             }
138         
139             var skipFactor;
140             if (this._zoomFactor < 1) {
141                 skipFactor = Math.round(1 / this._zoomFactor);
142             }
143             else {
144                 skipFactor = 1;
145             }
146             
147             /* draw the vertical main grid */
148             gmin = Math.floor(this._offsetX / mainGrid.sizeX) + 1;
149             gmax = Math.floor((this._offsetX + visibleWidth) / mainGrid.sizeX);
150             for (lineNo = gmin; lineNo <= gmax; lineNo++) {
151                 if (lineNo % skipFactor) {
152                     continue;
153                 }
154                 x = mainGrid.sizeX * lineNo;
155                 if (x === 0) {
156                     continue;
157                 }
158                 
159                 this.drawLine(new Webgram.Geometry.Point(x, this._offsetY), new Webgram.Geometry.Point(x, visibleHeight + this._offsetY - 1));
160                 this.paint(mainGridStrokeStyle, null);
161             }
162             
163             /* draw the horizontal main grid */
164             gmin = Math.floor(this._offsetY / mainGrid.sizeY) + 1;
165             gmax = Math.floor((this._offsetY + visibleHeight) / mainGrid.sizeY);
166             for (lineNo = gmin; lineNo <= gmax; lineNo++) {
167                 if (lineNo % skipFactor) {
168                     continue;
169                 }
170                 y = mainGrid.sizeY * lineNo;
171                 if (y === 0) {
172                     continue;
173                 }
174                 
175                 this.drawLine(new Webgram.Geometry.Point(this._offsetX, y), new Webgram.Geometry.Point(visibleWidth + this._offsetX - 1, y));
176                 this.paint(mainGridStrokeStyle, null);
177             }
178         }
179         
180         /* draw the vertical axis */
181         this.drawLine(new Webgram.Geometry.Point(0, this._offsetY), new Webgram.Geometry.Point(0, visibleHeight + this._offsetY - 1));
182         this.paint(axisStrokeStyle, null);
183         
184         /* draw the horizontal axis */
185         this.drawLine(new Webgram.Geometry.Point(this._offsetX, 0), new Webgram.Geometry.Point(visibleWidth + this._offsetX - 1, 0));
186         this.paint(axisStrokeStyle, null);
187     },

189     /**
190      * Draws the <em>snap visual feedback</em>, if present.
191      * This is internally called by {@link Webgram.DrawingElements.RootContainer#draw}.<br><br>
192      * <em>(could be overridden)</em>
193      * @see Webgram.DrawingElements.RootContainer#setSnapVisualFeedback
194      */
195     drawSnapVisualFeedback: function () {
196         if (this._mini) {
197             return;
198         }
199         
200         if (this._snapVisualFeedbackList == null || this._snapVisualFeedbackList.length === 0) {
201             return;
202         }
203         
204         if (!this.getSetting('snapVisualFeedback')) {
205             return;
206         }
207         
208         var strokeStyle = Webgram.Styles.getStrokeStyle('snap-visual-feedback');
209         var fillStyle = Webgram.Styles.createFillStyle({colors: strokeStyle.colors});
210         // var textStyle = Webgram.Styles.getTextStyle('snap-visual-feedback');
211         
212         var visibleWidth = this.canvas.getWidth() / this._zoomFactor;
213         var visibleHeight = this.canvas.getHeight() / this._zoomFactor;
214 
215         for (var i = 0; i < this._snapVisualFeedbackList.length; i++) {
216             var snapVisualFeedback = this._snapVisualFeedbackList[i];
217             
218             if (snapVisualFeedback.type === 'linear') {
219                 /* draw the vertical axis */
220                 if (snapVisualFeedback.x != null) {
221                     this.drawLine(
222                             new Webgram.Geometry.Point(snapVisualFeedback.x, this._offsetY),
223                             new Webgram.Geometry.Point(snapVisualFeedback.x, visibleHeight + this._offsetY - 1));
224                     this.paint(strokeStyle, null);
225                 }
226                 
227                 /* draw the horizontal axis */
228                 if (snapVisualFeedback.y != null) {
229                     this.drawLine(
230                             new Webgram.Geometry.Point(this._offsetX, snapVisualFeedback.y),
231                             new Webgram.Geometry.Point(visibleWidth + this._offsetX - 1, snapVisualFeedback.y));
232                     this.paint(strokeStyle, null);
233                 }
234             }
235             else { /* type == 'radial' */
236                 var center = new Webgram.Geometry.Point(snapVisualFeedback.x, snapVisualFeedback.y);
237                 var radius = 50;
238                 
239                 var angle = Webgram.Utils.normalizeAngle(snapVisualFeedback.angle);
240                 var angleDelta = 10 * Math.PI / radius / this._zoomFactor;
241                 
242                 /* draw the arc */
243                 this.drawArc(center, radius, radius, angle - angleDelta, angle + angleDelta, 0);
244                 this.paint(strokeStyle, null);
245     
246                 var arrowTop = center.getPointAt(radius, angle);
247                 var arrowLen = 10 / this._zoomFactor;
248                 var arrowAngle = Math.PI / 4;
249                 var arrowAngleOffset = 5 / radius / this._zoomFactor;
250                 
251                 /* draw the first arrow */
252                 this.drawLine(arrowTop, arrowTop.getPointAt(arrowLen, angle - Math.PI / 2 - arrowAngle / 2 - arrowAngleOffset));
253                 this.drawLine(arrowTop.getPointAt(arrowLen, angle - Math.PI / 2 + arrowAngle / 2 - arrowAngleOffset));
254                 this.drawLine(arrowTop);
255                 this.paint(null, fillStyle);
256     
257                 /* draw the second arrow */
258                 this.drawLine(arrowTop, arrowTop.getPointAt(arrowLen, angle + Math.PI / 2 - arrowAngle / 2 + arrowAngleOffset));
259                 this.drawLine(arrowTop.getPointAt(arrowLen, angle + Math.PI / 2 + arrowAngle / 2 + arrowAngleOffset));
260                 this.drawLine(arrowTop);
261                 this.paint(null, fillStyle);
262                 
263                 /* draw the angle axis */
264                 this.drawLine(center, center.getPointAt(radius * 100, angle));
265                 this.paint(strokeStyle, null);
266                 
267                 /* draw the angle text */
268 //                var textDist = 30 / this._zoomFactor;
269 //                var textPoint = arrowTop.getPointAt(textDist, 3 * Math.PI / 4 + angle);
270 //                var textBox = new Webgram.Geometry.Rectangle(textPoint.x, textPoint.y, textPoint.x, textPoint.y);
271 //                var textAngle = Math.round(angle * 180 / Math.PI * 2) / 2 + '';
272 //                this.drawText(textAngle + String.fromCharCode(0xB0), textBox, textStyle);
273             }
274         }

277     /**
278      * Draws the rulers of the root container.
279      * This is internally called by {@link Webgram.DrawingElements.RootContainer#draw}.<br><br>
280      * <em>(could be overridden)</em>
281      */
282     drawRulers: function () {
283         var fillStyle = Webgram.Styles.getFillStyle('rulers');
284         var strokeStyle = Webgram.Styles.getStrokeStyle('rulers');
285         var textStyle = Webgram.Styles.getTextStyle('rulers');
286         
287         var mainGrid = this.getSetting('mainGrid');
288         var snapGrid = this.getSetting('snapGrid');
289         var showRulers = this.getSetting('showRulers');
290         var showZoom = this.getSetting('showZoom');
291         
292         var barWidth, barHeight, gmin, gmax, lineNo, box;
293         
294         if (!this._mini) {
295             var visibleWidth = this.canvas.getWidth() / this._zoomFactor;
296             var visibleHeight = this.canvas.getHeight() / this._zoomFactor;
297             var zoomUnit = 1 / this._zoomFactor; /* used to adjust various coordinates */
298             
299             var gridSizeX = null, gridSizeY = null;
300             if (snapGrid != null && snapGrid.sizeX != null) {
301                 gridSizeX = snapGrid.sizeX;
302                 gridSizeY = snapGrid.sizeY;
303             }
304             else if (mainGrid != null && mainGrid.sizeX != null) {
305                 gridSizeX = Math.round(mainGrid.sizeX / 2);
306                 gridSizeY = Math.round(mainGrid.sizeY / 2);
307             }
308             
309             if (showRulers && gridSizeX != null) {
310                 barHeight = 15;
311                 barWidth = Math.max(('' + Math.round(this._offsetY)).length, 
312                         ('' + Math.round(this._offsetY + visibleHeight)).length) * 7;
313                 
314                 /* draw the ruler background */
315                 this.drawRect(new Webgram.Geometry.Rectangle(this._offsetX - zoomUnit, this._offsetY - zoomUnit,
316                         this._offsetX + visibleWidth, this._offsetY + barHeight / this._zoomFactor));
317                 this.drawRect(new Webgram.Geometry.Rectangle(this._offsetX - zoomUnit, this._offsetY - zoomUnit,
318                         this._offsetX + barWidth / this._zoomFactor, this._offsetY + visibleHeight));
319                 
320                 this.paint(null, fillStyle);
321                 
322                 /* draw the separator lines */
323                 this.drawLine(new Webgram.Geometry.Point(this._offsetX + barWidth / this._zoomFactor, this._offsetY + barHeight / this._zoomFactor),
324                         new Webgram.Geometry.Point(this._offsetX + visibleWidth, this._offsetY + barHeight / this._zoomFactor));
325                 this.paint(strokeStyle, null);
326                 
327                 this.drawLine(new Webgram.Geometry.Point(this._offsetX + barWidth / this._zoomFactor, this._offsetY + barHeight / this._zoomFactor),
328                         new Webgram.Geometry.Point(this._offsetX + barWidth / this._zoomFactor, this._offsetY + visibleHeight));
329                 this.paint(strokeStyle, null);
330                 
331                 var skipFactorX = Math.ceil(Math.round(10 / this._zoomFactor) / gridSizeX) * gridSizeX;
332                 
333                 /* draw the horizontal ruler */
334                 gmin = Math.floor(this._offsetX / gridSizeX) + Math.round(10 / this._zoomFactor);
335                 gmax = Math.floor((this._offsetX + visibleWidth) / gridSizeX) - 1;
336                 for (lineNo = gmin; lineNo <= gmax; lineNo++) {
337                     if (lineNo % skipFactorX) {
338                         continue;
339                     }
340                     var x = gridSizeX * lineNo;
341                     box = new Webgram.Geometry.Rectangle(x, this._offsetY, x, this._offsetY + barHeight / this._zoomFactor);
342                     
343                     this.drawText('' + x, box, textStyle);
344                 }
345                 
346                 var skipFactorY = Math.ceil(Math.round(10 / this._zoomFactor) / gridSizeY) * gridSizeY;
347                 
348                 /* draw the vertical ruler */
349                 gmin = Math.floor(this._offsetY / gridSizeY) + 2;
350                 gmax = Math.floor((this._offsetY + visibleHeight) / gridSizeY) - 1;
351                 for (lineNo = gmin; lineNo <= gmax; lineNo++) {
352                     if (lineNo % skipFactorY) {
353                         continue;
354                     }
355                     var y = gridSizeY * lineNo;
356                     box = new Webgram.Geometry.Rectangle(this._offsetX, y, this._offsetX + (barWidth - 3) / this._zoomFactor, y);
357                     
358                     this.drawText('' + y, box, textStyle.replace({justify: 'rc'}));
359                 }
360             }
361             
362             if (showZoom) {
363                 /* draw the zoom background */
364                 barWidth = 45;
365                 barHeight = 15;
366                 
367                 this.drawRect(new Webgram.Geometry.Rectangle(
368                         this._offsetX + visibleWidth - barWidth / this._zoomFactor, 
369                         this._offsetY + visibleHeight - barHeight / this._zoomFactor,
370                         this._offsetX + visibleWidth, this._offsetY + visibleHeight));
371                 
372                 this.paint(strokeStyle, fillStyle);
373                 
374                 /* draw the zoom text */
375                 var text;
376                 if (this._zoomFactor >= 2) {
377                     text = '' + this._zoomFactor + ' x';
378                 }
379                 else {
380                     text = '' + this._zoomFactor.toFixed(1) + ' x';
381                 }
382                 
383                 box = new Webgram.Geometry.Rectangle(
384                         this._offsetX + visibleWidth - barWidth / this._zoomFactor,
385                         this._offsetY + visibleHeight - barHeight / this._zoomFactor,
386                         this._offsetX + visibleWidth,
387                         this._offsetY + visibleHeight);
388                 
389                 this.drawText(text, box, textStyle.replace({justify: 'cc'}));
390             }
391         }

394     /**
395      * Displays the frames per second value (used for benchmarking).
396      * This is internally called by {@link Webgram.DrawingElements.RootContainer#draw}.<br><br>
397      * <em>(could be overridden)</em>
398      */
399     drawFPS: function () {
400         var fromLastDrawing = new Date().getTime() - this._drawingEndTime;
401         var fps = 1000 / fromLastDrawing;
402         
403         this.drawText('fps: ' + fps.toFixed(1), new Webgram.Geometry.Rectangle(-100, -100, 100, 100));
404     },
405     
406     draw: function () {
407         this._drawingStartTime = new Date().getTime();
408         
409         this.drawBackground();
410         if (this.webgram.settings.mainGrid) {
411             this.drawGrid();
412         }
413         
414         var drawingElement, i;
415         
416         /* elements themselves */
417         
418         /* the normal draw loop */
419         for (i = 0; i < this.drawingElements.length; i++) {
420             drawingElement = this.drawingElements[i];
421             
422             /* avoid drawing elements that don't have enough shape points */
423             if (drawingElement.getShape().points.length >= drawingElement.minPointCount) {
424                 drawingElement.draw();
425             
426                 if (!this._mini) {
427                     this._noZoom = true;
428                     drawingElement.drawNoZoom();
429                     this._noZoom = false;
430                 }
431             }
432         }
433         
434         /* the "on top" draw loop */
435         for (i = 0; i < this.drawingElements.length; i++) {
436             drawingElement = this.drawingElements[i];
437             
438             /* avoid drawing elements that don't have enough shape points */
439             if (drawingElement.getShape().points.length >= drawingElement.minPointCount) {
440                 drawingElement.drawTop();
441                 
442                 if (!this._mini) {
443                     this._noZoom = true;
444                     drawingElement.drawNoZoomTop();
445                     this._noZoom = false;
446                 }
447             }
448         }
449         
450         if (!this._mini) {
451             /* control points & sockets */
452             for (i = 0; i < this.drawingElements.length; i++) {
453                 drawingElement = this.drawingElements[i];
454                 this._noZoom = true;
455                 drawingElement.drawActionMenuItems();
456                 drawingElement.drawControlPoints();
457                 drawingElement.drawSockets();
458                 this._noZoom = false;
459             }
460         }
461         
462         this._noZoom = true;
463         this.drawSnapVisualFeedback();
464         this.drawRulers();
465         // this.drawFPS();
466         this._noZoom = false;
467         this._drawingEndTime = new Date().getTime();
468     },

473     drawLine: function (point1, point2) {
474         if (this._noZoom) {
475             point1 = this.transformZoomOffsetDirect(point1);
476             point2 = point2 && this.transformZoomOffsetDirect(point2);
477         }
478         
479         this.canvas.drawLine(point1, point2);
480     },
481     
482     drawPoly: function (poly, closed) {
483         if (this._noZoom) {
484             var newPoints = [];
485             for (var i = 0; i < poly.points.length; i++) {
486                 var point = poly.points[i];
487                 point = this.transformZoomOffsetDirect(point);
488                 newPoints.push(point);
489             }
490             
491             poly = new Webgram.Geometry.Poly(newPoints);
492         }
493         
494         this.canvas.drawPoly(poly, closed);
495     },
496     
497     drawRect: function (rectangle) {
498         var poly = rectangle.getPoly();
499         
500         this.drawPoly(poly, true);
501     },
502     
503     drawArc: function (center, radiusX, radiusY, startAngle, endAngle) {
504         if (this._noZoom) {
505             center = this.transformZoomOffsetDirect(center);
506         }
507         
508         this.canvas.drawArc(center, radiusX, radiusY, startAngle, endAngle);
509     },
510     
511     drawBezier: function (point1, point2, controlPoint1, controlPoint2) {
512         if (this._noZoom) {
513             point1 = this.transformZoomOffsetDirect(point1);
514             point2 = this.transformZoomOffsetDirect(point2);
515             if (controlPoint2 != null) {
516                 controlPoint1 = this.transformZoomOffsetDirect(controlPoint1);
517             }
518             controlPoint2 = this.transformZoomOffsetDirect(controlPoint2);
519         }
520         
521         this.canvas.drawBezier(point1, point2, controlPoint1, controlPoint2);
522     },
523     
524     drawImage: function (image, center, size, rotationAngle, alpha, transformSet) {
525         if (!transformSet) {
526             transformSet = new Webgram.Canvas.TransformSet();
527         }
528         
529         if (!this._noZoom) {
530             transformSet.addTranslation(-this._offsetX, -this._offsetY);
531             transformSet.addScaling(this._zoomFactor);
532         }
533         else {
534             center = this.transformZoomOffsetDirect(center);
535         }
536         
537         /* the method can be called with either an image dictionary as returned
538          * by the ImageStore or directly a javascript image object */
539         if (image.content) {
540             image = image.content;
541         }
542         
543         this.canvas.drawImage(image, center, size, rotationAngle, alpha, transformSet);
544     },
545     
546     drawText: function (text, box, textStyle, transformSet) {
547         if (box == null) {
548             box = this.getBaseRectangle();
549         }
550         
551         if (textStyle === undefined) {
552             textStyle = this.getTextStyle();
553         }
554         
555         if (!transformSet) {
556             transformSet = new Webgram.Canvas.TransformSet();
557         }
558         
559         if (!this._noZoom) {
560             transformSet.addTranslation(-this._offsetX, -this._offsetY);
561             transformSet.addScaling(this._zoomFactor);
562         }
563         else {
564             var topLeft = this.transformZoomOffsetDirect(box.getTopLeft());
565             var bottomRight = this.transformZoomOffsetDirect(box.getBottomRight());
566             
567             box = new Webgram.Geometry.Rectangle(topLeft.x, topLeft.y, bottomRight.x, bottomRight.y);
568         }
569         
570         this.canvas.drawText(text, box, textStyle, transformSet);
571     },
572 
573     paint: function (strokeStyle, fillStyle, transformSet) {
574         if (!transformSet) {
575             transformSet = new Webgram.Canvas.TransformSet();
576         }
577         
578         if (strokeStyle === undefined) {
579             strokeStyle = this.getStrokeStyle();
580         }
581         if (fillStyle === undefined) {
582             fillStyle = this.getFillStyle();
583         }
584         
585         if (!this._noZoom) {
586             transformSet.addTranslation(-this._offsetX, -this._offsetY);
587             transformSet.addScaling(this._zoomFactor);
588         }
589         else {
590             if (strokeStyle) {
591                 strokeStyle = strokeStyle.clone();
592             }
593             if (fillStyle) {
594                 fillStyle = fillStyle.clone();
595             }
596             
597             var styles = [strokeStyle, fillStyle];
598             for (var i = 0; i < styles.length; i++) {
599                 var style = styles[i];
600                 if (!style) {
601                     continue;
602                 }
603                 
604                 if (style.gradientPoint1) {
605                     style.gradientPoint1 = this.transformZoomOffsetDirect(style.gradientPoint1);
606                 }
607                 if (style.gradientPoint2) {
608                     style.gradientPoint2 = this.transformZoomOffsetDirect(style.gradientPoint2);
609                 }
610             }
611         }
612         
613         this.canvas.paint(strokeStyle, fillStyle, transformSet);
614     },

628     /**
629      * Returns the current zoom level of the root container.
630      * The zoom <em>level</em> is different from the <em>zoom</em> factor in that
631      * the zoom level is an integer number representing an index in a list
632      * of predefined zoom factors, while the zoom factor is effectively the
633      * ratio at which the scaling is being done.<br><br>
634      * <em>(should not be overridden)</em>
635      * @returns {Number} the current zoom level
636      */
637     getZoomLevel: function () {
638         return this._zoomLevel;
639     },

641     /**
642      * Sets the zoom level of the root container and optionally recenters. 
643      * The list of the available zoom levels is defined through the <tt>zoomLevels</tt> global setting.<br><br>
644      * <em>(should not be overridden)</em>
645      * @param {Number} zoomLevel the zoom level to set
646      * @param {Webgram.Geometry.Point} point the point of the webgram that should remain fixed
647      */
648     setZoomLevel: function (zoomLevel, point) {
649         this._zoomLevel = zoomLevel;
650         
651         var zoomFactors = this.getSetting('zoomFactors');
652         if (this._zoomLevel > zoomFactors.length - 1) {
653             this._zoomLevel = zoomFactors.length - 1;
654         }
655         
656         if (this._zoomLevel < 0) {
657             this._zoomLevel = 0;
658         }
659         
660         this.setZoomFactor(zoomFactors[this._zoomLevel], point);
661         
662         /* update all the DEs - the offsets of the various points might have changed with the zoom */
663         for (var i = 0; i < this.drawingElements.length; i++) {
664             this.drawingElements[i].invalidate();
665         }
666         
667         this.invalidateDrawing();
668     },
669     
670     /**
671      * Changes the zoom factor to the exact given value.<br><br>
672      * <em>(should not be overridden)</em>
673      * @param {Number} zoomFactor the zoom factor to set
674      * @param {Webgram.Geometry.Point} point the point in the webgram that should remain fixed
675      */
676     setZoomFactor: function (zoomFactor, point) {
677         var oldZoomFactor = this._zoomFactor;
678         this._zoomFactor = zoomFactor;
679         
680         /* recenter if a center point was provided */
681         if (point) {
682             var dx = point.x * (this._zoomFactor / oldZoomFactor - 1);
683             var dy = point.y * (this._zoomFactor / oldZoomFactor - 1);
684             
685             var offset = this.getOffset();
686             offset = offset.getTranslated(dx / this._zoomFactor, dy / this._zoomFactor);
687             this.setOffset(offset);
688         }
689         
690         this.invalidateDrawing(true);
691         this.webgram.onZoom.trigger(this._zoomLevel);
692     },

734         var offsetX = center.x - this.getWidth() / 2 / this._zoomFactor;

797     /**
798      * Transforms a point from the webgram coordinates system
799      * to the window coordinates system. The transformation takes into account
800      * the <em>zoom</em> and the <em>panning offset</em> of the root container.<br><br>
801      * <em>(should not be overridden)</em>
802      * @param {Webgram.Geometry.Point} point the point with webgram coordinates to transform
803      * @returns {Webgram.Geometry.Point} the corresponding point in window coordinates
804      */
805     transformZoomOffsetDirect: function (point) {
806         return new Webgram.Geometry.Point(
807                 (point.x - this._offsetX) * this._zoomFactor,
808                 (point.y - this._offsetY) * this._zoomFactor);
809     },

811     /**
812      * Transforms a point from the window coordinates system
813      * to the webgram coordinates system. The transformation takes into account
814      * the <em>zoom</em> and the <em>panning offset</em> of the root container.<br><br>
815      * <em>(should not be overridden)</em>
816      * @param {Webgram.Geometry.Point} point a point with window coordinates to transform
817      * @returns {Webgram.Geometry.Point} the corresponding point in webgram coordinates
818      */
819     transformZoomOffsetInverse: function (point) {
820         return new Webgram.Geometry.Point(
821                 point.x / this._zoomFactor + this._offsetX,
822                 point.y / this._zoomFactor + this._offsetY);

829         if (this._snapVisualFeedbackTimer != null) {
830             clearTimeout(this._snapVisualFeedbackTimer);
831             this._snapVisualFeedbackTimer = null;
832         }
833         
834         if (snapVisualFeedback == null || snapVisualFeedback.length === 0) {
835             this._snapVisualFeedbackList = null;
836             this.invalidateDrawing();
837             
838             return;
839         }
840         
841         if (snapVisualFeedback instanceof Array) {
842             this._snapVisualFeedbackList = snapVisualFeedback;
843         }
844         else {
845             this._snapVisualFeedbackList = [snapVisualFeedback];
846         }
847         
848         this.invalidateDrawing();
849         
850         var that = this;
851         
852         this._snapVisualFeedbackTimer = setTimeout(function () {
853             that._snapVisualFeedbackList = null;
854             that.invalidateDrawing();
855         }, 200);
856     },

1068                 drawingControl.handleDrawingElementRemove(drawingElement);
1069                 var afterSelectedDrawingElements = drawingControl.getSelectedDrawingElements();
1070                 
1071                 if (!Webgram.Utils.equals(beforeSelectedDrawingElements, afterSelectedDrawingElements) && this.webgram != null) {
1072                     this.webgram.onSelectionChange.trigger(afterSelectedDrawingElements);
1073                 }
1074             }
1075         }
