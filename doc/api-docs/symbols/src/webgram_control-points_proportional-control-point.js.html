 21 /**
 22  * @namespace Contains some common control point classes used by webgram.
 23  * @see Webgram.ControlPoint
 24  */
 25 Webgram.ControlPoints = Webgram.Namespace('Webgram.ControlPoints');
 26 
 27 /**
 28  * @class A control point that controls a <em>proportional point</em>.
 29  * A proportional point is a point that preserves its relative position
 30  * when the drawing element changes size. Proportional points are added
 31  * with {@link Webgram.DrawingElement#addProportionalPoint}.
 32  * @extends Webgram.ControlPoint
 33  * @param {Webgram.Geometry.Point} point the proportional point to be controlled
 34  */
 35 Webgram.ControlPoints.ProportionalControlPoint = function (point) {
 36     /**
 37      * The proportional point controlled by this control point.
 38      * @type Webgram.Geometry.Point
 39      */
 40     this.proportionalPoint = point;
 41 
 42     Webgram.ControlPoint.call(this);
 43 };
 44 
 45 Webgram.ControlPoints.ProportionalControlPoint.prototype = {
 46     getCursor: function () {
 47         return 'pointer';
 48     },
 49     
 50     computeAnchor: function () {
 51         return this.drawingElement.transformDirect(this.proportionalPoint);
 52     },
 53     
 54     processMove: function (point, vanillaPoint) {
 55         point = this.drawingElement.transformInverse(point);
 56         
 57         this.proportionalPoint.x = point.x;
 58         this.proportionalPoint.y = point.y;
 59         
 60         this.drawingElement.saveProportionalPoint(this.proportionalPoint);
 61         
 62         if (this.proportionalPoint._onChange != null) {
 63             this.proportionalPoint._onChange.call(this.drawingElement, this.proportionalPoint);
 64         }
 65     }
 66 };
 67 
 68 Webgram.Class('Webgram.ControlPoints.ProportionalControlPoint', Webgram.ControlPoint);
