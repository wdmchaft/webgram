 21 /**
 22  * @class The abstract base class for all control points.
 23  * A control point is a zoom-independent element on the canvas
 24  * that allows the user to easily interact with a drawing element.
 25  * Each control point instance belongs to a drawing element.
 26  */
 27 Webgram.ControlPoint = function () {
 28     /**
 29      * A control point has a radius, used to determine if the mouse is inside
 30      * inside it. This radius does not change with the zoom level.
 31      * @type Number
 32      */
 33     this.radius = 9;
 34     
 35     /**
 36      * This is a reference to the element that the control point belongs to.
 37      * Its value is <tt>null</tt> until the control point is added to the drawing element
 38      * using {@link Webgram.DrawingElement#addControlPoint}.
 39      * @type Webgram.DrawingElement
 40      */
 41     this.drawingElement = null;
 42 
 43     this._anchor = null;
 44     this._focusType = Webgram.ControlPoint.FOCUS_NONE;
 45     this._strokeStyle = Webgram.Styles.getStrokeStyle('default');
 46     this._fillStyle = Webgram.Styles.getFillStyle('default');
 47     this._snapToGridEnabled = false;
 48 
 49 
 50     /* events */
 51 
 52     /**
 53      * An event that is triggered when the control point moving begins.<br>
 54      * Handlers receive no arguments. 
 55      * @type Webgram.Event
 56      */
 57     this.onBeginMove = new Webgram.Event('begin move', this); /* () */
 58     
 59     /**
 60      * An event that is triggered when the control point moving ends.<br>
 61      * Handlers receive no arguments. 
 62      * @type Webgram.Event
 63      */
 64     this.onEndMove = new Webgram.Event('end move', this); /* () */
 65 
 66     /**
 67      * An event that is triggered whenever the control point is moved.<br>
 68      * Handlers receive the following arguments: <tt>(point)</tt> 
 69      * @type Webgram.Event
 70      */
 71     this.onMove = new Webgram.Event('move', this); /* (point) */
 72 
 73     /**
 74      * An event that is triggered when the control point is updated.<br>
 75      * Handlers receive no arguments.
 76      * @see Webgram.ControlPoint#update 
 77      * @type Webgram.Event
 78      */
 79     this.onUpdate = new Webgram.Event('update', this); /* () */
 80     
 81     /**
 82      * An event that is triggered whenever the mouse enters the area of this control point.
 83      * {@link Webgram.ControlPoint#pointInside} method is used to evaluate this condition.<br>
 84      * This is the equivalent of <tt>onMouseOver</tt> JavaScript event.<br>
 85      * Handlers receive the following arguments: <tt>(point, modifiers)</tt>. 
 86      * @type Webgram.Event
 87      */
 88     this.onMouseEnter = new Webgram.Event('mouse enter', this); /* (point, modifiers) */
 89 
 90     /**
 91      * An event that is triggered whenever the mouse leaves the area of this control point.
 92      * {@link Webgram.ControlPoint#pointInside} method is used to evaluate this condition.<br>
 93      * This is the equivalent of <tt>onMouseOut</tt> JavaScript event.<br>
 94      * Handlers receive the following arguments: <tt>(modifiers)</tt>. 
 95      * @type Webgram.Event
 96      */
 97     this.onMouseLeave = new Webgram.Event('mouse leave', this); /* (modifiers) */
 98     
 99     /**
100      * An event that is triggered when the control point is added to a drawing element,
101      * using {@link Webgram.DrawingElement#addControlPoint}.<br>
102      * Handlers receive the following arguments: <tt>(drawingElement)</tt>.
103      * @type Webgram.Event
104      */
105     this.onAdd = new Webgram.Event('add', this); /* (drawingElement) */
106 
107     /**
108      * An event that is triggered when the control point is removed from a drawing element,
109      * using {@link Webgram.DrawingElement.remControlPoint}.<br>
110      * Handlers receive the following arguments: <tt>(drawingElement)</tt>. 
111      * @type Webgram.Event
112      */
113     this.onRemove = new Webgram.Event('remove', this); /* (drawingElement) */
114 
115     /**
116      * An event that is triggered when the control point is selected by the active {@link Webgram.DrawingControl}.<br>
117      * Handlers receive no arguments. 
118      * @type Webgram.Event
119      */
120     this.onSelect = new Webgram.Event('select', this); /* () */
121 
122     /**
123      * An event that is triggered when the control point is unselected by the active {@link Webgram.DrawingControl}.<br>
124      * Handlers receive no arguments. 
125      * @type Webgram.Event
126      */
127     this.onUnselect = new Webgram.Event('unselect', this); /* () */
128 
129     /**
130      * An event that is triggered when the control points's focus type is changed,
131      * by a call to {@link Webgram.ControlPoint#setFocusType}.<br>
132      * Handlers receive no arguments. 
133      * @type Webgram.Event
134      */
135     this.onChangeFocus = new Webgram.Event('change focus', this); /* () */
136 };

138 /** Represents the normal focus type of a control point, when it's neither hovered, nor selected. */
139 Webgram.ControlPoint.FOCUS_NONE = 0;
140 /** Represents the hovered focus type of a control point. */
141 Webgram.ControlPoint.FOCUS_HOVERED = 1;
142 /** Represents the selected focus type of a control point. */
143 Webgram.ControlPoint.FOCUS_SELECTED = 2;

191     
192     /**
193      * This method is responsible for the way the control point looks on the canvas.
194      * Concrete drawing elements should implement this method by calling the various
195      * available drawing primitives (draw* methods), followed by a call to <tt>paint</tt>.<br>
196      * Figures drawn by a control point are not affected by the zoom.<br><br> 
197      * <em>(should be overridden)</em>
198      */
199     draw: function () {
200     },
201     
202     /**
203      * Draws a straight line between <tt>point1</tt> and <tt>point2</tt>. All coordinates are relative to the control point.
204      * If the last drawn point differs from <tt>point1</tt>, a straight line is drawn to join them.<br>
205      * This is a path-related drawing primitive.
206      * Set <tt>point2</tt> to <tt>null</tt> or <tt>undefined</tt> to obtain a "lineTo" effect.
207      * @param {Webgram.Geometry.Point} point1 the starting point of the line
208      * @param {Webgram.Geometry.Point} point2 the ending point of the line
209      */
210     drawLine: function (point1, point2) {
211         var anchor = this.getAnchor();
212         
213         point1 = this._scaleDownToZoomFactor(point1);
214         point2 = point2 && this._scaleDownToZoomFactor(point2);
215         
216         point1 = point1.getTranslated(anchor.x, anchor.y);
217         point2 = point2 && point2.getTranslated(anchor.x, anchor.y);
218         
219         this.drawingElement._parent.drawLine(point1, point2);
220     },
221     
222     /**
223      * Draws a rectangle. All coordinates are relative to the control point.
224      * If the last drawn point differs from <tt>rectangle</tt>'s top-left point, a straight line is drawn to join them.<br>
225      * This is a drawing primitive that creates or continues a path.
226      * @param {Webgram.Geometry.Rectangle} rectangle the rectangle to draw
227      */
228     drawRect: function (rectangle) {
229         this.drawPoly(rectangle.getPoly(), true);
230     },
231     
232     /**
233      * Draws a polygonal line. All coordinates are relative to the control point.
234      * If the last drawn point differs from the first point of <tt>poly</tt>, a straight line is drawn to join them.<br>
235      * This is a drawing primitive that creates or continues a path.
236      * @param {Webgram.Geometry.Poly} poly the poly to draw
237      * @param {Boolean} closed set to <tt>true</tt> to join the first and the last points of the poly, <tt>false</tt> otherwise
238      */
239     drawPoly: function (poly, closed) {
240         var anchor = this.getAnchor();
241         
242         var newPoints = [];
243         for (var i = 0; i < poly.points.length; i++) {
244             var point = poly.points[i];
245             
246             point = this._scaleDownToZoomFactor(point);
247             point = point.getTranslated(anchor.x, anchor.y);
248             
249             newPoints.push(point);
250         }
251         
252         poly = new Webgram.Geometry.Poly(newPoints);
253         
254         this.drawingElement._parent.drawPoly(poly, closed);
255     },
256     
257     /**
258      * Draws an ellipsoidal arc. All coordinates are relative to the control point.
259      * If the last drawn point differs from the staring point of the arc, a straight line is drawn to join them.<br>
260      * This is a drawing primitive that creates or continues a path.
261      * @param {Webgram.Geometry.Point} center the central point of the ellipsoid that the arc is based upon
262      * @param {Number} radiusX the X radius of the ellipsoid that the arc is based upon 
263      * @param {Number} radiusY the Y radius of the ellipsoid that the arc is based upon
264      * @param {Number} startAngle the starting angle of the arc
265      * @param {Number} startAngle the ending angle of the arc
266      */
267     drawArc: function (center, radiusX, radiusY, startAngle, endAngle) {
268         var anchor = this.getAnchor();
269         
270         center = center.getTranslated(anchor.x, anchor.y);
271         
272         this.drawingElement._parent.drawArc(center, radiusX, radiusY, startAngle, endAngle);
273     },
274     
275     /**
276      * Draws a quadratic or a cubic Bezier curve. All coordinates are relative to the control point.
277      * If the last drawn point differs from the staring point of the curve, a straight line is drawn to join them.
278      * Set <tt>controlPoint2</tt> to <tt>null</tt> or <tt>undefined</tt> to draw a quadratic curve; a cubic one is drawn, otherwise.<br>
279      * This is a drawing primitive that creates or continues a path. 
280      * @param {Webgram.Geometry.Point} point1 the starting point of the curve
281      * @param {Webgram.Geometry.Point} point2 the starting point of the curve 
282      * @param {Webgram.Geometry.Point} controlPoint1 the first control point of the curve
283      * @param {Webgram.Geometry.Point} controlPoint2 the second control point of the curve, in case of a cubic Bezier curve
284      */
285     drawBezier: function (point1, point2, controlPoint1, controlPoint2) {
286         var anchor = this.getAnchor();
287         
288         point1 = point1.getTranslated(anchor.x, anchor.y);
289         point2 = point2.getTranslated(anchor.x, anchor.y);
290         controlPoint1 = controlPoint1.getTranslated(anchor.x, anchor.y);
291         controlPoint2 = controlPoint2.getTranslated(anchor.x, anchor.y);
292         
293         this.drawingElement._parent.drawBezier(point1, point2, controlPoint1, controlPoint2);
294     },
295     
296     /**
297      * Draws a scalar image. All coordinates are relative to the control point.
298      * @param {Image} image the JavaScript image object with a loaded image
299      * @param {Webgram.Geometry.Point} center the central point of the drawn image
300      * @param {Webgram.Geometry.Size} size the size of the drawn image;
301      *  use <tt>image.width</tt> and <tt>image.height</tt> to draw the image unscaled
302      * @param {Number} rotationAngle the at which the image is rotated when drawn
303      * @param {Number} alpha alpha (transparency) value to use when drawing the image (between 0 and 1)
304      */
305     drawImage: function (image, center, size, rotationAngle, alpha) {
306         var anchor = this.getAnchor();
307         
308         center = center.getTranslated(anchor.x, anchor.y);
309         
310         this.drawingElement._parent.drawImage(image, center, size, rotationAngle, alpha);
311     },

313     /**
314      * Draws a piece of text. All coordinates are relative to the control point.
315      * @param {String} text the textual content to draw
316      * @param {Webgram.Geometry.Rectangle} box the bounding box of the drawn text
317      * @param {Webgram.Styles.TextStyle} textStyle the style to use when drawing the text
318      */
319     drawText: function (text, box, textStyle) {
320         if (box == null) {
321             box = this.getBaseRectangle();
322         }
323         
324         var anchor = this.getAnchor();
325         
326         box = box.getTranslated(anchor.x, anchor.y);
327         
328         this.drawingElement._parent.drawText(text, box, textStyle);
329     },

331     /**
332      * This method must be called after using one or more path-related drawing primitives.
333      * It actually commits the path-drawing operations that were previously queued.
334      * No path operations should remain "unpainted" (i.e. with no call to <tt>paint</tt> after them).   
335      * If <tt>strokeStyle</tt> is specified, a stroke is drawn along the path, using the given style.
336      * If <tt>strokeStyle</tt> is undefined, the control points's stroke style is used ({@link Webgram.ControlPoint#getStrokeStyle}).
337      * If <tt>strokeStyle</tt> is null, no stroke is drawn along the path at all.
338      * The same algorithm is used for filling the path, using the <tt>fillStyle</tt> argument.
339      * @param {Webgram.Styles.StrokeStyle} strokeStyle the stroke style to use when painting the primitives
340      * @param {Webgram.Styles.FillStyle} fillStyle the fill style to use when painting the primitives
341      */
342     paint: function (strokeStyle, fillStyle) {
343         if (strokeStyle === undefined) {
344             strokeStyle = this._strokeStyle;
345         }
346         if (fillStyle === undefined) {
347             fillStyle = this._fillStyle;
348         }
349         
350         if (strokeStyle) {
351             strokeStyle = strokeStyle.clone();
352         }
353         if (fillStyle) {
354             fillStyle = fillStyle.clone();
355         }
356         
357         var anchor = this.getAnchor();
358         
359         var styles = [strokeStyle, fillStyle];
360         for (var i = 0; i < styles.length; i++) {
361             var style = styles[i];
362             if (!style) {
363                 continue;
364             }
365             
366             if (style.gradientPoint1) {
367                 style.gradientPoint1 = this._scaleDownToZoomFactor(style.gradientPoint1);
368                 style.gradientPoint1 = style.gradientPoint1.getTranslated(anchor.x, anchor.y);
369             }
370             if (style.gradientPoint2) {
371                 style.gradientPoint2 = this._scaleDownToZoomFactor(style.gradientPoint2);
372                 style.gradientPoint2 = style.gradientPoint1.getTranslated(anchor.x, anchor.y);
373             }
374         }
375         
376         this.drawingElement._parent.paint(strokeStyle, fillStyle);
377     },

438     /**
439      * Determines if a given <tt>point</tt> (relative to the element's parent)
440      * is inside this control point or not.<br><br>
441      * <em>(could be overridden)</em> 
442      * @param {Webgram.Geometry.Point} point the point to test, relative to the element
443      * @returns {Boolean} <tt>true</tt> if the point is inside this control point, <tt>false</tt> otherwise
444      */
445     pointInside: function (point) {
446         var radius = this.radius / this.getZoomFactor();
447         
448         var anchor = this.getAnchor();
449         var rect = new Webgram.Geometry.Rectangle(anchor.x - radius, anchor.y - radius, anchor.x + radius, anchor.y + radius);
450 
451         return rect.pointInside(point);

497     /**
498      * This method is responsible for the effect generated by moving this control point.
499      * Implement this method in concrete control point classes and change the
500      * corresponding fields of the drawing element based on <tt>point</tt> and <tt>vanillaPoint</tt> arguments.<br><br>
501      * <em>(must be overridden)</em>
502      * @param {Webgram.Geometry.Point} point the point where the the control point was dragged with the mouse;
503      *  this argument contains the value after snapping and applying other limitations; its value is relative to the
504      *  element
505      * @param {Webgram.Geometry.Point} vanillaPoint the exact point of the mouse where the control point was dragged;
506      *  this argument holds the value before any snapping or alteration; its value is relative to the element's parent
507      */
508     processMove: function (point, vanillaPoint) {
509     },
510     
511     /**
512      * Moves the control point to the given <tt>point</tt>.
513      * This method is called by the active {@link Webgram.DrawingControl} as a consequence
514      * of dragging the control point with the mouse.<br><br>
515      * <em>(should not be overridden)</em>
516      * @param {Webgram.Geometry.Point} point the point that this control point will be moved to
517      */
518     move: function (point) {
519         var offsets = this._getOffsets();
520         var vanillaPoint = point;
521         point = this.drawingElement.transformInverse(point);
522         point = point.getTranslated(-offsets.x, -offsets.y);
523         
524         point = this.snap(point);
525         this.processMove(point, vanillaPoint);
526         this.onMove.trigger(point);
527         this.update();
528     },
529     
530     /**
531      * Recomputes the anchor of this element. If the anchor has changed,
532      * that is the control point was moved, an <tt>onUpdate</tt> event is triggered.
533      * This method is automatically called by the library when the control points
534      * are invalidated, but can be called by the user as well.<br><br>
535      * <em>(should not be overridden)</em>
536      */
537     update: function () {
538         var anchor = this.computeAnchor();
539         var offsets = this._getOffsets();
540         var oldAnchor = this._anchor;
541         
542         anchor = anchor.getTranslated(offsets.x, offsets.y);
543         anchor = this.drawingElement.transformDirect(anchor);
544         this._anchor = anchor;
545         
546         if (oldAnchor == null || this._anchor.x !== oldAnchor.x || this._anchor.y !== oldAnchor.y) {
547             this.onUpdate.trigger();
548         }
549     },
550     
551     /**
552      * Snaps the control point. The default implementation
553      * simply snaps to <em>grid</em>, if enabled.
554      * @param {Webgram.Geometry.Point} point the point to snap
555      * @returns {Webgram.Geometry.Point} the snapped point
556      */
557     snap: function (point) {
558         if (this.isSnapToGridEnabled() && this.getSetting('snapGrid') != null) {
559             point = this.snapToGrid(point);
560         }
561         
562         return point;
563     },
564     
565     /**
566      * Snaps the given point to the <em>grid</em>.
567      * @param {Webgram.Geometry.Point} point the point to snap
568      * @returns {Webgram.Geometry.Point} the snapped point
569      */
570     snapToGrid: function (point) {
571         var snapGrid = this.getSetting('snapGrid');
572         if (snapGrid == null) {
573             return point;
574         }
575         
576         point = new Webgram.Geometry.Point(Math.round(point.x / snapGrid.sizeX) * snapGrid.sizeX, Math.round(point.y / snapGrid.sizeY) * snapGrid.sizeY);
577         
578         return point;
579     },

621     _getOffsets: function () {
622         var zoomFactor = this.getZoomFactor();
623         var strokeStyle = this.drawingElement.getStrokeStyle();
624         var thickness = Math.round(strokeStyle ? strokeStyle.lineWidth / 2 : 0);
625         var offsets = this.getOffsets();
626         var x = offsets.x / zoomFactor;
627         var y = offsets.y / zoomFactor;
628         
629         if (offsets.x > 0) {
630             x += thickness;
631         }
632         else if (offsets.x < 0) {
633             x -= thickness;
634         }
635         
636         if (offsets.y > 0) {
637             y += thickness;
638         }
639         else if (offsets.y < 0) {
640             y -= thickness;
641         }
642         
643         return {x : x, y : y};
644     },
645     
646     _scaleDownToZoomFactor: function (point) {
647         var angle = Webgram.Geometry.Point.zero().getAngleTo(point);
648         var distance = Webgram.Geometry.Point.zero().getDistanceTo(point);
649         
650         return Webgram.Geometry.Point.zero().getPointAt(distance / this.getZoomFactor(), angle);
651     }
