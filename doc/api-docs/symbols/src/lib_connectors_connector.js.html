 21 /**
 22  * @namespace Contains the classes needed to implement <em>connectors</em> in webgram.
 23  * Connectors are polygonal drawing elements with certain <em>end points</em> that
 24  * remain connected to <em>sockets</em>, when these sockets move along with their drawing elements.
 25  */
 26 Webgram.Connectors = Webgram.Namespace('Webgram.Connectors');
 27 
 28 /**
 29  * @class A special polygonal element that has one or more instances of
 30  * {@link Webgram.Connectors.EndPoint} as control points and allows
 31  * a visual interconnection between drawing elements.
 32  * @extends Webgram.DrawingElements.PolyElement
 33  * @see Webgram.Connectors.EndPoint
 34  * @see Webgram.Connectors.Socket
 35  * @param {String} id a given identifier, can be <tt>null</tt>
 36  * @param {Array} points a list of {@link Webgram.Geometry.Point} objects
 37  */
 38 Webgram.Connectors.Connector = function (id, points) {
 39     Webgram.DrawingElements.PolyElement.call(this, id, points);
 40     
 41     this.zIndex = 10;
 42     
 43     /**
 44      * An event that is triggered whenever the connector is connected to a socket.<br>
 45      * Handlers receive the following arguments: <tt>(endPoint, socket)</tt> 
 46      * @type Webgram.Event
 47      */
 48     this.onConnect = new Webgram.Event('connect', this); /* (endPoint, socket) */
 49 
 50     /**
 51      * An event that is triggered whenever the connector is disconnected from a socket.<br>
 52      * Handlers receive the following arguments: <tt>(endPoint, socket)</tt> 
 53      * @type Webgram.Event
 54      */
 55     this.onDisconnect = new Webgram.Event('disconnect', this); /* (endPoint, socket) */
 56 };
 57 
 58 Webgram.Connectors.Connector.prototype = {
 59     draw: function () {
 60         this.drawPoly(this.getDrawingPoly(), false);
 61         this.paint(undefined, null);
 62     },
 63     
 64     saveParentRelativePosition: function () {
 65         var parentWidth = this._parent.getWidth();
 66         var parentHeight = this._parent.getHeight();
 67         
 68         this._parentRelativePositions = [];
 69         for (var i = 0; i < this.shape.points.length; i++) {
 70             var point = this.shape.points[i];
 71             point = this.rotateDirect(point);
 72             
 73             this._parentRelativePositions.push({
 74                 x: point.x / (parentWidth - 1),
 75                 y: point.y / (parentHeight - 1)
 76             });
 77         }
 78     },
 79     
 80     restoreParentRelativePosition: function () {
 81         var parentWidth = this._parent.getWidth();
 82         var parentHeight = this._parent.getHeight();
 83         
 84         /* update shape points */
 85         for (var i = 0; i < this._parentRelativePositions.length; i++) {
 86             var controlPoint = this.getPolyControlPointByIndex(i);
 87             if (controlPoint && (controlPoint instanceof Webgram.Connectors.EndPoint) && controlPoint.isConnected()) {
 88                 continue;
 89             }
 90             
 91             var percents = this._parentRelativePositions[i];
 92             
 93             var point = new Webgram.Geometry.Point(
 94                     percents.x * (parentWidth - 1),
 95                     percents.y * (parentHeight - 1)
 96             );
 97             
 98             this.shape.points[i] = point;
 99         }
100         
101         var newCenter = this.shape.getBoundingRectangle().getCenter();
102         this.shape= this.shape.getRotated(-this.getRotationAngle(), newCenter);
103         
104         this.invalidateBaseRectangle();
105     },
106     
107     remControlPoint: function (controlPoint) {
108         /* we have to override this method in order to disconnect
109          * an end point before removing actually it */
110         
111         if (controlPoint instanceof Webgram.Connectors.EndPoint && controlPoint.isConnected()) {
112             controlPoint.socket.disconnect(controlPoint);
113         }
114         
115         Webgram.DrawingElements.PolyElement.prototype.remControlPoint.call(this, controlPoint);
116     },
117     
118     getPolyControlPointClass: function (index) {
119         return Webgram.Connectors.EndPoint;
120     },
121     
122     snapExternally: function (index, point, fixedPoints, rotationAngle, center) {
123         var result = this._snapToSockets(index, point, fixedPoints, rotationAngle, center);
124         if (result != null) {
125             return result;
126         }
127         
128         return Webgram.DrawingElement.prototype.snapExternally.call(this, index, point, fixedPoints, rotationAngle, center);
129     },
130     
131     setEditEnabled: function (enabled) {
132         if (!enabled) {
133             /* disconnect all endPoints before removing them */
134             this._disconnectAllEndPoints();
135         }
136         
137         Webgram.DrawingElements.PolyElement.prototype.setEditEnabled.call(this, enabled);
138     },
139     
140     getEndPoints: function () {
141         return this.getControlPoints(Webgram.Connectors.EndPoint);
142     },
143     
144     moveTo: function (point) {
145         /* We override this function because we want to make it harder
146          * to move a connector when at least one of its end points is connected.
147          * The move distance has to be greater than a threshold in order to move
148          * the element. */
149         
150         var endPoints = this.getEndPoints();
151         var endPoint = null;
152         for (var i = 0; i < endPoints.length; i++) {
153             if (endPoints[i].isConnected()) {
154                 endPoint = endPoints[i];
155                 break;
156             }
157         }
158         
159         if (endPoint) { /* at least an end point is connected */
160             var center = this.getCenter();
161             var dist = center.getDistanceTo(point);
162             if (dist < 4 * endPoint.radius) {
163                 return;
164             }
165         }
166         
167         this._disconnectAllEndPoints();
168         
169         Webgram.DrawingElement.prototype.moveTo.call(this, point);
170     },
171     
172     flipHorizontally: function () {
173         /* we override this method because we want
174          * a connector to disconnect when flipped */
175         
176         if (!this.isFlipEnabled()) {
177             return;
178         }
179         
180         this._disconnectAllEndPoints();
181 
182         Webgram.DrawingElements.PolyElement.prototype.flipHorizontally.call(this);
183     },
184     
185     flipVertically: function () {
186         /* we override this method because we want
187          * a connector to disconnect when flipped */
188         
189         if (!this.isFlipEnabled()) {
190             return;
191         }
192         
193         this._disconnectAllEndPoints();
194 
195         Webgram.DrawingElements.PolyElement.prototype.flipVertically.call(this);
196     },
197 
198     continueCreate: function (point, size, mouseDown, click) {
199         /* we override this method because we want to stop the creation
200          * when an end point (different than the very first one) gets connected */
201         
202         var count = this.shape.points.length;
203         if (click && count > 1 && count >= this.minPointCount) {
204             var controlPoint = this.getPolyControlPointByIndex(count - 1);
205             if (controlPoint instanceof Webgram.Connectors.EndPoint && controlPoint.isConnected()) {
206                 this._pendingCreatePoint = false;
207                 return false; /* stop */
208             }
209         }
210         
211         return Webgram.DrawingElements.PolyElement.prototype.continueCreate.call(this, point, size, mouseDown, click);
212     },
213     
214     isSnapExternallyEnabled: function () {
215         /* a connector cannot function without this feature enabled */
216         
217         return true;
218     },
219     
220     _snapToSockets: function (index, point, fixedPoints, rotationAngle, center) {
221         var endPoint = this.getPolyControlPointByIndex(index);
222         if (!(endPoint instanceof Webgram.Connectors.EndPoint)) {
223             return null;
224         }
225         
226         /* rotate the point to get the actual position,
227          * relative to the parent */
228         point = point.getRotated(rotationAngle, center);
229         
230         /* if the endpoint is over a socket, 
231          * place it directly in the middle of the socket */
232         var socket = endPoint._getHoveredSocket(point);
233         if (socket) {
234             return socket.getAnchor().getRotated(-rotationAngle, center);
235         }
236     },
237     
238     _setRotationAngle: function (angle) {
239         this._disconnectAllEndPoints();
240         
241         Webgram.DrawingElements.PolyElement.prototype._setRotationAngle.call(this, angle);
242     },
243     
244     _disconnectAllEndPoints: function () {
245         var endPoints = this.getEndPoints();
246         
247         for (var i = 0; i < endPoints.length; i++) {
248             var endPoint = endPoints[i];
249             if (endPoint.isConnected()) {
250                 endPoint.socket.disconnect(endPoint);
251             }
252         }
253     }
254 };
255 
256 Webgram.Class('Webgram.Connectors.Connector', Webgram.DrawingElements.PolyElement);
