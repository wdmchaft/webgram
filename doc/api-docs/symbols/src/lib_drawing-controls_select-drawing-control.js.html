 28 /**
 29  * @class The main drawing control, used to manipulate the existing drawing elements,
 30  * as well as for zooming and panning.
 31  * @extends Webgram.DrawingControl
 32  * @param {Webgram.DrawingElements.RootContainer} rootContainer the root container element of the webgram
 33  */
 34 Webgram.DrawingControls.SelectDrawingControl = function (rootContainer) {
 35     Webgram.DrawingControl.call(this, rootContainer);
 36     
 37     /**
 38      * The currently hovered drawing element, or <tt>null</tt> if no element is hovered.
 39      * @type Webgram.DrawingElement
 40      */
 41     this.hoveredDrawingElement = null;
 42 
 43     /**
 44      * The currently selected drawing element, or <tt>null</tt> if no element is selected.
 45      * @type Webgram.DrawingElement
 46      */
 47     this.selectedDrawingElement = null;
 48     
 49     /**
 50      * The currently hovered control point, or <tt>null</tt> if no control point is hovered.
 51      * @type Webgram.ControlPoint
 52      */
 53     this.hoveredControlPoint = null;
 54 
 55     /**
 56      * The currently selected control point, or <tt>null</tt> if no control point is selected.
 57      * @type Webgram.ControlPoint
 58      */
 59     this.selectedControlPoint = null;
 60 
 61     /**
 62      * The currently hovered action menu item, or <tt>null</tt> if no action menu item is hovered.
 63      * @type Webgram.ActionMenuItem
 64      */
 65     this.hoveredActionMenuItem = null;
 66     
 67     /**
 68      * An event that is triggered when the current selection of drawing elements changes.<br>
 69      * Handlers receive the following arguments: <tt>(drawingElements)</tt>.
 70      * @type Webgram.Event
 71      */
 72     this.onSelectionChange = new Webgram.Event('selection change', this); /* (drawingElements) */
 73     
 74     this._mouseDownPoint = null;
 75     this._panMouseDownPoint = null;
 76     this._msContainerElement = null;
 77 
 78     this.onDeactivate.bind(function () {
 79         this.setSelectedDrawingElements(null);
 80         
 81         this.hoveredDrawingElement = null;
 82         this.selectedDrawingElement = null;
 83         this.hoveredControlPoint = null;
 84         this.selectedControlPoint = null;
 85         this.hoveredActionMenuItem = null;
 86         
 87         this._mouseDownPoint = null;
 88         this._msContainerElement = null;
 89     });
 90     
 91     this.onSelectionChange.bind(function (drawingElements) {
 92         this.rootContainer.webgram.onSelectionChange.trigger(drawingElements);
 93     });
 94 };

 96 Webgram.DrawingControls.SelectDrawingControl.prototype = {
 97     getSelectedDrawingElements: function () {
 98         if (this._msContainerElement) {
 99             return this._msContainerElement.drawingElements.slice(0);
100         }
101         else if (this.selectedDrawingElement) {
102             return [this.selectedDrawingElement];
103         }
104         else {
105             return [];
106         }
107     },
108     
109     setSelectedDrawingElements: function (drawingElements) {
110         /* clear the multiple selection, if any */
111         if (this._msContainerElement) {
112             this._remFromMultipleSelection();
113         }
114         
115         if (this.selectedDrawingElement) {
116             this.selectedDrawingElement.setFocusType(Webgram.DrawingElement.FOCUS_NONE);
117             this.selectedDrawingElement = null;
118         }
119         
120         if (drawingElements && drawingElements.length) {
121             if (drawingElements.length === 1) { /* single selected element */
122                 this.selectedDrawingElement = drawingElements[0];
123                 if (this.selectedDrawingElement) {
124                     this.selectedDrawingElement.setFocusType(Webgram.DrawingElement.FOCUS_SELECTED);
125                 }
126                 
127                 this.onSelectionChange.trigger([this.selectedDrawingElement]);
128             }
129             else { /* multiple selected elements */
130                 if (this.getSetting('multipleSelectionEnabled')) {
131                     for (var i = 0; i < drawingElements.length; i++) {
132                         drawingElements[i].setFocusType(Webgram.DrawingElement.FOCUS_SELECTED_MULTIPLE);
133                     }
134                     
135                     this._msContainerElement = new Webgram.DrawingControls.SelectDrawingControl._MultipleSelectionContainer(0, 0, 0, 0);
136                     this.rootContainer._addDrawingElement(this._msContainerElement);
137                     this.selectedDrawingElement = this._msContainerElement;
138                     this._finishMultipleSelection(); /* this will trigger the onSelectionChange event */
139                 }
140                 else {
141                     this.onSelectionChange.trigger([]);
142                 }
143             }
144         }
145         else {
146             this.onSelectionChange.trigger([]);            
147         }
148         
149         this.selectedControlPoint = null;   
150         this.hoveredControlPoint = null;
151         this.hoveredActionMenuItem = null;
152     },

157     /**
158      * This action copies the current selection of drawing elements into the clipboard.
159      * @returns {Boolean} <tt>true</tt> if the action succeeded, <tt>false</tt> otherwise
160      */
161     doCopyAction: function (byKeyboard) {
162         var drawingElements = this.getSelectedDrawingElements();
163         var jsonList = [];
164         
165         if (drawingElements.length === 0) {
166             return false; /* nothing to do */
167         }
168         
169         for (var i = 0; i < drawingElements.length; i++) {
170             var drawingElement = drawingElements[i];
171             var json = drawingElement.toJson();
172             jsonList.push(json);
173         }
174         
175         var webgram = this.getWebgram();
176         webgram.setClipboard('json/drawing-elements', jsonList);
177 
178         webgram.onCopyAction.trigger(drawingElements.slice(0), byKeyboard === true);
179         
180         return true;
181     },

183     /**
184      * This action pastes the contents of the clipboard onto the root container.
185      * @returns {Boolean} <tt>true</tt> if the action succeeded, <tt>false</tt> otherwise
186      */
187     doPasteAction: function (byKeyboard) {
188         var webgram = this.getWebgram();
189         var clipboard = webgram.getClipboard();
190         
191         if (clipboard.type === 'json/drawing-elements') {
192             var jsonList = clipboard.content;
193             var i, drawingElement, drawingElements = [];
194             
195             for (i = 0; i < jsonList.length; i++) {
196                 var json = jsonList[i];
197                 var klass = Webgram.Utils.getField(json.className);
198                 if (klass == null) { /* no such class, ignoring */
199                     continue;
200                 }
201                 
202                 delete json.id; /* we don`t duplicate the id */
203                 
204                 drawingElement = new klass(); /* create an instance of klass with default properties */
205                 drawingElement.fromJson(json);
206                 
207                 this.rootContainer._addDrawingElement(drawingElement, null, true);
208                 
209                 drawingElements.push(drawingElement);
210             }
211             
212             if (drawingElements.length === 0) {
213                 return false; /* no DE was actually pasted */
214             }
215             
216             /* move the newly created DEs next to the currently selected DE */
217             var selectedDrawingElements = this.getSelectedDrawingElements();
218             var center = null;
219             if (selectedDrawingElements.length > 0) {
220                 center = selectedDrawingElements[0].getCenter();
221             }
222             
223             var delta = this.getSetting('mainGrid.sizeX', 25);
224             var deltaX = delta, deltaY = delta;
225             if (center != null) {
226                 var oldCenter = drawingElements[0].getCenter();
227                 deltaX += center.x - oldCenter.x;
228                 deltaY += center.y - oldCenter.y;
229             }
230             
231             for (i = 0; i < drawingElements.length; i++) {
232                 drawingElement = drawingElements[i];
233                 drawingElement.moveTo(drawingElement.getCenter().getTranslated(deltaX, deltaY));
234             }
235             
236             this.setSelectedDrawingElements(drawingElements);
237             
238             webgram.onPasteAction.trigger(drawingElements.slice(0), byKeyboard === true);
239             
240             return true;
241         }
242         
243         return false;
244     },

246     /**
247      * This action duplicates the current selection of drawing elements on the root container.
248      * @returns {Boolean} <tt>true</tt> if the action succeeded, <tt>false</tt> otherwise
249      */
250     doDuplicateAction: function (byKeyboard) {
251         var drawingElement, drawingElements = this.getSelectedDrawingElements();
252         var i, json, jsonList = [];
253         var webgram = this.getWebgram();
254         
255         if (drawingElements.length === 0) {
256             return false; /* nothing to do */
257         }
258         
259         for (i = 0; i < drawingElements.length; i++) {
260             drawingElement = drawingElements[i];
261             json = drawingElement.toJson();
262             jsonList.push(json);
263         }
264         
265         drawingElements = [];
266         
267         for (i = 0; i < jsonList.length; i++) {
268             json = jsonList[i];
269             var klass = Webgram.Utils.getField(json.className);
270             if (klass == null) { /* no such class, ignoring */
271                 continue;
272             }
273             
274             delete json.id; /* we don`t duplicate the id */
275             
276             drawingElement = new klass(); /* create an instance of klass with default properties */
277             drawingElement.fromJson(json);
278             
279             this.rootContainer._addDrawingElement(drawingElement, null, true);
280             
281             drawingElements.push(drawingElement);
282         }
283         
284         if (drawingElements.length === 0) {
285             return false; /* no DE was actually duplicated */
286         }
287         
288         /* move the newly created DEs next to the currently selected DE */
289         var center = null;
290         if (this.selectedDrawingElement != null) {
291             center = this.selectedDrawingElement.getCenter();
292         }
293         
294         var mainGridSize = this.getSetting('mainGrid.sizeX', 25);
295         var snapDistance = this.getSetting('snapDistance', 10) + 1;
296         var delta = Math.max(mainGridSize, snapDistance);
297         
298         var deltaX = delta, deltaY = delta;
299         if (center != null) {
300             var oldCenter = drawingElements[0].getCenter();
301             deltaX += center.x - oldCenter.x;
302             deltaY += center.y - oldCenter.y;
303         }
304         
305         for (i = 0; i < drawingElements.length; i++) {
306             drawingElement = drawingElements[i];
307             drawingElement.moveTo(drawingElement.getCenter().getTranslated(deltaX, deltaY));
308         }
309         
310         this.setSelectedDrawingElements(drawingElements);
311         
312         webgram.onDuplicateAction.trigger(drawingElements.slice(0), byKeyboard === true);
313         
314         return true;
315     },

317     /**
318      * This action deletes the current selection of drawing elements from the root container.
319      * @returns {Boolean} <tt>true</tt> if the action succeeded, <tt>false</tt> otherwise
320      */
321     doDeleteAction: function (byKeyboard) {
322         var drawingElements = this.getSelectedDrawingElements();
323         if (drawingElements.length > 0) {
324             if (drawingElements.length > 1) { /* multiple selection */
325                 this.setSelectedDrawingElements();
326                 
327                 for (var i = 0; i < drawingElements.length; i++) {
328                     this.rootContainer.remDrawingElement(drawingElements[i]);
329                 }
330             }
331             else { /* single element selection */
332                 this.rootContainer.remDrawingElement(drawingElements[0]);
333             }
334             
335             this.getWebgram().onDeleteAction.trigger(drawingElements.slice(0), byKeyboard === true);
336             
337             return true;
338         }
339         
340         return false;
341     },

343     /**
344      * This action moves the current selection of drawing elements on the root container.
345      * The move distance is determined by the <tt>snapGrid</tt> global setting if the <tt>fine</tt>
346      * argument is <tt>true</tt>, and by the <tt>mainGrid</tt> global setting otherwise.
347      * @param {String} direction a string indicating the direction:
348      *  <tt>'top'</tt>, <tt>'right'</tt>, <tt>'bottom'</tt> or <tt>'left'</tt>
349      * @param {Boolean} fine pass <tt>true</tt> to perform a fine moving,
350      *  or <tt>false</tt><tt>undefined</tt> to perform a coarse moving
351      * @returns {Boolean} <tt>true</tt> if the action succeeded, <tt>false</tt> otherwise
352      */
353     doMoveAction: function (direction, fine, byKeyboard) {
354         var size;
355         if (fine) {
356             size = this.getSetting('snapGrid.sizeX', 5);
357         }
358         else {
359             size = this.getSetting('mainGrid.sizeX', 25);
360         }
361         
362         var drawingElement = this.selectedDrawingElement;
363         if (drawingElement == null) {
364             return false;
365         }
366         
367         var center;
368         if (direction === 'up') {
369             center = drawingElement.getCenter().getTranslated(0, -size);
370         }
371         else if (direction === 'right') {
372             center = drawingElement.getCenter().getTranslated(size, 0);
373         }
374         else if (direction === 'down') {
375             center = drawingElement.getCenter().getTranslated(0, size);
376         }
377         else if (direction === 'left') {
378             center = drawingElement.getCenter().getTranslated(-size, 0);
379         }
380         else {
381             return false;
382         }
383         
384         drawingElement.moveTo(center);
385         
386         this.invalidate(true);
387         
388         this.getWebgram().onMoveAction.trigger(this.getSelectedDrawingElements().slice(0), direction, byKeyboard === true);
389         
390         return true;
391     },

393     /**
394      * This action brings the current selection of drawing elements to front with one position.
395      * @returns {Boolean} <tt>true</tt> if the action succeeded, <tt>false</tt> otherwise
396      */
397     doBringToFrontAction: function (byKeyboard) {
398         var drawingElement = this.selectedDrawingElement;
399         if (drawingElement == null) {
400             return false;
401         }
402         
403         var index = this.rootContainer.getDrawingElementIndex(drawingElement);
404         this.rootContainer._setDrawingElementIndex(drawingElement, index + 1, true);
405         if (index !== this.rootContainer.getDrawingElementIndex(drawingElement)) {
406             this.getWebgram().onBringToFrontAction.trigger(this.getSelectedDrawingElements().slice(0), byKeyboard === true);
407             
408             return true;
409         }
410         
411         return false;
412     },

475     /**
476      * This action increases the zoom level if the maximum zoom level
477      * hasn't been reached yet.
478      * @returns {Boolean} <tt>true</tt> if the action succeeded, <tt>false</tt> otherwise
479      */
480     doZoomInAction: function (byKeyboard) {
481         var webgram = this.getWebgram();
482         var zoomLevel = webgram.getZoomLevel();
483         var zoomFactors = this.getSetting('zoomFactors');
484         var zoomLevelCount = zoomFactors.length;
485         
486         if (zoomLevel < zoomLevelCount - 1) {
487             webgram.setZoomLevel(zoomLevel + 1);
488             webgram.onZoomInAction.trigger(zoomLevel + 1, byKeyboard === true);
489             
490             return true;
491         }
492         
493         return false;
494     },

497      * This action decreases the zoom level if the minimum zoom level
498      * hasn't been reached yet.
499      * @returns {Boolean} <tt>true</tt> if the action succeeded, <tt>false</tt> otherwise
500      */
501     doZoomOutAction: function (byKeyboard) {
502         var webgram = this.getWebgram();
503         var zoomLevel = webgram.getZoomLevel();
504         
505         if (zoomLevel > 0) {
506             webgram.setZoomLevel(zoomLevel - 1);
507             webgram.onZoomOutAction.trigger(zoomLevel - 1, byKeyboard === true);
508             
509             return true;
510         }
511         
512         return false;

515     /**
516      * This action sets the zoom level to 1:1.
517      * @returns {Boolean} <tt>true</tt> if the action succeeded, <tt>false</tt> otherwise
518      */

520         var webgram = this.getWebgram();
521         var zoomFactor = webgram.getZoomFactor();
522         var zoomFactors = this.getSetting('zoomFactors');

524         if (zoomFactor !== 1) {
525             for (var i = 0; i < zoomFactors.length; i++) {
526                 if (zoomFactors[i] === 1) {
527                     webgram.setZoomLevel(i);
528                     webgram.onZoomInAction.trigger(i, byKeyboard === true);
529                     
530                     return true;
531                 }
532             }
533         }
534         
535         return false;

559     /* event handlers */
560     
561     handleMouseDown: function (point, button, modifiers) {
562         if (this.selectedDrawingElement) {
563             var result = this.selectedDrawingElement.onMouseDown.trigger(point, button, modifiers);
564             if (result) {
565                 return true;
566             }
567         }
568         else if (this.hoveredDrawingElement && this.hoveredDrawingElement.isHoveredControlPointsEnabled()) {
569             var result = this.hoveredDrawingElement.onMouseDown.trigger(point, button, modifiers);
570             if (result) {
571                 return true;
572             }
573         }
574         
575         if (button === 1) {
576             this.invalidate();
577             
578             this._mouseDownPoint = point;
579             
580             /* unselect any previously selected drawing element */
581             var prevSelectedDrawingElement = this.selectedDrawingElement; /* remember the previously selected DE */
582             if (this.selectedDrawingElement && this.selectedDrawingElement !== this.hoveredDrawingElement) {
583                 this.selectedDrawingElement.setFocusType(Webgram.DrawingElement.FOCUS_NONE);
584             }
585             
586             /* unselect any previously selected control point */
587             var prevSelectedControlPoint = this.selectedControlPoint; /* remember the previously selected CP */
588             if (this.selectedControlPoint && this.selectedControlPoint !== this.hoveredControlPoint) {
589                 this.selectedControlPoint.setFocusType(Webgram.ControlPoint.FOCUS_NONE);
590             }
591             
592             /* select the hovered drawing element and control point */
593             this.selectedControlPoint = this.hoveredControlPoint;
594             this.selectedDrawingElement = this.hoveredDrawingElement;
595             
596             /* trigger the (un)select events for DEs */
597             if (this.selectedDrawingElement !== prevSelectedDrawingElement) {
598                 if (prevSelectedDrawingElement) {
599                     prevSelectedDrawingElement.onUnselect.trigger();
600                     prevSelectedDrawingElement.disableShift();
601                 }
602                 if (this.selectedDrawingElement) {
603                     this.selectedDrawingElement.onSelect.trigger();
604                     this.onSelectionChange.trigger([this.selectedDrawingElement]);
605                 }
606                 else {
607                     this.onSelectionChange.trigger([]);
608                 }
609             }
610             
611             /* trigger the (un)select events for CPs */
612             if (this.selectedControlPoint !== prevSelectedControlPoint) {
613                 if (prevSelectedControlPoint) {
614                     prevSelectedControlPoint.onUnselect.trigger();
615                 }
616                 if (this.selectedControlPoint) {
617                     this.selectedControlPoint.onSelect.trigger();
618                 }
619             }
620             
621             if (this.selectedControlPoint) { /* clicked on a CP */
622                 this.selectedControlPoint.setFocusType(Webgram.ControlPoint.FOCUS_SELECTED);
623                 this.selectedControlPoint.onBeginMove.trigger();
624                 
625                 this.selectedDrawingElement.setFocusType(Webgram.DrawingElement.FOCUS_SELECTED);
626             }
627             else if (this.hoveredActionMenuItem) { /* clicked on an AMI */
628                 this.hoveredActionMenuItem.activate();
629                 
630                 this.selectedDrawingElement.setFocusType(Webgram.DrawingElement.FOCUS_SELECTED);
631             }
632             else if (this.selectedDrawingElement) { /* clicked on a DE */
633                 /* remember the mouse down coordinates relatively to the selected drawing element */
634                 var center = this.selectedDrawingElement.getCenter();
635                 this.selectedDrawingElement._mouseDownPoint = new Webgram.Geometry.Point(point.x - center.x, point.y - center.y);
636                 
637                 /* update the selected element's focus */
638                 this.selectedDrawingElement.setFocusType(Webgram.DrawingElement.FOCUS_SELECTED);
639                 
640                 if (modifiers.shift &&
641                         (!this._msContainerElement || (this._msContainerElement.getRotationAngle() === 0))) { /* add/remove to/from current selection */
642 
643                     if (this.selectedDrawingElement !== this._msContainerElement) {
644                         if (this._msContainerElement) { /* existing multiple selection */
645                             this._remFromMultipleSelection(true);
646                             this.selectedDrawingElement.setFocusType(Webgram.DrawingElement.FOCUS_SELECTED_MULTIPLE);
647                             this._finishMultipleSelection(point);
648                         }
649                         else { /* create new multiple selection */
650                             if (this.getSetting('multipleSelectionEnabled')) {
651                                 if (prevSelectedDrawingElement && prevSelectedDrawingElement !== this.selectedDrawingElement) {
652                                     prevSelectedDrawingElement.setFocusType(Webgram.DrawingElement.FOCUS_SELECTED_MULTIPLE);
653                                     this.selectedDrawingElement.setFocusType(Webgram.DrawingElement.FOCUS_SELECTED_MULTIPLE);
654                                     this._msContainerElement = new Webgram.DrawingControls.SelectDrawingControl._MultipleSelectionContainer(point.x, point.y, 0, 0);
655                                     this.rootContainer._addDrawingElement(this._msContainerElement);
656                                     this._finishMultipleSelection(point);
657                                 }
658                                 else {
659                                     this.selectedDrawingElement.setFocusType(Webgram.DrawingElement.FOCUS_SELECTED);
660                                 }
661                             }
662                         }
663                     }
664                 }
665                 else { /* no modifier pressed, normal click to select */
666                     if (this._msContainerElement && this.selectedDrawingElement !== this._msContainerElement) {
667                         /* clear any existing multiple selection */
668                         this._remFromMultipleSelection();
669                     }
670                 }
671             }
672             else { /* clicked outside of any DE */
673                 /* clear any existing multiple selection */
674                 if (this._msContainerElement) {
675                     this._remFromMultipleSelection();
676                 }
677                 
678                 /* start a new multiple selection */
679                 if (this.getSetting('multipleSelectionEnabled')) {
680                     /* snap the point to grid */
681                     
682                     /* snap the point */
683                     var snapGrid = this.getSetting('snapGrid');
684                     if (snapGrid != null) {
685                         point = new Webgram.Geometry.Point(
686                                 Math.round(point.x / snapGrid.sizeX) * snapGrid.sizeX,
687                                 Math.round(point.y / snapGrid.sizeY) * snapGrid.sizeY);
688                     }
689

691                     this.rootContainer._addDrawingElement(this._msContainerElement);
692                 }

694                 return true;
695             }
696         }
697         else if (button === 2 && this.getSetting('panEnabled')) { /* middle button, used for panning */
698             this._panMouseDownPoint = point;
699             
700             return true;
701         }

740     handleMouseMove: function (point, modifiers) {
741         if (this.hoveredDrawingElement) {
742             var result = this.hoveredDrawingElement.onMouseMove.trigger(point, modifiers);
743             if (result) {
744                 return result;
745             }
746         }
747 
748         if (this._panMouseDownPoint) { /* used for panning */
749             var deltaX = point.x - this._panMouseDownPoint.x;
750             var deltaY = point.y - this._panMouseDownPoint.y;
751             var offset = this.rootContainer.getOffset();
752             
753             this.rootContainer.setOffset(new Webgram.Geometry.Point(offset.x - deltaX, offset.y - deltaY));
754             this.invalidate(true);
755             
756             return 'move';
757         }
758         else if (this._mouseDownPoint) {
759             if (this.selectedControlPoint) { /* execute the control point's action */
760                 this.selectedControlPoint.move(point, true);
761                 
762                 this.invalidate();
763                 
764                 return this.selectedControlPoint.getCursor();
765             }
766             else if (this.selectedDrawingElement) {
767                 this.selectedDrawingElement.onMouseMove.trigger(point, modifiers);
768                 
769                 if (this.selectedDrawingElement.isMoveEnabled()) { /* move the whole drawing element */
770                     var deltaX = point.x - this._mouseDownPoint.x;
771                     var deltaY = point.y - this._mouseDownPoint.y;
772                     
773                     var x = this._mouseDownPoint.x - this.selectedDrawingElement._mouseDownPoint.x + deltaX;
774                     var y = this._mouseDownPoint.y - this.selectedDrawingElement._mouseDownPoint.y + deltaY;
775                     
776                     this.selectedDrawingElement.moveTo(new Webgram.Geometry.Point(x, y));
777                     
778                     this.invalidate();
779                     
780                     return 'move';
781                 }
782             }
783             else if (this._msContainerElement && this._msContainerElement.isResizeEnabled()) {
784                 /* continue an already started multiple selection */
785                 if (this._mouseDownPoint.x <= point.x) {
786                     if (this._mouseDownPoint.y <= point.y) {
787                         this._msContainerElement._setBottomRight(point, false);
788                     }
789                     else {
790                         this._msContainerElement._setTopRight(point, false);
791                     }
792                 }
793                 else {
794                     if (this._mouseDownPoint.y <= point.y) {
795                         this._msContainerElement._setBottomLeft(point, false);
796                     }
797                     else {
798                         this._msContainerElement._setTopLeft(point, false);
799                     }
800                 }
801                 
802                 this._markMultipleSelected();
803                 this.invalidate();
804                 
805                 return 'crosshair';
806             }
807         }
808         
809         /* determine the hovered drawing element & control point */
810         var hoveredDrawingElement = null;
811         var hoveredControlPoint = null;
812         var hoveredActionMenuItem = null;
813         for (var i = 0; i < this.rootContainer.drawingElements.length; i++) {
814             var drawingElement = this.rootContainer.drawingElements[i];
815             if (!drawingElement.isSelectEnabled()) {
816                 continue;
817             }
818             
819             if (((this.selectedDrawingElement === drawingElement) || drawingElement.isHoveredControlPointsEnabled())) {
820                 hoveredControlPoint = drawingElement.pointInsideControlPoint(point);
821                 if (hoveredControlPoint) {
822                     hoveredDrawingElement = drawingElement;
823                     break;
824                 }
825                 hoveredActionMenuItem = drawingElement.pointInsideActionMenuItem(point);
826                 if (hoveredActionMenuItem) {
827                     hoveredDrawingElement = drawingElement;
828                     break;
829                 }
830             }
831             
832             if (drawingElement.pointInside(point)) {
833                 hoveredDrawingElement = drawingElement;
834             }
835         }
836         
837         /* by this point we have the hovered control point and drawing element 
838          * we can also be sure the hoveredControlPoint and hoveredActionMenuItem both belong to the hoveredDrawingElement */
839         
840         if (hoveredDrawingElement && hoveredDrawingElement.getFocusType() > Webgram.DrawingElement.FOCUS_SELECTED_MULTIPLE) { /* selected remotely */
841             return 'not-allowed';
842         }
843         
844         var hoverChanged = false;
845         
846         if (hoveredDrawingElement !== this.hoveredDrawingElement) { /* not the same hovered element */
847             /* remove hover from the currently hovered element */
848             if (this.hoveredDrawingElement) {
849                 if (this.hoveredDrawingElement.getFocusType() === Webgram.DrawingElement.FOCUS_HOVERED) {
850                     this.hoveredDrawingElement.setFocusType(Webgram.DrawingElement.FOCUS_NONE);
851                 }
852                 this.hoveredDrawingElement.onMouseLeave.trigger(point, modifiers);
853             }
854             
855             /* hover the new element, unless it's selected */
856             if (hoveredDrawingElement) {
857                 if (hoveredDrawingElement.getFocusType() === Webgram.DrawingElement.FOCUS_NONE) {
858                     hoveredDrawingElement.setFocusType(Webgram.DrawingElement.FOCUS_HOVERED);
859                 }
860                 hoveredDrawingElement.onMouseEnter.trigger(point, modifiers);
861             }
862             
863             /* set the new hovered element */
864             this.hoveredDrawingElement = hoveredDrawingElement;
865             
866             hoverChanged = true;
867         }
868         
869         if (hoveredControlPoint !== this.hoveredControlPoint) { /* not the same hovered control point */
870             /* remove hover from the currently hovered element */
871             if (this.hoveredControlPoint && this.hoveredControlPoint.getFocusType() === Webgram.ControlPoint.FOCUS_HOVERED) {
872                 this.hoveredControlPoint.setFocusType(Webgram.ControlPoint.FOCUS_NONE);
873                 this.hoveredControlPoint.onMouseLeave.trigger(point, modifiers);
874                 this.hoveredControlPoint.onRemove.unbind(Webgram.DrawingControls.SelectDrawingControl._whenPointRemoved);
875             }
876             
877             /* hover the new element, unless it's selected */
878             if (hoveredControlPoint && hoveredControlPoint.getFocusType() === Webgram.ControlPoint.FOCUS_NONE) {
879                 hoveredControlPoint.setFocusType(Webgram.ControlPoint.FOCUS_HOVERED);
880                 hoveredControlPoint.onMouseEnter.trigger(point, modifiers);
881                 hoveredControlPoint.onRemove.bind(Webgram.DrawingControls.SelectDrawingControl._whenPointRemoved, this);
882             }
883             
884             /* set the new hovered element */
885             this.hoveredControlPoint = hoveredControlPoint;
886             
887             hoverChanged = true;
888         }
889         
890         if (hoveredActionMenuItem !== this.hoveredActionMenuItem) { /* not the same hovered action menu item */
891             /* remove hover from the currently hovered element */
892             if (this.hoveredActionMenuItem && this.hoveredActionMenuItem.getFocusType() === Webgram.ActionMenuItem.FOCUS_HOVERED) {
893                 this.hoveredActionMenuItem.setFocusType(Webgram.ActionMenuItem.FOCUS_NONE);
894                 this.hoveredActionMenuItem.onRemove.unbind(Webgram.DrawingControls.SelectDrawingControl._whenPointRemoved);
895             }
896             
897             /* hover the new CP , unless it's selected */
898             if (hoveredActionMenuItem && hoveredActionMenuItem.getFocusType() === Webgram.ActionMenuItem.FOCUS_NONE) {
899                 hoveredActionMenuItem.setFocusType(Webgram.ActionMenuItem.FOCUS_HOVERED);
900                 hoveredActionMenuItem.onRemove.bind(Webgram.DrawingControls.SelectDrawingControl._whenPointRemoved, this);
901             }
902             
903             /* set the new hovered action menu item */
904             this.hoveredActionMenuItem = hoveredActionMenuItem;
905             
906             hoverChanged = true;
907         }
908         
909         if (hoverChanged) {
910             this.invalidate();
911         }
912         
913         /* decide which cursor to set on the drawing canvas */
914         if (hoveredControlPoint) {
915             return hoveredControlPoint.getCursor();
916         }
917         if (hoveredActionMenuItem) {
918             return hoveredActionMenuItem.getCursor();
919         }
920         else if (hoveredDrawingElement) {
921             return hoveredDrawingElement.getCursor();
922         }
923         else {
924             return 'default';
925         }

1097     _markMultipleSelected: function (drawingElements) {
1098         if (!this._msContainerElement) {
1099             return;
1100         }
1101         
1102         var drawingElement, i;
1103         
1104         if (drawingElements) {
1105             for (i = 0; i < drawingElements.length; i++) {
1106                 drawingElement = drawingElements[i];
1107                 
1108                 if (drawingElement === this._msContainerElement) {
1109                     continue;
1110                 }
1111                 
1112                 if (!drawingElement.isSelectEnabled()) {
1113                     continue;
1114                 }
1115                 
1116                 drawingElement.setFocusType(Webgram.DrawingElement.FOCUS_SELECTED_MULTIPLE);
1117                 drawingElement.onSelect.trigger();
1118             }
1119         }
1120         else {
1121             var msBoundingRectangle = this._msContainerElement.getBoundingRectangle();
1122             for (i = 0; i < this.rootContainer.drawingElements.length; i++) {
1123                 drawingElement = this.rootContainer.drawingElements[i];
1124                 
1125                 if (drawingElement === this._msContainerElement) {
1126                     continue;
1127                 }
1128                 
1129                 if (!drawingElement.isSelectEnabled()) {
1130                     continue;
1131                 }
1132                 
1133                 var boundingRectangle = drawingElement.getBoundingRectangle();
1134                 
1135                 if ((msBoundingRectangle.x1 <= boundingRectangle.x1) &&
1136                         (msBoundingRectangle.x2 >= boundingRectangle.x2) &&
1137                         (msBoundingRectangle.y1 <= boundingRectangle.y1) &&
1138                         (msBoundingRectangle.y2 >= boundingRectangle.y2)) { /* we have a completely hovered element */
1139                     
1140                     drawingElement.setFocusType(Webgram.DrawingElement.FOCUS_SELECTED_MULTIPLE);
1141                     drawingElement.onSelect.trigger();
1142                 }
1143                 else {
1144                     drawingElement.setFocusType(Webgram.DrawingElement.FOCUS_NONE);
1145                     drawingElement.onUnselect.trigger();
1146                     drawingElement.disableShift();
1147                 }
1148             }

1152     /**
1153      * @param {Webgram.Geometry.Point} mouseDownPoint the point where
1154      *  the mouse event that triggered the multiple selection finish occured;
1155      *  the coordinates are relative to the block element of this drawing control.
1156      */
1157     _finishMultipleSelection: function (mouseDownPoint) {
1158         if (!this._msContainerElement) {
1159             return;
1160         }
1161 
1162         /* shrink the selection rectangle */
1163         var minX = Infinity;
1164         var minY = Infinity;
1165         var maxX = -Infinity;
1166         var maxY = -Infinity;
1167         var drawingElement, i;
1168         
1169         /* gather the multiple-selected drawing elements */
1170         var drawingElements = [];
1171         for (i = 0; i < this.rootContainer.drawingElements.length; i++) {
1172             drawingElement = this.rootContainer.drawingElements[i];
1173             
1174             if (drawingElement.getFocusType() !== Webgram.DrawingElement.FOCUS_SELECTED_MULTIPLE) {
1175                 continue;
1176             }
1177             
1178             drawingElements.push(drawingElement);
1179         }
1180         
1181         /* compute the minimum bounding rectangle */
1182         for (i = 0; i < drawingElements.length; i++) {
1183             drawingElement = drawingElements[i];
1184             
1185             var boundingRectangle = drawingElement.getBoundingRectangle();
1186             
1187             if (boundingRectangle.x1 < minX) {
1188                 minX = boundingRectangle.x1;
1189             }
1190             
1191             if (boundingRectangle.y1 < minY) {
1192                 minY = boundingRectangle.y1;
1193             }
1194             
1195             if (boundingRectangle.x2 > maxX) {
1196                 maxX = boundingRectangle.x2;
1197             }
1198             
1199             if (boundingRectangle.y2 > maxY) {
1200                 maxY = boundingRectangle.y2;
1201             }
1202         }
1203         
1204         if (drawingElements.length > 1) { /* more than one selected element, shrink the _msContainerElement */
1205             this._msContainerElement._setTopLeft(new Webgram.Geometry.Point(minX, minY), true);
1206             this._msContainerElement._setBottomRight(new Webgram.Geometry.Point(maxX, maxY), true);
1207             this._msContainerElement._setTopLeft(new Webgram.Geometry.Point(minX, minY), true); /* yeap, this is needed because of the minSize constraints */
1208             
1209             /* remember each element's index */
1210             var index, indexes = [];
1211             for (i = 0; i < drawingElements.length; i++) {
1212                 index = this.rootContainer.getDrawingElementIndex(drawingElements[i]);
1213                 indexes.push(index);
1214             }
1215             this._msContainerElement._drawingElementIndexes = indexes;
1216             
1217             /* place the MSCE right before the first selected element */
1218             this.rootContainer._setDrawingElementIndex(this._msContainerElement, indexes[0]);
1219             
1220             this._msContainerElement.setFocusType(Webgram.DrawingElement.FOCUS_SELECTED);
1221             this.selectedDrawingElement = this._msContainerElement;
1222             
1223             this.onSelectionChange.trigger(drawingElements.slice(0));
1224             
1225             if (mouseDownPoint) {
1226                 var center = this.selectedDrawingElement.getCenter();
1227                 this.selectedDrawingElement._mouseDownPoint = new Webgram.Geometry.Point(mouseDownPoint.x - center.x, mouseDownPoint.y - center.y);
1228             }
1229         }
1230         else if (drawingElements.length === 1) { /* a single selected element, select it directly instead of adding to multiple */
1231             drawingElements[0].setFocusType(Webgram.DrawingElement.FOCUS_SELECTED);
1232             this.selectedDrawingElement = drawingElements[0];
1233             
1234             this.onSelectionChange.trigger([this.selectedDrawingElement]);
1235             
1236             this.rootContainer._remDrawingElement(this._msContainerElement);
1237             this._msContainerElement = null;
1238             
1239             if (mouseDownPoint) {
1240                 var center = this.selectedDrawingElement.getCenter();
1241                 this.selectedDrawingElement._mouseDownPoint = new Webgram.Geometry.Point(mouseDownPoint.x - center.x, mouseDownPoint.y - center.y);
1242             }
1243             
1244             return;
1245         }
1246         else { /* no selected elements, cancel the multiple selection */
1247             this.rootContainer._remDrawingElement(this._msContainerElement);
1248             this._msContainerElement = null;
1249             
1250             return;
1251         }
1252         
1253         /* actually add the drawing elements to the multiple selection block element */
1254         var moveEnabled = true;
1255         var rotateEnabled = true;
1256         var resizeEnabled = true;
1257         var pendingToAdd = [];
1258         for (i = 0; i < this.rootContainer.drawingElements.length; i++) {
1259             var drawingElement = this.rootContainer.drawingElements[i];
1260             
1261             if (drawingElement.getFocusType() !== Webgram.DrawingElement.FOCUS_SELECTED_MULTIPLE) {
1262                 continue;
1263             }
1264             
1265             if (!drawingElement.isMoveEnabled()) {
1266                 moveEnabled = false;
1267             }
1268             
1269             if (!drawingElement.isRotateEnabled()) {
1270                 rotateEnabled = false;
1271             }
1272             
1273             if (drawingElement.isResizeEnabled && !drawingElement.isResizeEnabled()) {
1274                 resizeEnabled = false;
1275             }
1276             
1277             pendingToAdd.push(drawingElement);
1278         }
1279         
1280         for (i = 0; i < pendingToAdd.length; i++) {
1281             pendingToAdd[i]._enterContainerElement(this._msContainerElement);
1282             pendingToAdd[i].setFocusType(Webgram.DrawingElement.FOCUS_SELECTED_MULTIPLE);
1283         }
1284         
1285         if (!moveEnabled) {
1286             this._msContainerElement.setMoveEnabled(false);
1287         }
1288         
1289         if (!rotateEnabled) {
1290             this._msContainerElement.setRotateEnabled(false);
1291         }
1292         
1293         if (!resizeEnabled) {
1294             this._msContainerElement.setResizeEnabled(false);
1295         }
1296     },
1297 
1298     /**
1299      * Destroys the multiple selection by restoring the
1300      * multiple-selected elements to the block element
1301      * of this drawing control. 
1302      * @param {Boolean} temporary indicates if this is a
1303      *  temporary multiple-selection removal, in which case
1304      *  the multiple-selection block element is not removed
1305      */
1306     _remFromMultipleSelection: function (temporary) {
1307         if (!this._msContainerElement) {
1308             return;
1309         }
1310         
1311         var i, pendingToRem = [];
1312         
1313         /* create a copy of the list of children in the multiple selection block element */
1314         for (i = 0; i < this._msContainerElement.drawingElements.length; i++) {
1315             pendingToRem.push(this._msContainerElement.drawingElements[i]);
1316         }
1317         
1318         for (i = 0; i < pendingToRem.length; i++) {
1319             pendingToRem[i]._leaveContainerElement();
1320             
1321             if (temporary) {
1322                 pendingToRem[i].setFocusType(Webgram.DrawingElement.FOCUS_SELECTED_MULTIPLE);
1323             }
1324         }
1325         
1326         var indexes = this._msContainerElement._drawingElementIndexes;
1327         var index = this.rootContainer.getDrawingElementIndex(this._msContainerElement); 
1328         
1329         /* remove the MSCE first */
1330         this.rootContainer._remDrawingElement(this._msContainerElement);
1331         
1332         if (indexes != null) {
1333             /* restore the initial order of DEs */
1334             for (i = 0; i < pendingToRem.length; i++) {
1335                 var drawingElement = pendingToRem[i];
1336                 drawingElement._parent._setDrawingElementIndex(drawingElement, indexes[i]);
1337             }
1338         }
1339         else {
1340             /* place all the DEs right after the MSCE */
1341             for (i = 0; i < pendingToRem.length; i++) {
1342                 var drawingElement = pendingToRem[i];

1408     handleMouseDown: function (point, button, modifiers) {
1409         /* we add this handler because we want to detect clicks on
1410          * an inner element (for deselecting it with the "shift" modifier)
1411          * without enabling the select drawing control on this block element */
1412         
1413         var transformedPoint = this.transformInverse(point);
1414         var hoveredDrawingElement = null;
1415         
1416         for (var i = this.drawingElements.length - 1; i >= 0; i--) {
1417             var drawingElement = this.drawingElements[i];
1418             
1419             if (drawingElement.pointInside(transformedPoint)) {
1420                 hoveredDrawingElement = drawingElement;
1421                 break;
1422             }
1423         }
1424         
1425         /* if clicked on a child drawing element with shift pressed,
1426          * remove the child from the selection */
1427         if (hoveredDrawingElement && modifiers.shift && (this.getRotationAngle() === 0)) {
1428             var selectDrawingControl = this._parent.getActiveDrawingControl();
1429             selectDrawingControl._remFromMultipleSelection(true);
1430             hoveredDrawingElement.setFocusType(Webgram.DrawingElement.FOCUS_NONE);
1431             selectDrawingControl._finishMultipleSelection(point);
1432             selectDrawingControl._mouseDownPoint = null;
1433             
1434             return true;
1435         }
1436     },

1439         /* we override this function because we want to return false
1440          * when the mouse is not over a child of this block element */
1441         
1442         var transformedPoint = this.transformInverse(point);
1443         
1444         for (var i = 0; i < this.drawingElements.length; i++) {
1445             var drawingElement = this.drawingElements[i];
1446             
1447             if (drawingElement.pointInside(transformedPoint)) {
1448                 return true;
1449             }
1450         }
1451         
1452         return false;
1453     },
