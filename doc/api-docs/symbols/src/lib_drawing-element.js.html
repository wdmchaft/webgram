 21 /**
 22  * @class The abstract base class for all drawing elements.
 23  * @param {String} id a given identifier, can be <tt>null</tt>
 24  * @param {Array} points a list of {@link Webgram.Geometry.Point} objects
 25  */
 26 Webgram.DrawingElement = function (id, points) {
 27     if (points == null) {
 28         points = [];
 29     }
 30 
 31     /**
 32      * A reference to the main webgram object, or null if this hasn't been added to a webgram yet.
 33      * @type Webgram
 34      */
 35     this.webgram = null;
 36     
 37     /**
 38      * This field indicates the drawing order among elements of having the same parent.
 39      * It works just like the <tt>z-index</tt> CSS property. <br>
 40      * You should only set this field when constructing the object.
 41      * @type Number
 42      * @default 0
 43      */
 44     this.zIndex = 0;
 45     
 46     /**
 47      * This field specifies the minimum number of shape points that this drawing element needs
 48      * in order to be consistent, i.e. to function properly.

 50      * @type Number
 51      * @default 1
 52      */

235

237      * An event that is triggered whenever the rotation angle of this element changes.<br>
238      * Handlers receive the following arguments: <tt>(angle)</tt>. 
239      * @type Webgram.Event

241     this.onRotate = new Webgram.Event('rotate', this); /* (angle) */
242 
243     /**
244      * An event that is triggered when the user finishes the rotation of this element.

247      * @type Webgram.Event

249     this.onEndRotate = new Webgram.Event('end rotate', this); /* () */
250

252      * An event that is triggered when the user begins changing one of the styles.<br>
253      * Handlers receive no arguments. 
254      * @type Webgram.Event
255      */
256     this.onBeginStyleChange = new Webgram.Event('begin style change', this); /* () */
257 
258     /**
259      * An event that is triggered whenever one of the styles of this element changes.<br>
260      * Handlers receive no arguments. 
261      * @type Webgram.Event
262      */
263     this.onStyleChange = new Webgram.Event('style change', this); /* () */
264 
265     /**
266      * An event that is triggered when the user finishes changing one of the styles.<br>
267      * Handlers receive no arguments. 
268      * @type Webgram.Event
269      */
270     this.onEndStyleChange = new Webgram.Event('end style change', this); /* () */

273      * An event that is triggered when the shape starts to change.<br>

275      * @type Webgram.Event
276      */
277     this.onBeginShapeChange = new Webgram.Event('shape begin change', this); /* () */
278 
279     /**
280      * An event that is triggered whenever the shape changes.<br>
281      * Handlers receive no arguments.
282      * @type Webgram.Event

284     this.onShapeChange = new Webgram.Event('shape change', this); /* () */
285 
286     /**
287      * An event that is triggered when the shape finishes to change.<br>
288      * Handlers receive no arguments. 
289      * @type Webgram.Event

291     this.onEndShapeChange = new Webgram.Event('shape end change', this); /* () */
292     
293     /**
294      * An event that is triggered after the style, the shape
295      * or the rotation angle of this drawing element is modified.
296      * In addition to these, any additional attributes that contribute
297      * to the json form of the element should trigger this event.
298      * This event is linked with the
299      * more general {@link Webgram#onDrawingElementChange} event<br>
300      * Handlers receive no arguments.
301      * @type Webgram.Event
302      */
303     this.onChange = new Webgram.Event('change', this); /* () */
304     
305     /* link the onChange event to the webgram's onDrawingElementChange event */
306     this.onChange.bind(function () {
307         if (this.webgram != null) {
308             this.webgram.onDrawingElementChange.trigger(this);
309         }
310     });
311     
312     /* styles */
313     this._strokeStyle = Webgram.Styles.getStrokeStyle('default');
314     this._fillStyle = Webgram.Styles.getFillStyle('default');
315     this._textStyle = Webgram.Styles.getTextStyle('default');
316     this._guidesStyle = Webgram.Styles.getStrokeStyle('default-guides');
317     this._hoveredDecorationStyle = Webgram.Styles.getStrokeStyle('default-hovered-decoration');
318     this._selectedDecorationStyle = Webgram.Styles.getStrokeStyle('default-selected-decoration');
319     
320     this._shape = new Webgram.Geometry.Poly(points);
321     this._rotationAngle = 0;
322     this._rotationCenter = new Webgram.Geometry.Point(0, 0);
323     
324     this._flippedHorizontally = false;
325     this._flippedVertically = false;
326     this._parent = null;
327     this._id = id;
328     this._mouseDownPoint = null;
329     this._creating = false;
330     this._pasting = false;
331     this._duplicating = false;
332     this._baseRectangle = null;
333     this._shapeChanged = false;
334     this._rotationAngleChanged = false;
335     this._styleChanged = false;
336     this._focusType = Webgram.DrawingElement.FOCUS_NONE;
337     this._jsonFields = [];
338     this._jsonFuncs = [];
339     this._shiftBehaviors = [];
340     
341     this._hoveredControlPointsEnabled = false;
342     this._selectEnabled = true;
343     this._moveEnabled = true;
344     this._snapToAngleEnabled = false;
345     this._snapInternallyEnabled = false;
346     this._snapExternallyEnabled = false;
347     this._snapToGridEnabled = true;
348     this._rotateEnabled = false;
349     this._flipEnabled = true;
350     this._gradientEditEnabled = false;
351     this._shiftEnabled = false;
352     
353     this._controlPoints = [];
354     this._proportionalPoints = [];
355     this._sockets = [];
356     this._actionMenuItems = [];
357     this._parentRelativePositions = [];
358     this._parentRelativePositions = [];
359     this._dependentDrawingElements = [];
360     
361     /* compute the base rectangle, to avoid an initial shape change event */
362     this._baseRectangle = this._computeBaseRectangle();
363 
364     /* initial settings */
365     this.setRotateEnabled(false);
366 };

368 /** Represents the normal focus type of a drawing element, when it's neither hovered, nor selected. */
369 Webgram.DrawingElement.FOCUS_NONE = 0; 
370 /** Represents the hovered focus type of a drawing element. */
371 Webgram.DrawingElement.FOCUS_HOVERED = 1;
372 /** Represents the selected focus type of a drawing element. */
373 Webgram.DrawingElement.FOCUS_SELECTED = 2; 
374 /** Represents the multiple-selected focus type of a drawing element. */
375 Webgram.DrawingElement.FOCUS_SELECTED_MULTIPLE = 3;
376 /* any greater values are user specific */

378 Webgram.DrawingElement.prototype = {
379     /**
380      * Returns the given identifier of this element.
381      * @returns {String} the given identifier of this element.
382      */
383     getId: function () {
384         return this._id;
385     },
386 
387     /**
388      * Sets the given identifier of this element.
389      * @param {String} id the new given identifier
390      */
391     setId: function (id) {
392         this._id = id;
393     },
394     
395     /**
396      * Returns the current focus type of this element.
397      * @returns {Number} one of the Webgram.DrawingElement.FOCUS_* values or a greater, user-defined value
398      */
399     getFocusType: function () {
400         return this._focusType;
401     },
402 
403     /**
404      * Sets the focus type of this element.
405      * @param focusType {Number} one of the Webgram.DrawingElement.FOCUS_* values or a greater, user-defined value
406      */
407     setFocusType: function (focusType) {
408         var changed = false;
409         if (this._focusType !== focusType) {
410             changed = true;
411         }
412         
413         this._focusType = focusType;
414         
415         if (changed) {
416             this.onChangeFocus.trigger();
417         }
418         
419         if (focusType == Webgram.DrawingElement.FOCUS_SELECTED && this._parent) {
420             this._parent.setFocusType(Webgram.DrawingElement.FOCUS_SELECTED);
421         }
422     },
423     
424     /**
425      * Returns a string representation of this element.
426      * The default implementation returns the given identifier enclosed in parentheses.<br><br>
427      * <em>(could be overridden)</em>
428      * @returns {String} the string representation of this element
429      */
430     toString: function () {
431         var text = this.constructor.toString();
432         if (this._id === undefined) {
433             text += '(undefined)';
434         }
435         else if (this._id === null) {
436             text += '(null)';
437         }
438         else if (this._id !== undefined) {
439             text += '("' + this._id + '")';
440         }
441         
442         return text;
443     },
444     
445     /**
446      * Returns the <tt>cursor</tt> CSS property to set when the mouse is over this element.<br><br>
447      * <em>(could be overridden)</em>
448      * @returns {String} the <tt>cursor</tt> CSS property
449      */
450     getCursor: function () {
451         if (this.isMoveEnabled()) {
452             return 'move';
453         }
454         else {
455             return 'default';
456         }
457     },

461     
462     /**
463      * This method is responsible for the way the element looks on the canvas.
464      * Concrete drawing elements should implement this method by calling the various
465      * available drawing primitives (draw* methods), followed by a call to <tt>paint</tt>.<br>
466      * Figures drawn inside this method are affected by the zoom.<br><br> 
467      * <em>(should be overridden)</em>
468      */
469     draw: function () {
470     },
471 
472     /**
473      * This method behaves just like the {@link Webgram.DrawingElement#draw} method,
474      * except for the fact that everything drawn here appears on top of the other elements.<br><br> 
475      * <em>(could be overridden)</em>
476      */
477     drawTop: function () {
478     },
479 
480     /**
481      * This method behaves just like the {@link Webgram.DrawingElement#draw} method,
482      * except for the fact that the figures drawn inside this function are zoom-independent.<br><br> 
483      * <em>(could be overridden)</em>
484      */
485     drawNoZoom: function () {
486     },
487 
488     /**
489      * This method combines the {@link Webgram.DrawingElement#drawTop} method,
490      * with the {@link Webgram.DrawingElement#drawNoZoom} method, resulting in
491      * figures that are on top of the other elements, as well as zoom-independent.<br>
492      * The default implementation draws the decoration and the guides.<br><br> 
493      * <em>(could be overridden)</em>
494      */
495     drawNoZoomTop: function() {
496         this.drawDecoration();
497         
498         if (this.getFocusType() === Webgram.DrawingElement.FOCUS_SELECTED || this.getFocusType() === Webgram.DrawingElement.FOCUS_HOVERED) {
499             this.drawGuides();
500         }
501     },
502     
503     /**
504      * This method draws a decoration around the element.
505      * It's called by the {@link Webgram.DrawingElement#drawNoZoomTop} method,
506      * since we want decorations to be on top and zoom-independent.<br>
507      * The default implementation uses the hovered and selected decoration styles
508      * to draw the decoration, when the element is focused or respectively selected.<br><br> 
509      * <em>(could be overridden)</em>
510      */
511     drawDecoration: function () {
512         switch (this.getFocusType()) {
513             case Webgram.DrawingElement.FOCUS_NONE:
514                 break;
515             
516             case Webgram.DrawingElement.FOCUS_HOVERED:
517                 if (this._hoveredDecorationStyle) {
518                     this.drawPoly(this.getDecorationPoly(), true);
519                     this.paint(this._hoveredDecorationStyle, null);
520                 }
521                 
522                 break;
523             
524             case Webgram.DrawingElement.FOCUS_SELECTED:
525                 if (this._selectedDecorationStyle) {
526                     this.drawPoly(this.getDecorationPoly(), true);
527                     this.paint(this._selectedDecorationStyle, null);
528                 }
529                 
530                 break;
531         }
532     },
533 
534     /**
535      * This method draws the various guides that enhance the editing of the element.
536      * It's called by the {@link Webgram.DrawingElement#drawNoZoomTop} method,
537      * since we want guides to be on top and zoom-independent.<br>
538      * The default implementation uses the guides style to draw the rotation center link line,
539      * as well as the gradient link lines when gradient editing is enabled.<br><br> 
540      * <em>(could be overridden)</em>
541      */
542     drawGuides: function () {
543         if (this.isGradientEditEnabled()) {
544             this.drawGradientLines(this._fillStyle);
545             this.drawGradientLines(this._strokeStyle);
546         }
547         
548         this.drawRotationCenterLine();
549     },
550 
551     /**
552      * This method is responsible for the drawing of the rotation center link line.
553      * It's called by the {@link Webgram.DrawingElement#drawGuides} method.<br><br>
554      * <em>(should not be overridden)</em>
555      */
556     drawRotationCenterLine: function () {
557         this.drawLine(Webgram.Geometry.Point.zero(), this.getRotationCenter());
558         this.paint(this._guidesStyle, null);
559     },
560 
561     /**
562      * This method is responsible for the drawing of the gradient link lines.
563      * It's called by the {@link Webgram.DrawingElement#drawGuides} method.<br><br>
564      * <em>(should not be overridden)</em>
565      */
566     drawGradientLines: function (style) {
567         if (this.getFocusType() !== Webgram.DrawingElement.FOCUS_SELECTED) {
568             return;
569         }
570         
571         if (style == null) {
572             return;
573         }
574         
575         if (style.gradientPoint1 != null) { /* any kind of gradient form enabled */
576             if (style.gradientPoint1._radiusPoint1) { /* radial gradient */
577                 /* draw a line between the gradient point and its radius point */
578                 
579                 this.drawLine(style.gradientPoint1, style.gradientPoint1._radiusPoint1);
580                 this.drawLine(style.gradientPoint1, style.gradientPoint1._radiusPoint2);
581                 this.paint(this._guidesStyle, null);
582             }
583             else { /* linear gradient */
584                 /* draw a line between the gradient points */
585                 
586                 this.drawLine(style.gradientPoint1, style.gradientPoint2);
587                 this.paint(this._guidesStyle, null);
588             }
589         }
590     },
591 
592     /**
593      * This method is responsible for the drawing all the control points of this element.<br><br>
594      * <em>(should not be overridden)</em>
595      * @see Webgram.ControlPoint
596      */
597     drawControlPoints: function () {
598         if ((this._focusType === Webgram.DrawingElement.FOCUS_HOVERED && this.isHoveredControlPointsEnabled()) ||
599                 this._focusType === Webgram.DrawingElement.FOCUS_SELECTED) {
600             
601             for (var i = 0; i < this._controlPoints.length; i++) {
602                 var controlPoint = this._controlPoints[i];
603                 controlPoint.draw();
604             }
605         }
606     },
607 
608     /**
609      * This method is responsible for the drawing all the sockets of this element.<br><br>
610      * <em>(should not be overridden)</em>
611      * @see Webgram.Connectors.Socket
612      */
613     drawSockets: function () {
614         for (var i = 0; i < this._sockets.length; i++) {
615             var socket = this._sockets[i];
616             socket.draw();
617         }
618     },
619 
620     /**
621      * This method is responsible for the drawing all the action menu items of this element.
622      * @see Webgram.ActionMenuItem
623      * <em>(should not be overridden)</em>
624      */
625     drawActionMenuItems: function () {
626         if (this._focusType === Webgram.DrawingElement.FOCUS_SELECTED) {
627             
628             for (var i = 0; i < this._actionMenuItems.length; i++) {
629                 var actionMenuItem = this._actionMenuItems[i];
630                 actionMenuItem.draw();
631             }
632         }
633     },
634 
635     /**
636      * This tells webgram that something about this element has changed
637      * and that a redraw should be issued as soon as possible.<br><br>
638      * <em>(should not be overridden)</em>
639      * @param {Boolean} mini set to <tt>true</tt> if the mini webgram should
640      *  also be redrawn.
641      */
642     invalidateDrawing: function (mini) {
643         if (this.webgram) {
644             this.webgram.invalidateDrawing();
645             if (mini) {
646                 this.webgram.invalidateDrawingMini();
647             }
648         }
649     },

651     /**
652      * The decoration offset is the distance between the border of the element's shape
653      * and the decoration drawn around it. The default implementation returns the webgram
654      * <tt>decorationOffset</tt> setting.<br><br>
655      * <em>(could be overridden)</em>
656      * @returns {Number} the decoration offset for this element
657      */

1538     /**
1539      * Snaps the whole element, or just a single point,
1540      * using the <em>external</em> and <em>grid</em> snapping methods.
1541      * @param {Number} index the index of a point in the shape,
1542      *  or <tt>null</tt>/<tt>undefined</tt> to snap the whole element
1543      */
1544     snap: function (index) {
1545         if (index == null) { /* snap the whole element */
1546             /* find the best point to snap and move the whole shape accordingly */
1547             var i, deltaX, deltaY;
1548             var result, point;
1549             var minDeltaX = Infinity;
1550             var minDeltaY = Infinity;
1551             var minSnapX = null;
1552             var minSnapY = null;
1553             var snapVisualFeedback = {type: null, x: null, y: null, angle: null};
1554 
1555             var center = this.getCenter();
1556             var snappingPoints = this.getSnappingPoints();
1557 
1558             /* first try to snap externally (to siblings) */
1559             if (this.isSnapExternallyEnabled()) {
1560                 for (i = 0; i < snappingPoints.length; i++) {
1561                     var fixedPoints = snappingPoints.slice(0, i).concat(snappingPoints.slice(i + 1));
1562                     point = snappingPoints[i];
1563                     result = this.snapExternally(i, point, fixedPoints, 0, center);
1564                     
1565                     /* if point didn't snap, try the next one */
1566                     if (result == null) {
1567                         continue;
1568                     }
1569                     
1570                     if (result.x != null) {
1571                         deltaX = result.x - point.x;
1572                         if (Math.abs(deltaX) < Math.abs(minDeltaX)) {
1573                             minDeltaX = deltaX;
1574                             minSnapX = result.x;
1575                             snapVisualFeedback.x = result.x;
1576                             snapVisualFeedback.type = 'linear';
1577                         }
1578                     }
1579                     if (result.y != null) {
1580                         deltaY = result.y - point.y;
1581                         if (Math.abs(deltaY) < Math.abs(minDeltaY)) {
1582                             minDeltaY = deltaY;
1583                             minSnapY = result.y;
1584                             snapVisualFeedback.y = result.y;
1585                             snapVisualFeedback.type = 'linear';
1586                         }
1587                     }
1588                 }
1589             }
1590             
1591             var xSnapped = minSnapX != null;
1592             var ySnapped = minSnapY != null;
1593             
1594             /* if at least one of the two axes didn't snap, snap it to grid */
1595             if (this.isSnapToGridEnabled() && (minSnapX == null || minSnapY == null)) {
1596                 for (i = 0; i < snappingPoints.length; i++) {
1597                     point = snappingPoints[i];
1598                     /* avoid snapping the center point to grid */
1599                     if (point.x === center.x && point.y === center.y) {
1600                         continue;
1601                     }
1602                     
1603                     result = this.snapToGrid(point, 0, center);
1604                     
1605                     /* if point didn't snap, try the next one */
1606                     if (result == null) {
1607                         continue;
1608                     }
1609                     
1610                     if (result.x != null && !xSnapped) {
1611                         deltaX = result.x - point.x;
1612                         if (Math.abs(deltaX) < Math.abs(minDeltaX)) {
1613                             minDeltaX = deltaX;
1614                             minSnapX = result.x;
1615                         }
1616                     }
1617                     if (result.y != null && !ySnapped) {
1618                         deltaY = result.y - point.y;
1619                         if (Math.abs(deltaY) < Math.abs(minDeltaY)) {
1620                             minDeltaY = deltaY;
1621                             minSnapY = result.y;
1622                         }
1623                     }
1624                 }
1625             }
1626             
1627             if (minDeltaX < Infinity) {
1628                 this.setShape(this.getShape().getTranslated(minDeltaX, 0));
1629             }
1630             if (minDeltaY < Infinity) {
1631                 this.setShape(this.getShape().getTranslated(0, minDeltaY));
1632             }
1633             
1634             if (snapVisualFeedback.type == null) {
1635                 snapVisualFeedback = null;
1636             }
1637             
1638             this.setSnapVisualFeedback(snapVisualFeedback);
1639         }
1640         else { /* snap just the specified point */
1641             var rotatedShape = this.getShape(true);
1642             this._setTransformedShapePoint(index, rotatedShape.points[index], true);
1643         }
1644     },
