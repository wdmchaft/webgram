 21 /**
 22  * @class This is the base class for all drawing controls.
 23  *  A drawing control handles the mouse and keyboard events
 24  *  received by the webgram canvas. A webgram can have many associated drawing controls
 25  *  but only one of them can be active at once.
 26  * @param {Webgram.DrawingElements.RootContainer} rootContainer the root container element of the webgram
 27  */
 28 Webgram.DrawingControl = function (rootContainer) {
 29     /**
 30      * A drawing control operates directly on a root container element.
 31      * @type Webgram.DrawingElements.RootContainer
 32      */
 33     this.rootContainer = rootContainer;
 34     
 35     /**
 36      * An event that is triggered when the drawing control becomes active.<br>
 37      * Handlers receive no arguments.
 38      * @type Webgram.Event
 39      */
 40     this.onActivate = new Webgram.Event('activate', this); /* () */
 41 
 42     /**
 43      * An event that is triggered when the drawing control becomes inactive.<br>
 44      * Handlers receive no arguments.
 45      * @type Webgram.Event
 46      */
 47     this.onDeactivate = new Webgram.Event('deactivate', this); /* () */
 48 };
 49 
 50 Webgram.DrawingControl.prototype = {
 51     /**
 52      * This tells webgram that something about one of the drawing elements has changed
 53      * and that a redraw should be issued as soon as possible.<br><br>
 54      * <em>(should not be overridden)</em>
 55      * @param {Boolean} mini set to <tt>true</tt> if the mini webgram should
 56      *  also be redrawn.
 57      */
 58     invalidateDrawing: function (mini) {
 59         this.rootContainer.webgram.invalidateDrawing();
 60         if (mini) {
 61             this.rootContainer.webgram.invalidateDrawingMini();
 62         }
 63     },
 64 
 65     /**
 66      * Returns a webgram setting. These settings are global and are available to each
 67      * element added to the webgram instance.<br><br>
 68      * <em>(should not be overridden)</em>
 69      * @param {String} setting the full name of the setting
 70      * @param {any} def the default value to return if the setting is not present
 71      * @returns {any} the value of the setting, or <tt>def</tt> if it is not present
 72      */
 73     getSetting: function (setting, def) {
 74         if (!this.rootContainer) { /* not added yet */
 75             return null;
 76         }
 77         
 78         return this.rootContainer.webgram.getSetting(setting, def);
 79     },
 80     
 81     /**
 82      * Returns the webgram that this drawing control is attached to.
 83      * @returns {Webgram} the webgram that this drawing control is attached to
 84      */
 85     getWebgram: function () {
 86         if (this.rootContainer == null) {
 87             return null; /* not added yet */
 88         }
 89         
 90         return this.rootContainer.webgram;
 91     },
 92     
 93     /**
 94      * Activates the drawing control. If this drawing control is already active,
 95      * this method does nothing. If there's another active drawing control,
 96      * it will be deactivated first.<br><br>
 97      * <em>(should not be overridden)</em>
 98      */
 99     activate: function () {
100         if (this.rootContainer._activeDrawingControl === this) { /* already active */
101             return;
102         }
103         
104         if (this.rootContainer._activeDrawingControl) {
105             this.rootContainer._activeDrawingControl.onDeactivate.trigger();
106         }
107         
108         this.rootContainer._prevDrawingControl = this.rootContainer._activeDrawingControl;
109         this.rootContainer._activeDrawingControl = this;
110         
111         this.onActivate.trigger();
112         
113         /* issue an initial mouse move on the newly activated DC */
114         if (this.rootContainer._lastMousePoint) {
115             var result = this.handleMouseMove(this.rootContainer._lastMousePoint, {});
116             if (result != null) {
117                 this.rootContainer.webgram.htmlElement.style.cursor = result;
118             }
119         }
120         
121         /* make sure the webgram element has canvas */
122         var webgram = this.getWebgram();
123         if (!webgram.hasFocus()) {
124             webgram.htmlElement.focus();
125         }
126     },
127     
128     /**
129      * Deactivates the drawing control. If this drawing control is not active,
130      * this method does nothing.<br><br>
131      * <em>(should not be overridden)</em>
132      */
133     deactivate: function () {
134         if (this.rootContainer._activeDrawingControl !== this) { /* not active */
135             return;
136         }
137         
138         this.rootContainer._activeDrawingControl = null;
139         this.onDeactivate.trigger();
140     },
141     
142     /**
143      * Each drawing control remembers the drawing control that was previously active,
144      * before it became active. This method reactivates the previous drawing control, if any.<br><br>
145      * <em>(should not be overridden)</em>
146      * @returns {Webgram.DrawingControl} the drawing control that was reactivated
147      */
148     activatePrevious: function () {
149         this.deactivate();
150         
151         var prevDrawingControl = this.rootContainer._prevDrawingControl;
152         if (prevDrawingControl == null) { /* no previous DC */
153             return null;
154         }
155         
156         prevDrawingControl.activate();
157         
158         return prevDrawingControl;
159     },
160     
161     keyMatches: function (shortcut, key, modifiers) {
162         if (shortcut == null) {
163             return false;
164         }
165         
166         if (shortcut.indexOf('alt') >= 0) {
167             if (!modifiers.alt) {
168                 return false;
169             }
170         }
171         else {
172             if (modifiers.alt) {
173                 return false;
174             }
175         }
176         
177         if (shortcut.indexOf('ctrl') >= 0) {
178             if (!modifiers.ctrl) {
179                 return false;
180             }
181         }
182         else {
183             if (modifiers.ctrl) {
184                 return false;
185             }
186         }
187 
188         if (shortcut.indexOf('shift') >= 0) {
189             if (!modifiers.shift) {
190                 return false;
191             }
192         }
193         else {
194             if (modifiers.shift) {
195                 return false;
196             }
197         }
198         
199         var index = shortcut.lastIndexOf('-');
200         if (index >= 0) {
201             if (shortcut.substring(index + 1) != key) {
202                 return false;
203             }
204         }
205         else {
206             if (shortcut != key) {
207                 return false;
208             }
209         }
210         
211         return true;
212     },

247      */

258     },

274      * @param {Object} modifiers the modifiers (<tt>alt</tt>, <tt>ctrl</tt> and <tt>shift</tt>)
275      */
276     handleKeyPress: function (key, modifiers) {
277     },
278

280      * Performs the action corresponding to the key down event.<br><br>
281      * <em>(should be overridden)</em>
282      * @param {Number} key the code corresponding to the key that was pressed
283      * @param {Object} modifiers the modifiers (<tt>alt</tt>, <tt>ctrl</tt> and <tt>shift</tt>)
284      */
285     handleKeyDown: function (key, modifiers) {
286     },
287 
288     /**
289      * Performs the action corresponding to the key up event.<br><br>
290      * <em>(should be overridden)</em>
291      * @param {Number} key the code corresponding to the key that was released

293      */
294     handleKeyUp: function (key, modifiers) {
295     },
296 
297     /**
298      * Performs the action corresponding to the focus event.<br><br>
299      * <em>(should be overridden)</em>
300      */
301     handleFocus: function () {
302     },

305      * Performs the action corresponding to the blur (focus lost) event.<br><br>
306      * <em>(should be overridden)</em>
307      */
308     handleBlur: function () {
309     },
310     
311     /**
312      * Performs an action when a drawing element is added to the root container.<br><br>
313      * <em>(should be overridden)</em>
314      * @param {Webgram.DrawingElement} drawingElement the drawing element that was added

316     handleDrawingElementAdd: function (drawingElement) {
317     },
