 21 Webgram.Namespace('Webgram.ControlPoints');
 22 
 23 /**
 24  * @class A control point that controls a simple point.
 25  * @extends Webgram.ControlPoint
 26  * @param {Webgram.Geometry.Point} point the point to control
 27  */
 28 Webgram.ControlPoints.SimpleControlPoint = function (point) {
 29     /**
 30      * The point controlled by this control point.
 31      * @type Webgram.Geometry.Point
 32      */
 33     this.point = point;
 34 
 35     Webgram.ControlPoint.call(this);
 36 };
 37 
 38 Webgram.ControlPoints.SimpleControlPoint.prototype = {
 39     getCursor: function () {
 40         return 'pointer';
 41     },
 42     
 43     computeAnchor: function () {
 44         return this.drawingElement.transformInverse(this.point);
 45     },
 46     
 47     processMove: function (point, vanillaPoint) {
 48         point = this.drawingElement.transformDirect(point);
 49         
 50         this.point.x = point.x;
 51         this.point.y = point.y;
 52     }
 53 };
 54 
 55 Webgram.Class('Webgram.ControlPoints.SimpleControlPoint', Webgram.ControlPoint);
