 24 /**
 25  * @class A control point that controls a gradient point.
 26  * @extends Webgram.ControlPoints.ProportionalControlPoint
 27  * @param {Webgram.Geometry.Point} point the gradient point
 28  * @param {Webgram.Geometry.Point} siblingPoint the other gradient point
 29  * @param {Boolean} isFirst <tt>true</tt> if this controls the first gradient point,
 30  * <tt>false</tt> otherwise
 31  * @param {Object} either a fill style or a stroke style whose gradient is to be controlled
 32  */
 33 Webgram.ControlPoints.GradientControlPoint = function (point, siblingPoint, isFirst, style) {
 34     Webgram.ControlPoints.ProportionalControlPoint.call(this, point);
 35     
 36     this.radius = 9;
 37     
 38     /**
 39      * The other gradient point.
 40      * @type Webgram.Geometry.Point
 41      */
 42     this.siblingPoint = siblingPoint;
 43     
 44     /**
 45      * <tt>true</tt> if this controls the first gradient point, <tt>false</tt> otherwise.
 46      * @type Boolean
 47      */
 48     this.isFirst = isFirst;
 49     
 50     /**
 51      * Either a fill style or a stroke style whose gradient is controlled by this control point.
 52      * @type Object
 53      */
 54     this.style = style;
 55     
 56     this.setSnapToGridEnabled(true);
 57 };
 58 
 59 Webgram.ControlPoints.GradientControlPoint.prototype = {
 60     draw: function () {
 61         var gradientPoint = this.proportionalPoint;
 62         var image;
 63         if (gradientPoint._radiusPoint1 != null) { /* radial gradient */
 64             image = this.getImageStore().get('gradient-point');
 65             if (!image) {
 66                 return;
 67             }
 68 
 69             this.drawImage(image.content, Webgram.Geometry.Point.zero(), image.size, 0);
 70             
 71             var radius = 7;
 72             this.drawArc(Webgram.Geometry.Point.zero(), radius, radius, 0, 2 * Math.PI);
 73             
 74             this.paint(null, Webgram.Styles.createFillStyle({
 75                 'colors': this.style.colors,
 76                 'gradientPoint1': new Webgram.Geometry.Point(0, 0),
 77                 'gradientRadius1': 0,
 78                 'gradientRadius2': radius
 79             }));
 80 
 81             radius = 15;
 82             this.drawLine(new Webgram.Geometry.Point(0, -radius), new Webgram.Geometry.Point(0, radius));
 83             this.paint(this.drawingElement._guidesStyle, null);
 84             this.drawLine(new Webgram.Geometry.Point(-radius, 0), new Webgram.Geometry.Point(radius, 0));
 85             this.paint(this.drawingElement._guidesStyle, null);
 86         }
 87         else { /* linear gradient */
 88             image = this.getImageStore().get('gradient-point');
 89             if (!image) {
 90                 return;
 91             }
 92 
 93             this.drawImage(image.content, Webgram.Geometry.Point.zero(), image.size, 0);
 94             
 95             this.drawArc(Webgram.Geometry.Point.zero(), 7, 7, 0, 2 * Math.PI);
 96             
 97             var color;
 98             if (this.isFirst) {
 99                 color = this.style.colors[0];
100             }
101             else {
102                 color = this.style.colors.slice(-1)[0];
103             }
104             this.paint(null, this._fillStyle.replace({'colors': [color]}));
105         }
106     },
107     
108     processMove: function (point, vanillaPoint) {
109         this.drawingElement.beginStyleChange();
110         
111         Webgram.ControlPoints.ProportionalControlPoint.prototype.processMove.call(this, point, vanillaPoint);
112         
113         var gradientPoint = this.proportionalPoint;
114         if (gradientPoint._radiusPoint1 != null) {
115             var anchor = this.drawingElement.transformInverse(this.getAnchor());
116             var deltaX = point.x - anchor.x;
117             var deltaY = point.y - anchor.y;
118             
119             gradientPoint._radiusPoint1.x += deltaX;
120             gradientPoint._radiusPoint1.y += deltaY;
121             this.drawingElement.saveProportionalPoint(gradientPoint._radiusPoint1);
122     
123             gradientPoint._radiusPoint2.x += deltaX;
124             gradientPoint._radiusPoint2.y += deltaY;
125             this.drawingElement.saveProportionalPoint(gradientPoint._radiusPoint2);
126             
127             if (gradientPoint._radiusControlPoint1 != null) {
128                 gradientPoint._radiusControlPoint1.update();
129                 gradientPoint._radiusControlPoint2.update();
130             }
131         }
132         
133         this.drawingElement.onStyleChange.trigger();
134     }
135 };

140 /**
141  * @class A control point that controls a gradient radius.
142  * @extends Webgram.ControlPoints.ProportionalControlPoint
143  * @param {Webgram.Geometry.Point} point the gradient radius point
144  * @param {Webgram.Geometry.Point} siblingPoint the other gradient point
145  * @param {Boolean} isFirst <tt>true</tt> if this controls the first gradient point,
146  * <tt>false</tt> otherwise
147  * @param {Object} either a fill style or a stroke style whose gradient is to be controlled
148  */
149 Webgram.ControlPoints.GradientRadiusControlPoint = function (point, gradientPoint, isFirst, style) {
150     Webgram.ControlPoints.ProportionalControlPoint.call(this, point);
151     
152     this.radius = 9;
153     
154     /**
155      * the gradient radius point
156      * @type Webgram.Geometry.Point
157      */
158     this.gradientPoint = gradientPoint;
159     
160     /**
161      * <tt>true</tt> if this controls the first gradient point, <tt>false</tt> otherwise.
162      * @type Boolean
163      */
164     this.isFirst = isFirst;
165     
166     /**
167      * Either a fill style or a stroke style whose gradient is to be controlled
168      * @type Object
169      */
170     this.style = style;
171 };
172 
173 Webgram.ControlPoints.GradientRadiusControlPoint.prototype = {
174     draw: function () {
175         var image = this.getImageStore().get('gradient-point');
176         if (!image) {
177             return;
178         }
179 
180         this.drawImage(image.content, Webgram.Geometry.Point.zero(), image.size, 0);
181         
182         this.drawArc(Webgram.Geometry.Point.zero(), 7, 7, 0, 2 * Math.PI);
183         
184         var color;
185         if (this.isFirst) {
186             color = this.style.colors[0];
187         }
188         else {
189             color = this.style.colors.slice(-1)[0];
190         }
191         this.paint(null, this._fillStyle.replace({'colors': [color]}));
192     },
193     
194     snap: function (point) {
195         var gradientPoint = this.style.gradientPoint1;
196         gradientPoint = this.drawingElement.transformDirect(gradientPoint);
197         var distance = gradientPoint.getDistanceTo(point);
198         var snapDistance = this.getSetting('snapDistance');
199         if (snapDistance == null) {
200             return point;
201         }
202         
203         if (distance < 10) {
204             return gradientPoint;
205         }
206         else {
207             return point;
208         }
209     },
210     
211     processMove: function (point, vanillaPoint) {
212         this.drawingElement.beginStyleChange();
213         
214         Webgram.ControlPoints.ProportionalControlPoint.prototype.processMove.call(this, point, vanillaPoint);
215         
216         this.drawingElement.onStyleChange.trigger();
217     }
218 };
