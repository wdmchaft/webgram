 24 /**
 25  * @class A special control point that is capable to drag along
 26  * the end points connected to it.
 27  * @extends Webgram.ControlPoint
 28  * @see Webgram.Connectors.EndPoint
 29  * @param {Function} anchorFunc a function used to compute the anchor of the socket;
 30  * it's called with the drawing element as <tt>this</tt> and receives the socket as argument
 31  */
 32 Webgram.Connectors.Socket = function (anchorFunc) {
 33     Webgram.ControlPoint.call(this);
 34     
 35     this.radius = 10;
 36 
 37     this._endPoints = [];
 38     this._anchorFunc = anchorFunc;
 39     
 40     /**
 41      * An event that is triggered when an end point is connected to this socket.<br>
 42      * Handlers receive the following arguments: <tt>(endPoint)</tt>. 
 43      * @type Webgram.Event
 44      */
 45     this.onConnect = new Webgram.Event('connect', this); /* (endPoint) */
 46 
 47     /**
 48      * An event that is triggered when an end point is disconnected from this socket.<br>
 49      * Handlers receive the following arguments: <tt>(endPoint)</tt>. 
 50      * @type Webgram.Event
 51      */
 52     this.onDisconnect = new Webgram.Event('disconnect', this); /* (endPoint) */
 53 };
 54 
 55 Webgram.Connectors.Socket.prototype = {
 56     draw: function () {
 57         var image;
 58         var alpha = 1;
 59         
 60         if (this.getFocusType() !== Webgram.ControlPoint.FOCUS_HOVERED) {
 61             return;
 62         }
 63         
 64         image = this.getImageStore().get('socket');
 65         if (!image) {
 66             return;
 67         }
 68         
 69         this.drawImage(image.content, Webgram.Geometry.Point.zero(), image.size, 0, alpha);
 70     },
 71     
 72     computeAnchor: function () {
 73         var anchor = this._anchorFunc.call(this.drawingElement, this);
 74         
 75         return this.drawingElement.transformDirect(anchor);
 76     },
 77     
 78     update: function () {
 79         Webgram.ControlPoint.prototype.update.call(this);
 80 
 81         /* (indirectly) move all endpoints connected to this socket */
 82         for (var i = 0; i < this._endPoints.length; i++) {
 83             var endPoint = this._endPoints[i];
 84             endPoint.update();
 85             endPoint.movePoint(this.getAnchor());
 86         }
 87     },
 88     
 89     /**
 90      * Tells whether this socket has a connected end point or not.
 91      * @returns {Boolean} <tt>true</tt> if it's connected, <tt>false</tt> otherwise
 92      */
 93     isConnected: function () {
 94         return this._endPoints.length > 0;
 95     },
 96     
 97     /**
 98      * Connects an end point to this socket.<br><br>
 99      * <em>(should not be overridden)</em>
100      * @param {Webgram.Connectors.EndPoint} endPoint the end point to connect
101      */
102     connect: function (endPoint) {
103         this._endPoints.push(endPoint);
104         endPoint.socket = this;
105         endPoint.onConnect.trigger(this);
106         endPoint.movePoint(this.getAnchor());
107         
108         this.onConnect.trigger(endPoint);
109         
110         var connector = endPoint.drawingElement;
111         connector.onConnect.trigger(endPoint, this);
112         
113         if (connector.webgram) {
114             connector.webgram.onDrawingElementInteract.trigger(connector, this.drawingElement, 'connect');
115         }
116     },
117     
118     /**
119      * Disconnects an end point from this socket.<br><br>
120      * <em>(should not be overridden)</em>
121      * @param {Webgram.Connectors.EndPoint} endPoint the end point to disconnect
122      */
123     disconnect: function (endPoint) {
124         if (endPoint != null) {
125             var index = Webgram.Utils.indexOf(this._endPoints, endPoint);
126             
127             if (index === -1) { /* not connected to this endPoint */
128                 return;
129             }
130             
131             this.onDisconnect.trigger(endPoint);
132             endPoint.onDisconnect.trigger(this);
133             endPoint.socket = null;
134             this._endPoints.splice(index, 1);
135             
136             var connector = endPoint.drawingElement;
137             connector.onDisconnect.trigger(endPoint, this);
138         }
139         else {
140             this.onDisconnect.trigger(endPoint);
141             
142             for (var i = 0; i < this._endPoints.length; i++) {
143                 var endPoint = this._endPoints[i];
144                 endPoint.onDisconnect.trigger(this);
145                 endPoint.socket = null;
146 
147                 var connector = endPoint.drawingElement;
148                 connector.onDisconnect.trigger(endPoint, this);
149             }
150             
151             this._endPoints = [];
152         }
153     
154         if (connector.webgram) {
155             connector.webgram.onDrawingElementInteract.trigger(connector, this.drawingElement, 'disconnect');
156         }
157     },
158     
159     /**
160      * Tells whether an end point can be connected to this socket.
161      * The default implementation accepts all end points.<br><br>
162      * <em>(could be overridden)</em>
163      * @param {Webgram.Connectors.EndPoint} endPoint the end point to be tested
164      * @returns {Boolean} <tt>true</tt> if the end point can be connected to this socket, <tt>false</tt> otherwise
165      */
166     accepts: function (endPoint) {
167         return true;
168     }
169 };
170 
171 Webgram.Class('Webgram.Connectors.Socket', Webgram.ControlPoint);
