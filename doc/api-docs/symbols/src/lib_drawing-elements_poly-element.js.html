 24 /**
 25  * @class A base class for all the polygonal drawing elements.
 26  * @extends Webgram.DrawingElement
 27  * @param {String} id a given identifier, can be <tt>null</tt>
 28  * @param {Array} points a list of {@link Webgram.Geometry.Point} objects
 29  * that form the shape of this polygonal element 
 30  */
 31 Webgram.DrawingElements.PolyElement = function (id, points) {
 32     Webgram.DrawingElement.call(this, id, points);
 33 
 34     /**
 35      * <tt>true</tt> if the polygonal element is a closed shape, <tt>false</tt> otherwise.
 36      * For example, a <em>polygon</em> is closed, while a <em>polyline</em> is not.
 37      * @type Boolean
 38      */
 39     this.closed = false; /* a polygon is closed while a polyline is not */
 40     
 41     /**
 42      * The maximal number of points that the polygonal element is allowed to have.
 43      * @type Number
 44      */
 45     this.maxPointCount = Infinity;
 46     
 47     this._hoveredPoint = null;
 48     this._pendingCreatePoint = false;
 49     
 50     this.setHoveredControlPointsEnabled(true);
 51     this.setRotateEnabled(false);
 52     this.setAddRemovePointsEnabled(true);
 53     this.setEditEnabled(true);
 54 
 55     
 56     /* events */
 57     
 58     /**
 59      * An event that is triggered when a poly point is added to this polygonal element.<br>
 60      * Handlers receive the following arguments: <tt>(index, point)</tt> 
 61      * @type Webgram.Event
 62      */
 63     this.onAddPoint = new Webgram.Event('add point', this); /* (index, point) */
 64 
 65     /**
 66      * An event that is triggered when a poly point is removed from this polygonal element.<br>
 67      * Handlers receive the following arguments: <tt>(index, point)</tt> 
 68      * @type Webgram.Event
 69      */
 70     this.onRemovePoint = new Webgram.Event('remove point', this); /* (index, point) */
 71 
 72     this.onKeyUp.bind(function (key, modifiers) {
 73         if (key === 16) { /* shift */
 74             this._markHoveredPoint(null);
 75             
 76             return false; /* yes, false; we want to continue the event propagation */
 77         }
 78     });
 79     
 80     this.onMouseDown.bind(function (point, button, modifiers) {
 81         if (modifiers.shift && this.isAddRemovePointsEnabled()) {
 82             this._addHoveredPoint();
 83         }
 84     });
 85 
 86     this.onMouseMove.bind(function (point, modifiers) {
 87         if (modifiers.shift && this.isAddRemovePointsEnabled() && this.getPolyControlPoints().length < this.maxPointCount) {
 88             this._markHoveredPoint(point);
 89         }
 90         else {
 91             this._markHoveredPoint(null);
 92         }
 93     });
 94 
 95     this.onMouseLeave.bind(function (modifiers) {
 96         this._markHoveredPoint(null);
 97     });
 98 };

100 Webgram.DrawingElements.PolyElement.prototype = {
101     getCursor: function () {
102         if (this._hoveredPoint) {
103             return 'pointer';
104         }
105         
106         return Webgram.DrawingElement.prototype.getCursor.call(this);
107     },

109     drawNoZoom: function () {
110         Webgram.DrawingElement.prototype.drawNoZoom.call(this);
111         
112         if (this.isAddRemovePointsEnabled() &&
113                 (this.getFocusType() === Webgram.DrawingElement.FOCUS_HOVERED ||
114                  this.getFocusType() === Webgram.DrawingElement.FOCUS_SELECTED)) {
115             
116             this.drawHoveredPoint();
117         }
118     },
119     
120     drawDecoration: function () {
121         if (this.getShape().points.length == 2) {
122             switch (this.getFocusType()) {
123                 case Webgram.DrawingElement.FOCUS_NONE:
124                     break;
125                 
126                 case Webgram.DrawingElement.FOCUS_HOVERED:
127                     if (this._hoveredDecorationStyle) {
128                         this.drawPoly(this.getLineTolerancePoly(), true);
129                         this.paint(this._hoveredDecorationStyle, null);
130                     }
131                     
132                     break;
133                 
134                 case Webgram.DrawingElement.FOCUS_SELECTED:
135                     if (this._selectedDecorationStyle) {
136                         this.drawPoly(this.getLineTolerancePoly(), true);
137                         this.paint(this._selectedDecorationStyle, null);
138                     }
139                     
140                     break;
141             }
142         }
143         else {
144             Webgram.DrawingElement.prototype.drawDecoration.call(this);
145         }
146     },

148     /**
149      * Draws the point to be created, when using the
150      * interactive poly point creation.
151      */
152     drawHoveredPoint: function () {
153         if (!this._hoveredPoint) {
154             return;
155         }
156         
157         var image = this.getImageStore().get('poly-point');
158         if (!image) {
159             return;
160         }
161         
162         this.drawImage(image.content, this._hoveredPoint, image.size, 0, 0.5);
163     },

166     /* shape-related methods */
167     
168     pointInside: function (point) {
169         if (this.getShape().points.length <= 1) { /* a single point or nothing - either way the answer is no */
170             return false;
171         }
172         
173         if (this.closed) {
174             return this.getShape(true).pointInside(point);
175         }
176         else {
177             return this.transformDirect(this.getLineTolerancePoly()).pointInside(point);
178         }
179     },
180 
181     /**
182      * Returns the tolerance (the thickness) of the edges of this element.
183      * This tolerance is used to determine if a point is over the edges or not.
184      * Zooming affects the value of this tolerance.<br><br>
185      * <em>(could be overridden)</em>
186      * @returns {Number} the line tolerance
187      */
188     getLineTolerance: function () {
189         var strokeStyle = this.getStrokeStyle();
190         var distance = Math.round(strokeStyle.lineWidth / 2) + 5 / this.getZoomFactor();
191         
192         return distance;
193     },
194         
195     /**
196      * Returns a polygon that wraps all the edges of this element.
197      * This polygon is directly used to determine if a point is over
198      * the edges of this element.<br><br>
199      * <em>(should not be overridden)</em>
200      * @returns {Webgram.Geometry.Poly} the line tolerance polygon
201      */
202     getLineTolerancePoly: function () {
203         var lineTolerance = this.getLineTolerance();
204         
205         if (this.getShape().points.length === 0) {
206             return null;
207         }
208         else if (this.getShape().points.length === 1) {
209             var point = this.translateInverse(this.getShapePoint(0));
210             return new Webgram.Geometry.Poly([
211                 new Webgram.Geometry.Point(point.x - lineTolerance, point.y - lineTolerance),
212                 new Webgram.Geometry.Point(point.x + lineTolerance, point.y - lineTolerance),
213                 new Webgram.Geometry.Point(point.x + lineTolerance, point.y + lineTolerance),
214                 new Webgram.Geometry.Point(point.x - lineTolerance, point.y + lineTolerance)
215             ]);
216         }
217         else if (this.getShape().points.length === 2) {
218             /* just two points */
219             var p1 = this.translateInverse(this.getShapePoint(0));
220             var p2 = this.translateInverse(this.getShapePoint(1));
221             var alpha = Math.atan2(p2.y - p1.y, p2.x - p1.x);
222             var len = lineTolerance * 1.4142135623730951; /* sqrt(2); */
223             
224             var a = p1.getPointAt(len, -3 * Math.PI / 4 + alpha);
225             var b = p2.getPointAt(len, -Math.PI / 4 + alpha);
226             var c = p2.getPointAt(len, Math.PI / 4 + alpha);
227             var d = p1.getPointAt(len, 3 * Math.PI / 4 + alpha);
228             
229             return new Webgram.Geometry.Poly([a, b, c, d]);
230         }
231         else {
232             /* more than two points */
233             var polyPoints = [];
234             var points = this.getDrawingPoly().points;
235             var len = lineTolerance * 1.4142135623730951; /* sqrt(2); */
236             var pi4 = Math.PI / 4;
237             var p1, p2, p3;
238             var alpha, alpha1, alpha2;
239             var radius;
240             
241             for (var i = 0; i < points.length - 1; i++) {
242                 p1 = points[i];
243                 p2 = points[i + 1];
244                 
245                 if (i === 0) {
246                     alpha1 = Webgram.Utils.normalizeAngle(p1.getAngleTo(p2));
247                     polyPoints.push(p1.getPointAt(len, -3 * pi4 + alpha1));
248                 }
249                 
250                 if (i === points.length - 2) {
251                     alpha1 = Webgram.Utils.normalizeAngle(p1.getAngleTo(p2));
252                     polyPoints.push(p2.getPointAt(len, -pi4 + alpha1));
253                 }
254                 else {
255                     p3 = points[i + 2];
256                     alpha1 = Webgram.Utils.normalizeAngle(p2.getAngleTo(p1));
257                     alpha2 = Webgram.Utils.normalizeAngle(p2.getAngleTo(p3));
258                     radius = lineTolerance + (len - lineTolerance) * Math.abs(Math.sin(alpha2 - alpha1));
259                     alpha = (alpha1 + alpha2) / 2;
260                     if (alpha1 > alpha2) {
261                         alpha += Math.PI;
262                     }
263                     polyPoints.push(p2.getPointAt(radius, alpha));
264                 }
265             }
266             
267             for (var i = points.length - 2; i >= 0; i--){
268                 p1 = points[i];
269                 p2 = points[i + 1];
270                 
271                 if (i === points.length - 2) {
272                     alpha1 = Webgram.Utils.normalizeAngle(p1.getAngleTo(p2));
273                     polyPoints.push(p2.getPointAt(len, pi4 + alpha1));
274                 }
275                 else {
276                     p3 = points[i + 2];
277                     alpha1 = Webgram.Utils.normalizeAngle(p2.getAngleTo(p1));
278                     alpha2 = Webgram.Utils.normalizeAngle(p2.getAngleTo(p3));
279                     radius = lineTolerance + (len - lineTolerance) * Math.abs(Math.sin(alpha2 - alpha1));
280                     alpha = (alpha1 + alpha2) / 2 + Math.PI;
281                     if (alpha1 > alpha2) {
282                         alpha -= Math.PI;
283                     }
284                     polyPoints.push(p2.getPointAt(radius, alpha));
285                 }
286                 
287                 if (i === 0) {
288                     alpha1 = Webgram.Utils.normalizeAngle(p1.getAngleTo(p2));
289                     polyPoints.push(p1.getPointAt(len, 3 * pi4 + alpha1));
290                 }
291             }
292             
293             return new Webgram.Geometry.Poly(polyPoints);
294         }
295     },

351     /**
352      * Returns the currently selected point index among the points
353      * of this polygonal element.
354      * @returns {Number} the selected point index, or <tt>-1</tt> if no point is selected
355      */
356     getSelectedPointIndex: function () {
357         var polyControlPoints = this.getPolyControlPoints();
358         
359         for (var i = 0; i < polyControlPoints.length; i++) {
360             var controlPoint = polyControlPoints[i];
361             
362             if (controlPoint.getFocusType() === Webgram.ControlPoint.FOCUS_SELECTED) {
363                 return controlPoint.polyPointIndex;
364             }
365         }
366         
367         return -1;
368     },

370     /**
371      * Sets the selected point of this polygonal element.
372      * @param {Number} index the index of the new selected point
373      */
374     setSelectedPointIndex: function (index) {
375         var polyControlPoints = this.getPolyControlPoints();
376         
377         for (var i = 0; i < polyControlPoints.length; i++) {
378             var controlPoint = polyControlPoints[i];
379             
380             if (controlPoint.getFocusType() === Webgram.ControlPoint.FOCUS_SELECTED) {
381                 controlPoint.getFocusType(Webgram.ControlPoint.FOCUS_NONE); 
382             }
383             
384             if (i === index) {
385                 controlPoint.getFocusType(Webgram.ControlPoint.FOCUS_SELECTED);
386             }
387         }

408     /* creation */
409     
410     beginCreate: function (point) {
411         this.addPoint(0, point);
412         if (this.maxPointCount > 1) {
413             this._pendingCreatePoint = true;
414             this.addPoint(0, point);
415         }
416         
417         return true; /* accept creation */
418     },
419     
420     continueCreate: function (point, size, mouseDown, click) {
421         /* move the temporary point along the mouse */
422         var controlPoint = this.getPolyControlPointByIndex(this.getShape().points.length - 1);
423         if (controlPoint != null) { /* a control point has been created for the last point */
424             controlPoint.move(point, true);
425         }
426         else { /* no control point has been created, move the shape point*/
427             var shape = this.getShape();
428             this.setShapePoint(shape.points.length - 1, point);
429         }
430         
431         if (click) {
432             if (controlPoint != null) {
433                 controlPoint.onEndMove.trigger();
434             }
435 
436             /* stop the creation if we've got enough points */
437             if (this.getShape().points.length >= this.maxPointCount) {
438                 this._pendingCreatePoint = false;
439                 return false; /* stop */
440             }
441             
442             this.addPoint(this.getShape().points.length, point);
443             
444             controlPoint = this.getPolyControlPointByIndex(this.getShape().points.length - 1);
445             if (controlPoint != null) {
446                 controlPoint.move(point, true);
447             }
448         }
449         
450         return true; /* continue */
451     },
452     
453     endCreate: function () {
454         if (this._pendingCreatePoint) {
455             this._pendingCreatePoint = false;
456             
457             var controlPoint = this.getPolyControlPointByIndex(this.getShape().points.length - 1);
458             if (controlPoint != null) { /* we have a control point for this point */
459                 controlPoint.onEndMove.trigger();
460                 this.remPoint(controlPoint.polyPointIndex); /* remove the temporary poly point */
461             }
462             else { /* no control point for this point, remove the shape point */
463                 this.remPoint(this.getShape().points.length - 1);
464             }
465         }
466         
467         if (this.getShape().points.length < this.minPointCount) {
468             return false; /* failed */
469         }
470         
471         return true; /* succeeded */
472     },

477     shapeFromJson: function (json) {
478         var i, points = [];
479         for (i = 0; i < json.length; i++) {
480             var point = json[i];
481             var newPoint = new Webgram.Geometry.Point(point.x, point.y);
482             
483             points.push(newPoint);
484         }
485         
486         /* add, remove and change the points according to the shape changes */
487         if (points.length > this.getShape().points.length) { /* points were added */
488             for (i = 0; i < this.getShape().points.length; i++) {
489                 this._setTransformedShapePoint(i, points[i]);
490             }
491             
492             for (i = this.getShape().points.length; i < points.length; i++) {
493                 this._addPoint(this.getShape().points.length, points[i]);
494             }
495         }
496         else { /* points were removed or just nothing was added */
497             for (i = 0; i < points.length; i++) {
498                 this._setTransformedShapePoint(i, points[i]);
499             }
500             
501             while (this.getShape().points.length > points.length) {
502                 this._remPoint(this.getShape().points.length - 1);
503             }
504         }
505         
506         this._recreatePolyControlPoints();
507     },

614      * Enables or disables the editing of points for this element.
615      * @param {Boolean} enabled <tt>true</tt> to enable the feature, <tt>false</tt> to disable it
616      */
617     setEditEnabled: function (enabled) {
618         if (enabled) {
619             if (this._editEnabled) {
620                 return;
621             }
622             
623             this._recreatePolyControlPoints();
624             
625             this._editEnabled = true;
626             this._selectedDecorationStyle = Webgram.Styles.getStrokeStyle('default-selected-decoration');
627         }
628         else {
629             if (!this._editEnabled) {
630                 return;
631             }
632             
633             var polyControlPoints = this.getPolyControlPoints();
634             for (var i = 0; i < polyControlPoints.length; i++) {
635                 this.remControlPoint(polyControlPoints[i]);
636             }
637             
638             this._editEnabled = false;
639             this._selectedDecorationStyle = this._hoveredDecorationStyle;
640         }

643     _snapShapePoint: function (index, point, fixedPoints, rotationAngle, center) {
644         /* if the requested point is marked for removal,
645          * don't bother snapping it, just return it as it is */
646         var markedForRemovalPointIndex = this.getMarkedForRemovalPointIndex();
647         if (markedForRemovalPointIndex === index) {
648             return {
649                 point: point,
650                 snapVisualFeedback: null,
651                 xSnapped: false,
652                 ySnapped: false
653             };
654         }
655         
656         return Webgram.DrawingElement.prototype._snapShapePoint.call(this, index, point, fixedPoints, rotationAngle, center);
657     },

659     _snapToAngleInternally: function (index, point, rotationAngle, center) {
660         function getLinesIntersection(point1, angle1, point2, angle2) {
661             angle1 = Webgram.Utils.normalizeAngle(angle1);
662             angle2 = Webgram.Utils.normalizeAngle(angle2);
663             
664             if (Math.abs(angle1 - angle2) < 0.000001) {
665                 if (point1.equals(point2)) {
666                     return point1;
667                 }
668                 else {
669                     return null;
670                 }
671             }
672             else {
673                 var m1, n1, m2, n2, x, y;
674                 /* the special 90 deg for the first line */
675                 if (Math.abs(angle1 - Math.PI / 2) < 0.000001) {
676                     m2 = Math.tan(angle2);
677                     n2 = point2.y - m2 * point2.x;
678                     x = point1.x;
679                     y = m2 * x + n2;
680                 }
681                 /* the special 90 deg for the second line */
682                 else if (Math.abs(angle2 - Math.PI / 2) < 0.000001) {
683                     m1 = Math.tan(angle1);
684                     n1 = point1.y - m1 * point1.x;
685                     x = point2.x;
686                     y = m1 * x + n1;
687                 }
688                 /* none of them is at 90 deg */
689                 else {
690                     m1 = Math.tan(angle1);
691                     n1 = point1.y - m1 * point1.x;
692                     m2 = Math.tan(angle2);
693                     n2 = point2.y - m2 * point2.x;
694                     x = (n1 - n2) / (m2 - m1);
695                     y = (n1 * m2 - n2 * m1) / (m2 - m1);
696                 }
697 
698                 return new Webgram.Geometry.Point(x, y);
699             }
700         }
701         
702         var points = this.getShape().points;
703         if (points.length < 2) {
704             return;
705         }
706 
707         var snapAngle = this.getSetting('snapAngle');
708         var snapAngleThreshold = this.getSetting('snapAngleThreshold');
709         
710         if (snapAngle == null || snapAngleThreshold == null) {
711             return;
712         }
713         
714         /* gather previous points */
715         var prevPoints = [];
716         if (index >= 2) {
717             prevPoints.push(points[index - 1]);
718             prevPoints.push(points[index - 2]);
719         }
720         else if (index == 1) {
721             prevPoints.push(points[0]);
722             if (this.closed && points.length > 2) {
723                 prevPoints.push(points[points.length - 1]);
724             }
725         }
726         else {  /* if (index == 0) */
727             if (this.closed && points.length > 2) {
728                 prevPoints.push(points[points.length - 1]);
729                 prevPoints.push(points[points.length - 2]);
730             }
731         }
732 
733         /* gather next points */
734         var nextPoints = [];
735         if (index <= points.length - 3) {
736             nextPoints.push(points[index + 1]);
737             nextPoints.push(points[index + 2]);
738         }
739         else if (index == points.length - 2) {
740             nextPoints.push(points[points.length - 1]);
741             if (this.closed && points.length > 2) {
742                 nextPoints.push(points[0]);
743             }
744         }
745         else {  /* if (index == points.length - 1) */
746             if (this.closed && points.length > 2) {
747                 nextPoints.push(points[0]);
748                 nextPoints.push(points[1]);
749             }
750         }
751         
752         /* all the angles in angle sets are treated modulo PI */
753         
754         var i, j, m, n, angle, offsetAngle, snappedPoint;
755         
756         /* compute previous point and angle sets */
757         var prevAngleSet = [];
758         var prevPoint = null;
759         if (prevPoints.length > 0) {
760             prevPoint = prevPoints[0];
761             offsetAngle = 0;
762             if (prevPoints.length == 2) {
763                 offsetAngle = prevPoint.getAngleTo(prevPoints[1]);
764             }
765             
766             angle = 0;
767             while (angle < Math.PI) {
768                 prevAngleSet.push(Webgram.Utils.normalizeAngle(angle + offsetAngle) % Math.PI);
769                 /* add the round angle relative to the system */
770                 if (!Webgram.Utils.angleMultipleOf(offsetAngle + rotationAngle, snapAngle)) {
771                     prevAngleSet.push(Webgram.Utils.normalizeAngle(angle - rotationAngle) % Math.PI);
772                 }
773                 angle += snapAngle;
774             }
775         }
776 
777         /* compute next point and angle sets */
778         var nextAngleSet = [];
779         var nextPoint = null;
780         if (nextPoints.length > 0) {
781             nextPoint = nextPoints[0];
782             offsetAngle = 0;
783             if (nextPoints.length == 2) {
784                 offsetAngle = nextPoint.getAngleTo(nextPoints[1]);
785             }
786             
787             angle = 0;
788             while (angle < Math.PI) {
789                 nextAngleSet.push(Webgram.Utils.normalizeAngle(angle + offsetAngle) % Math.PI);
790                 /* add the round angle relative to the system */
791                 if (!Webgram.Utils.angleMultipleOf(offsetAngle + rotationAngle , snapAngle)) {
792                     nextAngleSet.push(Webgram.Utils.normalizeAngle(angle - rotationAngle) % Math.PI);
793                 }
794                 angle += snapAngle;
795             }
796         }
797         
798         /* if the two angle sets are available, compute the solutions
799          * based on intersections of lines at given angles */
800         var solution, solutions = [];
801         if (prevAngleSet.length > 0 && nextAngleSet.length > 0) {
802             /* consider the straight line between the two points as a potential solution */
803             if (Math.abs(prevPoint.x - nextPoint.x) < 0.000001) { /* special 90 deg case */
804                 solution = {x: null, y: null, m: Infinity, n: Infinity, x0: prevPoint.x};
805             }
806             else {
807                 m = (prevPoint.y - nextPoint.y) / (prevPoint.x - nextPoint.x);
808                 n = prevPoint.y - m * prevPoint.x;
809                 
810                 solution = {x: null, y: null, m: m, n: n, x0: null};
811             }
812             
813             solutions.push(solution);
814             
815             for (i = 0; i < prevAngleSet.length; i++) {
816                 for (j = 0; j < nextAngleSet.length; j++) {
817                     if (Math.abs(prevAngleSet[i] - nextAngleSet[j]) < 0.000001) {
818                         /* this solution is in fact the one given by a straight line
819                          * between the prev and next points */
820                         continue;
821                     }
822                     
823                     var intersectionPoint = getLinesIntersection(
824                             prevPoint, prevAngleSet[i], nextPoint, nextAngleSet[j]);
825                     
826                     if (intersectionPoint != null) {
827                         solution = {x: intersectionPoint.x, y: intersectionPoint.y, m: null, n: null, x0: null};
828                         solutions.push(solution);
829                     }
830                 }
831             }
832         }
833         
834         /* compute the best match coming from the intersection of the lines at given angles */
835         var angle1, angle2, minAngle = Infinity;
836         var dist1, dist2, dist3, cosVal;
837         var bestSolution = null;
838         for (i = 0; i < solutions.length; i++) {
839             solution = solutions[i];
840 
841             if (solution.m != null) { /* the straight line between prevPoint and nextPoint */
842                 /* apply the law of cosines to determine the angle between the previous point, point and next point */
843                 dist1 = point.getDistanceTo(prevPoint);
844                 dist2 = point.getDistanceTo(nextPoint);
845                 dist3 = nextPoint.getDistanceTo(prevPoint);
846                 
847                 /* compute the cosine value and bound it to -1..1 */
848                 cosVal = (dist1 * dist1 + dist2 * dist2 - dist3 * dist3) / (2 * dist1 * dist2);
849                 if (cosVal > 1) {
850                     cosVal = 1;
851                 }
852                 if (cosVal < -1) {
853                     cosVal = -1;
854                 }
855 
856                 angle = Math.acos(cosVal);
857                 angle = Math.PI - angle;
858             }
859             else { /* a point */
860                 /* apply the law of cosines twice to determine the differences in angles */
861                 snappedPoint = new Webgram.Geometry.Point(solution.x, solution.y);
862                 dist3 = point.getDistanceTo(snappedPoint);
863                 
864                 dist1 = point.getDistanceTo(prevPoint);
865                 dist2 = snappedPoint.getDistanceTo(prevPoint);
866                 cosVal = (dist1 * dist1 + dist2 * dist2 - dist3 * dist3) / (2 * dist1 * dist2);
867                 if (cosVal > 1) {
868                     cosVal = 1;
869                 }
870                 if (cosVal < -1) {
871                     cosVal = -1;
872                 }
873                 
874                 angle1 = Math.acos(cosVal);
875                 
876                 dist1 = point.getDistanceTo(nextPoint);
877                 dist2 = snappedPoint.getDistanceTo(nextPoint);
878                 cosVal = (dist1 * dist1 + dist2 * dist2 - dist3 * dist3) / (2 * dist1 * dist2);
879                 if (cosVal > 1) {
880                     cosVal = 1;
881                 }
882                 if (cosVal < -1) {
883                     cosVal = -1;
884                 }
885                 
886                 angle2 = Math.acos(cosVal);
887                 
888                 angle = angle1 + angle2;
889             }
890             
891             /* minimum angle difference (the sum of the two) wins */
892             if (angle < minAngle && angle < snapAngleThreshold) {
893                 minAngle = angle;
894                 bestSolution = solution;
895             }
896         }
897         
898         /* if there is a match coming from an intersection of the lines, return it right away */
899         
900         var xValue, yValue, preferredPoint = null;
901         if (bestSolution != null) {
902             snapVisualFeedback = null;
903             
904             if (bestSolution.m != null) { /* line */
905                 if (bestSolution.m === 0) {
906                     xValue = null;
907                     yValue = bestSolution.n;
908                 }
909                 else if (bestSolution.m === Infinity) {
910                     xValue = bestSolution.x0;
911                     yValue = null;
912                 }
913                 else {
914                     /** @ignore */
915                     xValue = function (y) { return (y - bestSolution.n) / bestSolution.m; };
916                     /** @ignore */
917                     yValue = function (x) { return bestSolution.m * x + bestSolution.n; };
918                     m = bestSolution.m;
919                     n = bestSolution.n;
920                     preferredPoint = new Webgram.Geometry.Point(
921                             (point.y * m + point.x - m * n) / (m * m + 1),
922                             (m * m * point.y + m * point.x + n) / (m * m + 1));
923                 }
924             }
925             else { /* point */
926                 xValue = bestSolution.x;
927                 yValue = bestSolution.y;
928                 bestPoint = new Webgram.Geometry.Point(xValue, yValue);
929                 
930                 snapVisualFeedback = [];
931                 snapVisualFeedback.push({
932                     type: 'radial',
933                     x: prevPoint.x,
934                     y: prevPoint.y,
935                     angle: prevPoint.getAngleTo(bestPoint)
936                 });
937                 snapVisualFeedback.push({
938                     type: 'radial',
939                     x: nextPoint.x,
940                     y: nextPoint.y,
941                     angle: nextPoint.getAngleTo(bestPoint)
942                 });
943             }
944             
945             return {
946                 x: xValue,
947                 y: yValue,
948                 preferredPoint: preferredPoint,
949                 snapVisualFeedback: snapVisualFeedback
950             };
951         }
952         
953         /* if no intersection matched, compute the solutions formed by lines at given angles */
954         /* the best match among these solutions is the one that yields the smallest angle difference */
955         minAngle = null;
956         var minPoint = null;
957         var minAngleDelta = Infinity;
958         var angleDelta;
959         var fullAngleSet = [];
960         
961         for (i = 0; i < prevAngleSet.length; i++) {
962             fullAngleSet.push({
963                 angle: prevAngleSet[i],
964                 point: prevPoint
965             });
966 
967             /* since the angle set contains angles from 0 to PI,
968              * we must manually add the rest of the angles from PI to 2*PI */
969             fullAngleSet.push({
970                 angle: prevAngleSet[i] + Math.PI,
971                 point: prevPoint
972             });
973         }
974         
975         for (i = 0; i < nextAngleSet.length; i++) {
976             fullAngleSet.push({
977                 angle: nextAngleSet[i],
978                 point: nextPoint
979             });
980 
981             /* since the angle set contains angles from 0 to PI,
982              * we must manually add the rest of the angles from PI to 2*PI */
983             fullAngleSet.push({
984                 angle: nextAngleSet[i] + Math.PI,
985                 point: nextPoint
986             });
987         }
988         
989         for (i = 0; i < fullAngleSet.length; i++) {
990             angle = fullAngleSet[i].angle;
991             
992             angle1 = Webgram.Utils.normalizeAngle(fullAngleSet[i].point.getAngleTo(point));
993             angleDelta = Math.abs(angle - angle1);
994             if (angleDelta > Math.PI) {
995                 angleDelta = 2 * Math.PI - angleDelta;
996             }
997             if (angleDelta < minAngleDelta && angleDelta < snapAngleThreshold) {
998                 minAngleDelta = angleDelta;
999                 minAngle = angle;
1000                 minPoint = fullAngleSet[i].point;
1001             }
1002         }
1003         
1004         /* if we have a solution based on lines at given angles,
1005          * we return it among with a radial display snapping info */
1006         if (minAngle != null) {
1007             minAngle = Webgram.Utils.normalizeAngle(minAngle); /* the angle is now between 0 and 2*PI */
1008             preferredPoint = null;
1009             if (Math.abs(minAngle - Math.PI / 2) < 0.000001 || Math.abs(minAngle - 3 * Math.PI / 2) < 0.000001) { /* the angle is odd multiple of PI/2 */
1010                 xValue = minPoint.x;
1011                 yValue = null;
1012             }
1013             else if (Math.abs(minAngle) < 0.000001 || Math.abs(minAngle - Math.PI) < 0.000001) { /* the angle is even multiple of PI/2 */
1014                 xValue = null;
1015                 yValue = minPoint.y;
1016             }
1017             else {
1018                 m = Math.tan(minAngle);
1019                 n = minPoint.y - m * minPoint.x;
1020                 xValue = function (y) { return (y - n) / m; };
1021                 yValue = function (x) { return m * x + n; };
1022                 preferredPoint = new Webgram.Geometry.Point(
1023                         (point.y * m + point.x - m * n) / (m * m + 1),
1024                         (m * m * point.y + m * point.x + n) / (m * m + 1));
1025             }
1026 
1027             return {
1028                 x: xValue,
1029                 y: yValue,
1030                 preferredPoint: preferredPoint,
1031                 snapVisualFeedback: {
1032                     type: 'radial',
1033                     x: minPoint.x,
1034                     y: minPoint.y,
1035                     angle: minAngle
1036                 }
1037             };
1038         }
1039     },

1041     _addPoint: function (index, point, triggerEvents) {
1042         if (index < 0 || index > this.getShape().points.length) {
1043             return false;
1044         }
1045         
1046         var controlPoint;
1047         
1048         /* update poly point indexes */
1049         var polyControlPoints = this.getPolyControlPoints();
1050         for (var i = 0; i < polyControlPoints.length; i++) {
1051             controlPoint = polyControlPoints[i];
1052             
1053             if (controlPoint.polyPointIndex >= index) {
1054                 controlPoint.polyPointIndex += 1;
1055             }
1056         }
1057         
1058         /* effectively add the point */
1059         var points = this.getShape().points;
1060         points.splice(index, 0, point);
1061         this.setShape(new Webgram.Geometry.Poly(points));
1062         
1063         var ControlPointClass = this.getPolyControlPointClass(index);
1064         if (ControlPointClass != null) {
1065             this.addControlPoint(new ControlPointClass(index));
1066         }
1067             
1068         this._recreatePolyControlPoints();
1069         this.snap(index);
1070         
1071         if (triggerEvents) {
1072             this.onAddPoint.trigger(index, point);
1073         }
1074         
1075         return true;
1076     },

1078     _remPoint: function (index, triggerEvents) {
1079         if (index < 0 || index >= this.getShape().points.length) {
1080             return false;
1081         }
1082         
1083         if (this.getShape().points.length <= 2) { /* we need at least two points in a polyElement */
1084             return false;
1085         }
1086         
1087         var controlPoint, i;
1088         
1089         /* remove the given point & associated control point */
1090         var polyControlPoints = this.getPolyControlPoints();
1091         for (i = 0; i < polyControlPoints.length; i++) {
1092             controlPoint = polyControlPoints[i];
1093             
1094             if (controlPoint.polyPointIndex === index) {
1095                 this.remControlPoint(controlPoint);
1096             }
1097         }
1098         
1099         /* update the polyPointIndices */
1100         polyControlPoints = this.getPolyControlPoints();
1101         for (i = 0; i < polyControlPoints.length; i++) {
1102             controlPoint = polyControlPoints[i];
1103             
1104             if (controlPoint.polyPointIndex > index) {
1105                 controlPoint.polyPointIndex -= 1;
1106             }
1107         }
1108         
1109         /* remove the shape point */
1110         var point = this.getShapePoint(index);
1111         var points = this.getShape().points;
1112         points.splice(index, 1);
1113         this.setShape(new Webgram.Geometry.Poly(points));
1114         
1115         this._recreatePolyControlPoints();
1116         
1117         if (triggerEvents) {
1118             this.onRemovePoint.trigger(index, point);
1119         }
1120         
1121         return true;
1122     },

1127             this.invalidateDrawing();
1128             
1129             return;

1131

1133

1135     },

1138         if (!this._hoveredPoint) {
1139             return;
1140         }
1141         
1142         var hoveredPointIndex = this._getHoveredPointIndex();
1143         if (hoveredPointIndex === -1) { /* not found, for some reason */
1144             return;
1145         }
1146         
1147         var point = this.translateDirect(this._hoveredPoint);    
1148         
1149         this.addPoint(hoveredPointIndex, point);
1150         this.setSelectedPointIndex(hoveredPointIndex);
1151

1160             this.invalidateDrawing();
1161         }

1164     _getHoveredPointIndex: function () {
1165         var point = this.transformDirect(this._hoveredPoint);
1166         
1167         /* determine the segment that the point hovers */
1168         var hoveredPointIndex = -1;
1169         var points = this.getShape(true).points;
1170         for (var i = 0; i < points.length; i++) {
1171             var point1 = points[i];
1172             var point2;
1173             if (i < points.length - 1) {
1174                 point2 = points[i + 1];
1175             }
1176             else {
1177                 point2 = points[0];
1178             }
1179             
1180             if (Webgram.Utils.pointInSegment(point, point1, point2, 3)) { /* a line tolerance of 3 should suffice */
1181                 
1182                 hoveredPointIndex = i + 1;
1183                 break;
1184             }
1185         }
1186         
1187         return hoveredPointIndex;
1188     },

1190     _computeHoveredPoint: function (mousePoint) {
1191         /* determine the segment that the point hovers */
1192         var points = this.getShape(true).points;
1193         for (var i = 0; i < points.length; i++) {
1194             if (!this.closed && i === points.length - 1) {
1195                 break;
1196             }
1197             
1198             var point1 = points[i];
1199             var point2;
1200             if (i < points.length - 1) {
1201                 point2 = points[i + 1];
1202             }
1203             else {
1204                 point2 = points[0];
1205             }
1206             
1207             if (Webgram.Utils.pointInSegment(mousePoint, point1, point2, this.getLineTolerance())) {
1208                 var x, y;
1209                 if (point2.y === point1.y) {
1210                     y = point1.y;
1211                     x = mousePoint.x;
1212                 }
1213                 else if (point2.x === point1.x) {
1214                     x = point1.x;
1215                     y = mousePoint.y;
1216                 }
1217                 else {
1218                     var m = (point2.y - point1.y) / (point2.x - point1.x);
1219                     x = (mousePoint.y - point1.y + mousePoint.x / m + point1.x * m) / (m + 1 / m);
1220                     y = point1.y + m * (x - point1.x);
1221                 }
1222                 
1223                 var hoveredPoint = new Webgram.Geometry.Point(x, y);
1224                 
1225                 /* don't mark the hovered point if over an existing CP */
1226                 var minDistThresh = this._controlPoints[0].radius * 1.5 / this.getZoomFactor();
1227                 for (i = 0; i < points.length; i++) {
1228                     if (hoveredPoint.getDistanceTo(points[i]) < minDistThresh) {
1229                         return null;
1230                     }
1231                 }
1232                 
1233                 return this.transformInverse(hoveredPoint);
1234             }
1235         }
1236         
1237         return null;
1238     },
1239     
1240     _recreatePolyControlPoints: function () {
1241         var i, controlPoints = this.getPolyControlPoints();
1242         var controlPointsByIndex = [];
1243         
1244         /* gather the existing poly control points */
1245         for (i = 0; i < controlPoints.length; i++) {
1246             controlPointsByIndex[controlPoints[i].polyPointIndex] = controlPoints[i];
1247         }
1248 
1249         /* add the new points */
1250         for (i = 0; i < this.getShape().points.length; i++) {
1251             var ControlPointClass = this.getPolyControlPointClass(i);
1252             var oldControlPoint = controlPointsByIndex[i];
1253             var oldControlPointClass = oldControlPoint && oldControlPoint.constructor;
1254             
1255             if (ControlPointClass != oldControlPointClass) { /* something has changed */
1256                 if (oldControlPoint != null) {
1257                     this.remControlPoint(oldControlPoint);
1258                 }
1259                 
1260                 if (ControlPointClass != null) {
1261                     this.addControlPoint(new ControlPointClass(i));
1262                 }
1263             }
1264         }
1265     }
