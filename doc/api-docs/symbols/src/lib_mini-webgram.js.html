 21 /**
 22  * @class A mini-webgram is a small version of a normal {@link Webgram} instance,
 23  * used as an overview of the whole drawing, enabling the user to pan and zoom
 24  * to certain areas that are not displayed by the main webgram. A webgram is paired
 25  * with a mini-webgram using {@link Webgram#setMiniWebgram}.
 26  * @extends Webgram
 27  * @param {HtmlElement} htmlElement the <em>HTML</em> element to be used to receive the mouse and keyboard events;
 28  *  this is normally the <em>HTML Canvas</em> element whose 2D context is used to create the canvas object passed as the second argument
 29  * @param {Webgram.Canvas} canvas the canvas to be used
 30  *  by this instance of mini-webgram to paint everything.
 31  */
 32 Webgram.MiniWebgram = function (htmlElement, canvas) {
 33     Webgram.call(this, htmlElement, canvas);
 34 
 35     /**
 36      * A reference to the main webgram that this mini-webgram was paired with.
 37      * @type Webgram
 38      */
 39     this.bigWebgram = null;
 40     
 41     /**
 42      * The amount of space to leave for the borders, expressed as a percent.
 43      * @type Number
 44      */
 45     this.marginFactor = 0.1;
 46     
 47     this.settings.mainGrid = null;
 48     this.settings.snapGrid = null;
 49     this.settings.snapAngle = null;
 50     this.settings.snapDistance = null;
 51     this.settings.showRulers = false;
 52     this.settings.showZoom = false;
 53     this.settings.panEnabled = false;
 54     this.settings.zoomEnabled = false;
 55     
 56     var miniWebgram = this;
 57     
 58     this.rootContainer._fillStyle = Webgram.Styles.getFillStyle('mini-background');
 59     this.rootContainer._strokeStyle = Webgram.Styles.getStrokeStyle('mini-background');
 60     
 61     this.rootContainer.drawBackground = function () {
 62         if (this._fillStyle != null) {
 63             miniWebgram.bigWebgram._drawMini();
 64             
 65             var miniRect = this.webgram.getVisibleArea().getShrinked(-1);
 66             var visibleRect = this.webgram.visibleArea.getBaseRectangle();
 67           
 68             var points = [
 69                 new Webgram.Geometry.Point(visibleRect.x1, miniRect.y2),
 70                 new Webgram.Geometry.Point(visibleRect.x1, visibleRect.y1),
 71                 new Webgram.Geometry.Point(visibleRect.x2, visibleRect.y1),
 72                 new Webgram.Geometry.Point(visibleRect.x2, visibleRect.y2),
 73                 new Webgram.Geometry.Point(visibleRect.x1, visibleRect.y2),
 74           
 75                 new Webgram.Geometry.Point(visibleRect.x1, miniRect.y2),
 76                 new Webgram.Geometry.Point(miniRect.x2, miniRect.y2),
 77                 new Webgram.Geometry.Point(miniRect.x2, miniRect.y1),
 78                 new Webgram.Geometry.Point(miniRect.x1, miniRect.y1),
 79                 new Webgram.Geometry.Point(miniRect.x1, miniRect.y2),
 80               
 81                 new Webgram.Geometry.Point(visibleRect.x1, miniRect.y2)
 82             ];
 83           
 84             this.drawPoly(new Webgram.Geometry.Poly(points), true);
 85             this.paint(null, undefined);
 86             
 87             this.drawRect(visibleRect);
 88             this.paint(undefined, null);
 89         }
 90     };
 91     
 92     this.rootContainer.drawGrid = function () {
 93         /* avoid drawing the grid */
 94     };
 95     
 96     this.onMouseDown.bind(function (point, button, modifiers) {
 97         if (button === 1) {
 98             point = this.rootContainer.transformZoomOffsetInverse(point);
 99             
100             if (!this.visibleArea.pointInside(point)) {
101                 /* center on point */
102                 this.visibleArea.moveTo(point);
103                 return true;
104             }
105         }
106     });
107 };
108 
109 Webgram.MiniWebgram.prototype = {
110     setVisibleArea: function (topLeft, bottomRight) {
111         this._updatingMini = true;
112         this.visibleArea._setTopLeft(topLeft, true);
113         this.visibleArea._setBottomRight(bottomRight, true);
114         
115         delete this._updatingMini;
116     },
117     
118     setWidth: function (width) {
119         /* we override this method to restore the visible center to (0, 0)*/
120         Webgram.prototype.setWidth.call(this, width);
121         
122         this.setVisibleCenter(new Webgram.Geometry.Point(0, 0));
123         this.bigWebgram._updateMini();
124     },
125     
126     setHeight: function (height) {
127         /* we override this method to restore the visible center to (0, 0)*/
128         Webgram.prototype.setHeight.call(this, height);
129         
130         this.setVisibleCenter(new Webgram.Geometry.Point(0, 0));
131         this.bigWebgram._updateMini();
132     },
133     
134     handleMouseDown: function (point, button, modifiers) {
135         if (button === 3) { /* panning with mouse right click is not allowed */
136             return;
137         }
138         
139         return Webgram.prototype.handleMouseDown.call(this, point, button, modifiers);
140     },
141     
142     handleMouseScroll: function (point, up, modifiers) {
143         var level = this.bigWebgram.getZoomLevel();
144         var center = new Webgram.Geometry.Point(this.bigWebgram.getWidth() / 2, this.bigWebgram.getHeight() / 2);
145         
146         if (up) {
147             this.bigWebgram.setZoomLevel(level - 1, center);
148         }
149         else {
150             this.bigWebgram.setZoomLevel(level + 1, center);
151         }
152         
153         return true;
154     },
155     
156     _setRedrawLoop: function () {
157         /* no redraw loop, redrawing handled by the big webgram */
158     },
159     
160     _initDrawingElements: function () {
161         this.visibleArea = new Webgram.MiniWebgram._MiniVisibleArea();
162         this.addDrawingElement(this.visibleArea);
163     },
164 };
165 
166 Webgram.Class('Webgram.MiniWebgram', Webgram);
167 
168 
169 Webgram.MiniWebgram._MiniVisibleArea = function () {
170     Webgram.DrawingElements.RectangularElement.call(this, ':special:mini-visible-area', -20, -20, 41, 41);
171     
172     this.setRotateEnabled(false);
173     this.setResizeEnabled(false);
174     
175     this.onMove.bind(this._updateBigWebgram);
176     this.onResize.bind(this._updateBigWebgram);
177 };
178 
179 Webgram.MiniWebgram._MiniVisibleArea.prototype = {
180     draw: function () {
181     },
182     
183     drawGuides: function () {
184     },
185     
186     drawDecoration: function () {
187     },
188     
189     getDrawingRectangle: function () {
190         var width = this.getDrawingWidth();
191         var height = this.getDrawingHeight();
192         
193         var x1 = Math.round(-width / 2);
194         var y1 = Math.round(-height / 2);
195         var x2 = Math.round(width / 2);
196         var y2 = Math.round(height / 2);
197         
198         return new Webgram.Geometry.Rectangle(x1, y1, x2, y2);
199     },
200 
201     snap: function () {
202     },
203     
204     _updateBigWebgram: function () {
205         if (this.webgram._updatingMini) { /* avoid recurrence */
206             return;
207         }
208         
209         var bigWebgram = this.webgram.bigWebgram;
210         
211         bigWebgram._updateFromMini();
212     }
213 };
214 
215 Webgram.Class('Webgram.MiniWebgram._MiniVisibleArea', Webgram.DrawingElements.RectangularElement);
