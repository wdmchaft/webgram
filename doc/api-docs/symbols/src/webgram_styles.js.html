 21 /**
 22  * @namespace Contains style classes and related functions
 23  * that determine how <em>stroke</em>, <em>fill</em> and <em>text</em> are painted on the canvas.
 24  * This namespace behaves like a factory/store for these styles.
 25  */
 26 Webgram.Styles = {
 27     _strokeStyles: {},
 28     _fillStyles: {},
 29     _textStyles: {},
 30 
 31     /**
 32      * Adds a stroke style to the store. If a style with the given name
 33      * already exists, it is replaced by the new style.
 34      * @param {String} name the name of the style
 35      * @param {Webgram.Styles.StrokeStyle} strokeStyle the stroke style to set
 36      */
 37     setStrokeStyle: function (name, strokeStyle) {
 38         if (!(strokeStyle instanceof Webgram.Styles.StrokeStyle) && strokeStyle != null) {
 39             strokeStyle = this.createStrokeStyle(strokeStyle);
 40         }
 41         
 42         this._strokeStyles[name] = strokeStyle;
 43     },
 44     
 45     /**
 46      * Adds a fill style to the store. If a style with the given name
 47      * already exists, it is replaced by the new style.
 48      * @param {String} name the name of the style
 49      * @param {Webgram.Styles.FillStyle} fillStyle the fill style to set
 50      */
 51     setFillStyle: function (name, fillStyle) {
 52         if (!(fillStyle instanceof Webgram.Styles.FillStyle) && fillStyle != null) {
 53             fillStyle = this.createFillStyle(fillStyle);
 54         }
 55         
 56         this._fillStyles[name] = fillStyle;
 57     },
 58     
 59     /**
 60      * Adds a text style to the store. If a style with the given name
 61      * already exists, it is replaced by the new style.
 62      * @param {String} name the name of the style
 63      * @param {Webgram.Styles.TextStyle} textStyle the text style to set
 64      */
 65     setTextStyle: function (name, textStyle) {
 66         if (!(textStyle instanceof Webgram.Styles.TextStyle) && textStyle != null) {
 67             textStyle = this.createTextStyle(textStyle);
 68         }
 69         
 70         this._textStyles[name] = textStyle;
 71     },
 72     
 73     /**
 74      * Retrieves the stroke style with the given name
 75      * from the store.
 76      * @param {String} name the name of the style
 77      * @returns {Webgram.Styles.StrokeStyle} the stroke style having the given name
 78      *  or <tt>null</tt> if no style with this name is found
 79      */
 80     getStrokeStyle: function (name) {
 81         var style = this._strokeStyles[name];
 82         if (style == null) {
 83             return null;
 84         }
 85             
 86         return style.clone();
 87     },
 88     
 89     /**
 90      * Retrieves the fill style with the given name
 91      * from the store.
 92      * @param {String} name the name of the style
 93      * @returns {Webgram.Styles.FillStyle} the fill style having the given name
 94      *  or <tt>null</tt> if no style with this name is found
 95      */
 96     getFillStyle: function (name) {
 97         var style = this._fillStyles[name];
 98         if (style == null) {
 99             return null;
100         }
101         
102         return style.clone();
103     },
104     
105     /**
106      * Retrieves the text style with the given name
107      * from the store.
108      * @param {String} name the name of the style
109      * @returns {Webgram.Styles.TextStyle} the text style having the given name
110      *  or <tt>null</tt> if no style with this name is found
111      */
112     getTextStyle: function (name) {
113         var style = this._textStyles[name];
114         if (style == null) {
115             return null;
116         }
117         
118         return style.clone();
119     },
120     
121     /**
122      * Creates a stroke style from attributes. This method does not
123      * add the newly created style to the store.
124      * For a full descriptions of these attributes,
125      * see {@link Webgram.Styles.StrokeStyle}.
126      * @param {Object} attributes an object with:<ul>
127      *  <li><tt>lineWidth</tt></li>
128      *  <li><tt>lineCap</tt></li>
129      *  <li><tt>lineJoin</tt></li>
130      *  <li><tt>miterLimit</tt></li>
131      *  <li><tt>colors</tt></li>
132      *  <li><tt>pattern</tt></li>
133      *  <li><tt>gradientPoint1</tt></li>
134      *  <li><tt>gradientPoint2</tt></li>
135      *  <li><tt>gradientRadius1</tt></li>
136      *  <li><tt>gradientRadius2</tt></li>
137      * </ul>
138      * @returns {Webgram.Styles.StrokeStyle} the newly created style
139      */
140     createStrokeStyle: function (attributes) {
141         var gradientPoint1 = attributes.gradientPoint1;
142         var gradientPoint2 = attributes.gradientPoint2;
143         
144         /* make sure the gradient points are instances of Point */
145         if (gradientPoint1 != null) {
146             gradientPoint1 = new Webgram.Geometry.Point(gradientPoint1.x, gradientPoint1.y);
147         }
148         if (gradientPoint2 != null) {
149             gradientPoint2 = new Webgram.Geometry.Point(gradientPoint2.x, gradientPoint2.y);
150         }
151                 
152         return new Webgram.Styles.StrokeStyle(
153                 attributes.lineWidth,
154                 attributes.lineCap,
155                 attributes.lineJoin,
156                 attributes.miterLimit,
157                 attributes.colors,
158                 attributes.pattern,
159                 gradientPoint1,
160                 gradientPoint2,
161                 attributes.gradientRadius1,
162                 attributes.gradientRadius2);
163     },
164     
165     /**
166      * Creates a fill style from attributes. This method does not
167      * add the newly created style to the store.
168      * For a full descriptions of these attributes,
169      * see {@link Webgram.Styles.FillStyle}.
170      * @param {Object} attributes an object with:<ul>
171      *  <li><tt>colors</tt></li>
172      *  <li><tt>gradientPoint1</tt></li>
173      *  <li><tt>gradientPoint2</tt></li>
174      *  <li><tt>gradientRadius1</tt></li>
175      *  <li><tt>gradientRadius2</tt></li>
176      * </ul>
177      * @returns {Webgram.Styles.FillStyle} the newly created style
178      */
179     createFillStyle: function (attributes) {
180         var gradientPoint1 = attributes.gradientPoint1;
181         var gradientPoint2 = attributes.gradientPoint2;
182         
183         /* make sure the gradient points are instances of Point */
184         if (gradientPoint1 != null) {
185             gradientPoint1 = new Webgram.Geometry.Point(gradientPoint1.x, gradientPoint1.y);
186         }
187         if (gradientPoint2 != null) {
188             gradientPoint2 = new Webgram.Geometry.Point(gradientPoint2.x, gradientPoint2.y);
189         }
190                 
191         return new Webgram.Styles.FillStyle(
192                 attributes.colors,
193                 gradientPoint1,
194                 gradientPoint2,
195                 attributes.gradientRadius1,
196                 attributes.gradientRadius2);
197     },
198 
199     /**
200      * Creates a text style from attributes. This method does not
201      * add the newly created style to the store.
202      * For a full descriptions of these attributes,
203      * see {@link Webgram.Styles.TextStyle}.
204      * @param {Object} attributes an object with:<ul>
205      *  <li><tt>color</tt></li>
206      *  <li><tt>font</tt></li>
207      *  <li><tt>size</tt></li>
208      *  <li><tt>bold</tt></li>
209      *  <li><tt>italic</tt></li>
210      *  <li><tt>justify</tt></li>
211      * </ul>
212      * @returns {Webgram.Styles.TextStyle} the newly created style
213      */
214     createTextStyle: function (attributes) {
215         return new Webgram.Styles.TextStyle(
216                 attributes.color,
217                 attributes.font,
218                 attributes.size,
219                 attributes.bold,
220                 attributes.italic,
221                 attributes.justify);
222     }

228 /**
229  * @class Represents a style used to draw the lines on the canvas.
230  * Any of the following arguments can be omitted (or passed as <tt>null</tt>),
231  * in which case the default values will be used for that attribute.
232  * @param {Number} lineWidth the width (thickness) of the line
233  * @param {String} lineCap one of <tt>'butt'</tt>, <tt>'round'</tt> or <tt>'square'</tt>
234  * @param {String} lineJoin one of <tt>'round'</tt>, <tt>'bevel'</tt> or <tt>'miter'</tt>
235  * @param {Number} miterLimit the miter limit, when <tt>lineJoin</tt> is set to <tt>'miter'</tt>
236  * @param {Array} colors a list of colors to use; if a single color is wanted, it can be passed directly (i.e. not enclosed in an array)
237  * @param {Array} pattern a list of two numbers that form the pattern;
238  * the first is the "pencil on" length, the second is the "pencil off" length;
239  * if omitted, no stroke pattern will be used
240  * @param {Webgram.Geometry.Point} gradientPoint1 the start gradient point;
241  * no gradient will be used if this is omitted
242  * @param {Webgram.Geometry.Point} gradientPoint2 the end gradient point;
243  * if not <tt>null</tt> a <em>linear</em> gradient will be used,
244  * otherwise, if <tt>gradientPoint1</tt> is not <tt>null</tt>, a <em>radial</em>
245  * gradient will be used
246  * @param {Number} gradientRadius1 the start radius of the radial gradient
247  * @param {Number} gradientRadius2 the end radius of the radial gradient
248  */
249 Webgram.Styles.StrokeStyle = function (lineWidth, lineCap, lineJoin, miterLimit, 
250         colors, pattern, gradientPoint1, gradientPoint2, gradientRadius1, gradientRadius2) {
251     
252     if (lineWidth == null) {
253         lineWidth = 1;
254     }
255     if (lineCap == null) {
256         lineCap = 'square';
257     }
258     if (lineJoin == null) {
259         lineJoin = 'miter';
260     }
261     if (miterLimit == null) {
262         miterLimit = 10;
263     }
264     if (colors == null) {
265         colors = 'black';
266     }
267     
268     /**
269      * The width (thickness) of the line.
270      * @type Number
271      */
272     this.lineWidth = lineWidth;
273     
274     /**
275      * One of <tt>'butt'</tt>, <tt>'round'</tt> or <tt>'square'</tt>.
276      * @type String
277      */
278     this.lineCap = lineCap;
279     
280     /**
281      * One of <tt>'round'</tt>, <tt>'bevel'</tt> or <tt>'miter'</tt>.
282      * @type String
283      */
284     this.lineJoin = lineJoin;
285     
286     /**
287      * The miter limit, when <tt>lineJoin</tt> is set to <tt>'miter'</tt>.
288      * @type Number
289      */
290     this.miterLimit = miterLimit;
291     
292     /**
293      * A list of colors to use. If a single color is used, this field
294      * is set directly to that color instead of being wrapped inside an array.
295      * @type Array 
296      */
297     this.colors = colors;
298     
299     /**
300      * A list of two numbers that form the pattern.
301      * The first is the "pencil on" length, the second is the "pencil off" length.
302      * This is set to <tt>null</tt> if no pattern is used.
303      * @type Array
304      */
305     this.pattern = pattern;
306     
307     /**
308      * The start gradient point. This is set to <tt>null</tt> if no gradient is used.
309      * @type Webgram.Geometry.Point
310      */
311     this.gradientPoint1 = gradientPoint1;
312     
313     /**
314      * The end gradient point. This is set to a non-<tt>null</tt> point, if a <em>linear</em> gradient is used.
315      * @type Webgram.Geometry.Point
316      */
317     this.gradientPoint2 = gradientPoint2;
318     
319     /**
320      * The start radius of the radial gradient. This is set to <tt>null</tt> if a <em>linear</em> gradient
321      * or no gradient is used.
322      * @type Number
323      */
324     this.gradientRadius1 = gradientRadius1;
325     
326     /**
327      * The end radius of the radial gradient. This is set to <tt>null</tt> if a <em>linear</em> gradient
328      * or no gradient is used.
329      * @type Number
330      */
331     this.gradientRadius2 = gradientRadius2;
332 };

334 Webgram.Styles.StrokeStyle.prototype = {
335     /**
336      * Creates an identical copy of this stroke style.
337      * @returns {Webgram.Styles.StrokeStyle} the clone of this stroke style
338      */
339     clone: function () {
340         return new Webgram.Styles.StrokeStyle(
341                 this.lineWidth,
342                 this.lineCap,
343                 this.lineJoin,
344                 this.miterLimit,
345                 Webgram.Utils.clone(this.colors),
346                 Webgram.Utils.clone(this.pattern),
347                 Webgram.Utils.clone(this.gradientPoint1),
348                 Webgram.Utils.clone(this.gradientPoint2),
349                 this.gradientRadius1,
350                 this.gradientRadius2);
351     },
352     
353     /**
354      * Creates a copy of this stroke style by selectively modifying some attributes.
355      * This method does not affect the original style.
356      * @param {Object} attributes the attributes to change;
357      * for more details on these attributes, see {@link Webgram.Styles.StrokeStyle}
358      * @returns {Webgram.Styles.StrokeStyle} the modified stroke style
359      */
360     replace: function (attributes) {
361         if (attributes.lineWidth === undefined) {
362             attributes.lineWidth = this.lineWidth;
363         }
364         if (attributes.lineCap === undefined) {
365             attributes.lineCap = this.lineCap;
366         }
367         if (attributes.lineJoin === undefined) {
368             attributes.lineJoin = this.lineJoin;
369         }
370         if (attributes.miterLimit === undefined) {
371             attributes.miterLimit = this.miterLimit;
372         }
373         if (attributes.colors === undefined) {
374             attributes.colors = Webgram.Utils.clone(this.colors);
375         }
376         if (attributes.pattern === undefined) {
377             attributes.pattern = Webgram.Utils.clone(this.pattern);
378         }
379         if (attributes.gradientPoint1 === undefined) {
380             attributes.gradientPoint1 = Webgram.Utils.clone(this.gradientPoint1);
381         }
382         if (attributes.gradientPoint2 === undefined) {
383             attributes.gradientPoint2 = Webgram.Utils.clone(this.gradientPoint2);
384         }
385         if (attributes.gradientRadius1 === undefined) {
386             attributes.gradientRadius1 = this.gradientRadius1;
387         }
388         if (attributes.gradientRadius2 === undefined) {
389             attributes.gradientRadius2 = this.gradientRadius2;
390         }
391         
392         return Webgram.Styles.createStrokeStyle(attributes);
393     },
394     
395     /**
396      * Generates a json object that completely characterizes this style.
397      * The result should be enough to restore the style using
398      * {@link Webgram.Styles.StrokeStyle#fromJson}.
399      * @returns {Object} the json object
400      */
401     toJson: function () {
402         return {
403             lineWidth: this.lineWidth,
404             lineCap: this.lineCap,
405             lineJoin: this.lineJoin,
406             miterLimit: this.miterLimit,
407             colors: this.colors,
408             pattern: this.pattern,
409             gradientPoint1: (this.gradientPoint1 != null) ? {
410                 x: this.gradientPoint1.x,
411                 y: this.gradientPoint1.y,
412             } : null,
413             gradientPoint2: (this.gradientPoint2 != null) ? {
414                 x: this.gradientPoint2.x,
415                 y: this.gradientPoint2.y,
416             } : null,
417             gradientRadius1: this.gradientRadius1,
418             gradientRadius2: this.gradientRadius2
419         };
420     },
421     
422     /**
423      * Restores the style from a json object. This method is the opposite
424      * of {@link Webgram.Styles.StrokeStyle#toJson}.
425      * @param {Object} json the json object to restore the style from
426      */
427     fromJson: function (json) {
428         if (json.lineWidth !== undefined) {
429             this.lineWidth = json.lineWidth;
430         }
431         
432         if (json.lineCap !== undefined) {
433             this.lineCap = json.lineCap;
434         }
435         
436         if (json.lineJoin !== undefined) {
437             this.lineJoin = json.lineJoin;
438         }
439         
440         if (json.miterLimit !== undefined) {
441             this.miterLimit = json.miterLimit;
442         }
443         
444         if (json.colors !== undefined) {
445             this.colors = Webgram.Utils.clone(json.colors);
446         }
447         
448         if (json.pattern !== undefined) {
449             this.pattern = Webgram.Utils.clone(json.pattern);
450         }
451         
452         if (json.gradientPoint1 !== undefined) {
453             this.gradientPoint1 = json.gradientPoint1;
454             if (this.gradientPoint1 != null) {
455                 this.gradientPoint1 = new Webgram.Geometry.Point(this.gradientPoint1.x, this.gradientPoint1.y);
456             }
457         }
458         
459         if (json.gradientPoint2 !== undefined) {
460             this.gradientPoint2 = json.gradientPoint2;
461             if (this.gradientPoint2 != null) {
462                 this.gradientPoint2 = new Webgram.Geometry.Point(this.gradientPoint2.x, this.gradientPoint2.y);
463             }
464         }
465         
466         if (json.gradientRadius1 !== undefined) {
467             this.gradientRadius1 = json.gradientRadius1;
468         }
469         
470         if (json.gradientRadius2 !== undefined) {
471             this.gradientRadius2 = json.gradientRadius2;
472         }
473     }
474 };

632 /**
633  * @class Represents a style used to draw the text on the canvas.
634  * Any of the following arguments can be omitted (or passed as <tt>null</tt>),
635  * in which case the default values will be used for that attribute.
636  * @param {String} color the color to be used to draw the text
637  * @param {String} font the font name to be used to draw the text
638  * @param {Number} size the size of the text
639  * @param {Boolean} bold whether to draw a bold text or not
640  * @param {Boolean} italic whether to draw the text in italics or not
641  * @param {Array} justify an array of two letters; the first one indicates the horizontal
642  * alignment and is one of <tt>'l'</tt>, <tt>'c'</tt> or <tt>'r'</tt>,
643  * for left, center or right alignment, respectively;
644  * the second one indicates the vertical
645  * alignment and is one of <tt>'t'</tt>, <tt>'c'</tt> or <tt>'b'</tt>,
646  * for top, center or bottom alignment, respectively
647  */
648 Webgram.Styles.TextStyle = function (color, font, size, bold, italic, justify) {
649     /**
650      * The color used to draw the text.
651      * @type String
652      */
653     this.color = color;
654     
655     /**
656      * The font name used to draw the text.
657      * @type String
658      */
659     this.font = font;
660     
661     /**
662      * The size of the text.
